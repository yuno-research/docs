1use std::fmt;
2use std::hash::BuildHasher;
3use std::sync::atomic::{AtomicUsize, Ordering};
4
5use arrow::array::builder::StaticArrayBuilder;
6use arrow::array::{Array, MutableUtf8Array, Utf8Array, Utf8ViewArrayBuilder};
7use arrow::datatypes::ArrowDataType;
8use polars_error::{PolarsResult, polars_bail};
9use polars_utils::aliases::PlSeedableRandomStateQuality;
10use polars_utils::parma::raw::RawTable;
11
12use super::CatSize;
13
14pub struct CategoricalMapping {
15    str_to_cat: RawTable<str, CatSize>,
16    cat_to_str: boxcar::Vec<&'static str>,
17    max_categories: usize,
18    upper_bound: AtomicUsize,
19    hasher: PlSeedableRandomStateQuality,
20}
21
22impl CategoricalMapping {
23    pub fn new(max_categories: usize) -> Self {
24        Self::with_hasher(max_categories, PlSeedableRandomStateQuality::default())
25    }
26
27    pub fn with_hasher(max_categories: usize, hasher: PlSeedableRandomStateQuality) -> Self {
28        Self {
29            str_to_cat: RawTable::default(),
30            cat_to_str: boxcar::Vec::default(),
31            max_categories,
32            upper_bound: AtomicUsize::new(0),
33            hasher,
34        }
35    }
36
37    #[inline(always)]
38    pub fn hasher(&self) -> &PlSeedableRandomStateQuality {
39        &self.hasher
40    }
41
42    pub fn set_max_categories(&mut self, max_categories: usize) {
43        assert!(max_categories >= self.num_cats_upper_bound());
44        self.max_categories = max_categories
45    }
46
47    /// Try to convert a string to a categorical id, but don't insert it if it is missing.
48    #[inline(always)]
49    pub fn get_cat(&self, s: &str) -> Option<CatSize> {
50        let hash = self.hasher.hash_one(s);
51        self.get_cat_with_hash(s, hash)
52    }
53
54    /// Same as get_cat, but with the hash pre-computed.
55    #[inline(always)]
56    pub fn get_cat_with_hash(&self, s: &str, hash: u64) -> Option<CatSize> {
57        self.str_to_cat.get(hash, |k| k == s).copied()
58    }
59
60    /// Convert a string to a categorical id.
61    #[inline(always)]
62    pub fn insert_cat(&self, s: &str) -> PolarsResult<CatSize> {
63        let hash = self.hasher.hash_one(s);
64        self.insert_cat_with_hash(s, hash)
65    }
66
67    /// Same as to_cat, but with the hash pre-computed.
68    #[inline(always)]
69    pub fn insert_cat_with_hash(&self, s: &str, hash: u64) -> PolarsResult<CatSize> {
70        self.str_to_cat
71            .try_get_or_insert_with(
72                hash,
73                s,
74                |k| k == s,
75                |k| {
76                    let old_upper_bound = self.upper_bound.fetch_add(1, Ordering::Relaxed);
77                    if old_upper_bound + 1 > self.max_categories {
78                        self.upper_bound.fetch_sub(1, Ordering::Relaxed);
79                        polars_bail!(ComputeError: "attempted to insert more categories than the maximum allowed");
80                    }
81                    let idx = self
82                        .cat_to_str
83                        .push(unsafe { core::mem::transmute::<&str, &'static str>(k) });
84                    Ok(idx as CatSize)
85                },
86            )
87            .copied()
88    }
89
90    /// Try to convert a categorical id to its corresponding string, returning
91    /// None if the string is not in the data structure.
92    #[inline(always)]
93    pub fn cat_to_str(&self, cat: CatSize) -> Option<&str> {
94        self.cat_to_str.get(cat as usize).copied()
95    }
96
97    /// Get the string corresponding to a categorical id.
98    ///
99    /// # Safety
100    /// The categorical id must have been returned from `to_cat`, and you must
101    /// have synchronized with the call which inserted it.
102    #[inline(always)]
103    pub unsafe fn cat_to_str_unchecked(&self, cat: CatSize) -> &str {
104        unsafe { self.cat_to_str.get_unchecked(cat as usize) }
105    }
106
107    /// Returns an upper bound such that all strings inserted into the CategoricalMapping
108    /// have a categorical id less than it. Note that due to parallel inserts which
109    /// you have not synchronized with, there may be gaps when using `from_cat`.
110    #[inline(always)]
111    pub fn num_cats_upper_bound(&self) -> usize {
112        // We need to clamp to self.max_categories because a `fetch_add` may
113        // have (temporarily) pushed it beyond the max allowed.
114        self.upper_bound
115            .load(Ordering::Relaxed)
116            .min(self.max_categories)
117    }
118
119    /// Returns the number of categories in this mapping.
120    #[inline(always)]
121    pub fn len(&mut self) -> usize {
122        *self.upper_bound.get_mut()
123    }
124
125    #[inline(always)]
126    pub fn is_empty(&mut self) -> bool {
127        self.len() == 0
128    }
129
130    pub fn to_arrow(&self, as_views: bool) -> Box<dyn Array> {
131        let n = self.num_cats_upper_bound();
132        if as_views {
133            let mut builder = Utf8ViewArrayBuilder::new(ArrowDataType::Utf8View);
134            builder.reserve(n);
135            for i in 0..n {
136                let s = self.cat_to_str(i as CatSize).unwrap_or_default();
137                builder.push_value_ignore_validity(s);
138            }
139            builder.freeze().boxed()
140        } else {
141            let mut builder = MutableUtf8Array::new();
142            builder.reserve(n, 0);
143            for i in 0..n {
144                let s = self.cat_to_str(i as CatSize).unwrap_or_default();
145                builder.push(Some(s));
146            }
147            let arr: Utf8Array<i64> = builder.into();
148            arr.boxed()
149        }
150    }
151}

153impl fmt::Debug for CategoricalMapping {
154    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
155        f.debug_struct("CategoricalMapping")
156            .field("max_categories", &self.max_categories)
157            .field("upper_bound", &self.upper_bound.load(Ordering::Relaxed))
158            .finish()
159    }
