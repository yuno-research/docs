9pub fn new_dummy_x509_certificate(
10    keypair: &Keypair,
11) -> (
12    rustls::pki_types::CertificateDer<'static>,
13    rustls::pki_types::PrivateKeyDer<'static>,
14) {
15    // Unfortunately, rustls does not accept a "raw" Ed25519 key.
16    // We have to convert it to DER and pass it to the library.
17
18    // Convert private key into PKCS#8 v1 object.
19    // RFC 8410, Section 7: Private Key Format
20    // https://www.rfc-editor.org/rfc/rfc8410#section-7
21    //
22    // The hardcoded prefix decodes to the following ASN.1 structure:
23    //
24    //   PrivateKeyInfo SEQUENCE (3 elem)
25    //     version Version INTEGER 0
26    //     privateKeyAlgorithm AlgorithmIdentifier SEQUENCE (1 elem)
27    //       algorithm OBJECT IDENTIFIER 1.3.101.112 curveEd25519 (EdDSA 25519 signature algorithm)
28    //     privateKey PrivateKey OCTET STRING (34 byte)
29    const PKCS8_PREFIX: [u8; 16] = [
30        0x30, 0x2e, 0x02, 0x01, 0x00, 0x30, 0x05, 0x06, 0x03, 0x2b, 0x65, 0x70, 0x04, 0x22, 0x04,
31        0x20,
32    ];
33
34    let key_pkcs8_der = {
35        let keypair_secret_bytes = keypair.secret_bytes();
36        let keypair_secret_len = keypair_secret_bytes.len();
37        if keypair_secret_len != 32 {
38            panic!("Unexpected secret key length!");
39        }
40        let buffer_size = PKCS8_PREFIX
41            .len()
42            .checked_add(keypair_secret_len) //clippy being overly guarded here but optimizer will elide checked_add
43            .expect("Unexpected secret key length!");
44        let mut key_pkcs8_der = Vec::<u8>::with_capacity(buffer_size);
45        key_pkcs8_der.extend_from_slice(&PKCS8_PREFIX);
46        key_pkcs8_der.extend_from_slice(keypair_secret_bytes);
47        key_pkcs8_der
48    };
49
50    // Create a dummy certificate. Only the SubjectPublicKeyInfo field
51    // is relevant to the peer-to-peer protocols. The signature of the
52    // X.509 certificate is deliberately invalid. (Peer authenticity is
53    // checked in the TLS 1.3 CertificateVerify)
54    // See https://www.itu.int/rec/T-REC-X.509-201910-I/en for detailed definitions.
55
56    let mut cert_der = Vec::<u8>::with_capacity(0xf4);
57    //    Certificate SEQUENCE (3 elem)
58    //      tbsCertificate TBSCertificate SEQUENCE (8 elem)
59    //        version [0] (1 elem)
60    //          INTEGER  2
61    //        serialNumber CertificateSerialNumber INTEGER (62 bit)
62    //        signature AlgorithmIdentifier SEQUENCE (1 elem)
63    //          algorithm OBJECT IDENTIFIER 1.3.101.112 curveEd25519 (EdDSA 25519 signature algorithm)
64    //        issuer Name SEQUENCE (1 elem)
65    //          RelativeDistinguishedName SET (1 elem)
66    //            AttributeTypeAndValue SEQUENCE (2 elem)
67    //              type AttributeType OBJECT IDENTIFIER 2.5.4.3 commonName (X.520 DN component)
68    //              value AttributeValue [?] UTF8String Solana
69    //        validity Validity SEQUENCE (2 elem)
70    //          notBefore Time UTCTime 1970-01-01 00:00:00 UTC
71    //          notAfter Time GeneralizedTime 4096-01-01 00:00:00 UTC
72    //        subject Name SEQUENCE (0 elem)
73    //        subjectPublicKeyInfo SubjectPublicKeyInfo SEQUENCE (2 elem)
74    //          algorithm AlgorithmIdentifier SEQUENCE (1 elem)
75    //            algorithm OBJECT IDENTIFIER 1.3.101.112 curveEd25519 (EdDSA 25519 signature algorithm)
76    //          subjectPublicKey BIT STRING (256 bit)
77    cert_der.extend_from_slice(&[
78        0x30, 0x81, 0xf6, 0x30, 0x81, 0xa9, 0xa0, 0x03, 0x02, 0x01, 0x02, 0x02, 0x08, 0x01, 0x01,
79        0x01, 0x01, 0x01, 0x01, 0x01, 0x01, 0x30, 0x05, 0x06, 0x03, 0x2b, 0x65, 0x70, 0x30, 0x16,
80        0x31, 0x14, 0x30, 0x12, 0x06, 0x03, 0x55, 0x04, 0x03, 0x0c, 0x0b, 0x53, 0x6f, 0x6c, 0x61,
81        0x6e, 0x61, 0x20, 0x6e, 0x6f, 0x64, 0x65, 0x30, 0x20, 0x17, 0x0d, 0x37, 0x30, 0x30, 0x31,
82        0x30, 0x31, 0x30, 0x30, 0x30, 0x30, 0x30, 0x30, 0x5a, 0x18, 0x0f, 0x34, 0x30, 0x39, 0x36,
83        0x30, 0x31, 0x30, 0x31, 0x30, 0x30, 0x30, 0x30, 0x30, 0x30, 0x5a, 0x30, 0x00, 0x30, 0x2a,
84        0x30, 0x05, 0x06, 0x03, 0x2b, 0x65, 0x70, 0x03, 0x21, 0x00,
85    ]);
86    cert_der.extend_from_slice(&keypair.pubkey().to_bytes());
87    //        extensions [3] (1 elem)
88    //          Extensions SEQUENCE (2 elem)
89    //            Extension SEQUENCE (3 elem)
90    //              extnID OBJECT IDENTIFIER 2.5.29.17 subjectAltName (X.509 extension)
91    //              critical BOOLEAN true
92    //              extnValue OCTET STRING (13 byte) encapsulating
93    //                SEQUENCE (1 elem)
94    //                [2] (9 byte) localhost
95    //            Extension SEQUENCE (3 elem)
96    //              extnID OBJECT IDENTIFIER 2.5.29.19 basicConstraints (X.509 extension)
97    //              critical BOOLEAN true
98    //              extnValue OCTET STRING (2 byte) encapsulating
99    //                SEQUENCE (0 elem)
100    //      signatureAlgorithm AlgorithmIdentifier SEQUENCE (1 elem)
101    //        algorithm OBJECT IDENTIFIER 1.3.101.112 curveEd25519 (EdDSA 25519 signature algorithm)
102    //        signature BIT STRING (512 bit)
103    cert_der.extend_from_slice(&[
104        0xa3, 0x29, 0x30, 0x27, 0x30, 0x17, 0x06, 0x03, 0x55, 0x1d, 0x11, 0x01, 0x01, 0xff, 0x04,
105        0x0d, 0x30, 0x0b, 0x82, 0x09, 0x6c, 0x6f, 0x63, 0x61, 0x6c, 0x68, 0x6f, 0x73, 0x74, 0x30,
106        0x0c, 0x06, 0x03, 0x55, 0x1d, 0x13, 0x01, 0x01, 0xff, 0x04, 0x02, 0x30, 0x00, 0x30, 0x05,
107        0x06, 0x03, 0x2b, 0x65, 0x70, 0x03, 0x41, 0x00, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff,
108        0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff,
109        0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff,
110        0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff,
111        0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff,
112    ]);
113
114    (
115        rustls::pki_types::CertificateDer::from(cert_der),
116        rustls::pki_types::PrivateKeyDer::try_from(key_pkcs8_der).unwrap(),
117    )
118}

120pub fn get_pubkey_from_tls_certificate(
121    der_cert: &rustls::pki_types::CertificateDer,
122) -> Option<Pubkey> {
123    let (_, cert) = X509Certificate::from_der(der_cert.as_ref()).ok()?;
124    match cert.public_key().parsed().ok()? {
125        PublicKey::Unknown(key) => Pubkey::try_from(key).ok(),
126        _ => None,
127    }
128}

130/// Translate a SocketAddr into a valid SNI for the purposes of QUIC connection
131///
132/// We do not actually check if the server holds a cert for this server_name
133/// since Solana does not rely on DNS names, but we need to provide a unique
134/// one to ensure that we present correct QUIC tokens to the correct server.
135pub fn socket_addr_to_quic_server_name(peer: SocketAddr) -> String {
136    format!("{}.{}.sol", peer.ip(), peer.port())
137}

154    fn test_generate_tls_certificate() {
155        let keypair = Keypair::new();

157        if let Some(pubkey) = get_pubkey_from_tls_certificate(&cert) {
