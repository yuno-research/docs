234/// Spawn a streamer instance in the current tokio runtime.
235pub(crate) fn spawn_server_with_cancel_and_qos<Q, C>(
236    name: &'static str,
237    stats: Arc<StreamerStats>,
238    sockets: impl IntoIterator<Item = UdpSocket>,
239    keypair: &Keypair,
240    packet_sender: Sender<PacketBatch>,
241    quic_server_params: QuicStreamerConfig,
242    qos: Arc<Q>,
243    cancel: CancellationToken,
244) -> Result<SpawnNonBlockingServerResult, QuicServerError>
245where
246    Q: QosController<C> + Send + Sync + 'static,
247    C: ConnectionContext + Send + Sync + 'static,
248{
249    let sockets: Vec<_> = sockets.into_iter().collect();
250    info!("Start {name} quic server on {sockets:?}");
251    let (config, _) = configure_server(keypair)?;
252
253    let endpoints = sockets
254        .into_iter()
255        .map(|sock| {
256            Endpoint::new(
257                EndpointConfig::default(),
258                Some(config.clone()),
259                sock,
260                Arc::new(TokioRuntime),
261            )
262            .map_err(QuicServerError::EndpointFailed)
263        })
264        .collect::<Result<Vec<_>, _>>()?;
265    let (packet_batch_sender, packet_batch_receiver) =
266        bounded(quic_server_params.accumulator_channel_size);
267    task::spawn_blocking({
268        let cancel = cancel.clone();
269        let stats = stats.clone();
270        move || {
271            run_packet_batch_sender(packet_sender, packet_batch_receiver, stats, cancel);
272        }
273    });
274
275    let max_concurrent_connections = quic_server_params.max_concurrent_connections();
276    let handle = tokio::spawn({
277        let endpoints = endpoints.clone();
278        let stats = stats.clone();
279        async move {
280            let tasks = run_server(
281                name,
282                endpoints.clone(),
283                packet_batch_sender,
284                stats.clone(),
285                quic_server_params,
286                cancel,
287                qos,
288            )
289            .await;
290            tasks.close();
291            tasks.wait().await;
292        }
293    });
294
295    Ok(SpawnNonBlockingServerResult {
296        endpoints,
297        stats,
298        thread: handle,
299        max_concurrent_connections,
300    })
301}

303/// struct ease tracking connections of all stages, so that we do not have to

306/// reduced when it is dropped.
307pub struct ClientConnectionTracker {

312impl fmt::Debug for ClientConnectionTracker {
313    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
314        f.debug_struct("StreamerClientConnection")

323impl Drop for ClientConnectionTracker {
324    /// When this is dropped, reduce the open connection count.
325    fn drop(&mut self) {
326        self.stats.open_connections.fetch_sub(1, Ordering::Relaxed);
327    }

330impl ClientConnectionTracker {
331    /// Check the max_concurrent_connections limit and if it is within the limit
332    /// create ClientConnectionTracker and increment open connection count. Otherwise returns Err
333    fn new(stats: Arc<StreamerStats>, max_concurrent_connections: usize) -> Result<Self, ()> {
334        let open_connections = stats.open_connections.fetch_add(1, Ordering::Relaxed);
335        if open_connections >= max_concurrent_connections {
336            stats.open_connections.fetch_sub(1, Ordering::Relaxed);
337            debug!(
338                "There are too many concurrent connections opened already: open: \
339                 {open_connections}, max: {max_concurrent_connections}"
340            );
341            return Err(());
342        }
343
344        Ok(Self { stats })
345    }
346}

348#[allow(clippy::too_many_arguments)]
349async fn run_server<Q, C>(
350    name: &'static str,
351    endpoints: Vec<Endpoint>,
352    packet_batch_sender: Sender<PacketAccumulator>,
353    stats: Arc<StreamerStats>,
354    quic_server_params: QuicStreamerConfig,
355    cancel: CancellationToken,
356    qos: Arc<Q>,
357) -> TaskTracker
358where
359    Q: QosController<C> + Send + Sync + 'static,
360    C: ConnectionContext + Send + Sync + 'static,
361{
362    let quic_server_params = Arc::new(quic_server_params);
363    let rate_limiter = Arc::new(ConnectionRateLimiter::new(
364        quic_server_params.max_connections_per_ipaddr_per_min,
365    ));
366    let overall_connection_rate_limiter = Arc::new(TotalConnectionRateLimiter::new(
367        TOTAL_CONNECTIONS_PER_SECOND,
368    ));
369
370    const WAIT_FOR_CONNECTION_TIMEOUT: Duration = Duration::from_secs(1);
371    debug!("spawn quic server");
372    let mut last_datapoint = Instant::now();
373    stats
374        .quic_endpoints_count
375        .store(endpoints.len(), Ordering::Relaxed);
376
377    let mut accepts = endpoints
378        .iter()
379        .enumerate()
380        .map(|(i, incoming)| {
381            Box::pin(EndpointAccept {
382                accept: incoming.accept(),
383                endpoint: i,
384            })
385        })
386        .collect::<FuturesUnordered<_>>();
387
388    let tasks = TaskTracker::new();
389    loop {
390        let timeout_connection = select! {
391            ready = accepts.next() => {
392                if let Some((connecting, i)) = ready {
393                    accepts.push(
394                        Box::pin(EndpointAccept {
395                            accept: endpoints[i].accept(),
396                            endpoint: i,
397                        }
398                    ));
399                    Ok(connecting)
400                } else {
401                    // we can't really get here - we never poll an empty FuturesUnordered
402                    continue
403                }
404            }
405            _ = tokio::time::sleep(WAIT_FOR_CONNECTION_TIMEOUT) => {
406                Err(())
407            }
408            _ = cancel.cancelled() => break,
409        };
410
411        if last_datapoint.elapsed().as_secs() >= 5 {
412            stats.report(name);
413            last_datapoint = Instant::now();
414        }
415
416        if let Ok(Some(incoming)) = timeout_connection {
417            stats
418                .total_incoming_connection_attempts
419                .fetch_add(1, Ordering::Relaxed);
420
421            // first do per IpAddr rate limiting
422            if rate_limiter.len() > CONNECTION_RATE_LIMITER_CLEANUP_SIZE_THRESHOLD {
423                rate_limiter.retain_recent();
424            }
425            stats
426                .connection_rate_limiter_length
427                .store(rate_limiter.len(), Ordering::Relaxed);
428
429            let Ok(client_connection_tracker) = ClientConnectionTracker::new(
430                stats.clone(),
431                quic_server_params.max_concurrent_connections(),
432            ) else {
433                stats
434                    .refused_connections_too_many_open_connections
435                    .fetch_add(1, Ordering::Relaxed);
436                incoming.refuse();
437                continue;
438            };
439
440            stats
441                .outstanding_incoming_connection_attempts
442                .fetch_add(1, Ordering::Relaxed);
443            let connecting = incoming.accept();
444            match connecting {
445                Ok(connecting) => {
446                    let rate_limiter = rate_limiter.clone();
447                    let overall_connection_rate_limiter = overall_connection_rate_limiter.clone();
448                    tasks.spawn(setup_connection(
449                        connecting,
450                        rate_limiter,
451                        overall_connection_rate_limiter,
452                        client_connection_tracker,
453                        packet_batch_sender.clone(),
454                        stats.clone(),
455                        quic_server_params.clone(),
456                        qos.clone(),
457                        tasks.clone(),
458                    ));
459                }
460                Err(err) => {
461                    stats
462                        .outstanding_incoming_connection_attempts
463                        .fetch_sub(1, Ordering::Relaxed);
464                    debug!("Incoming::accept(): error {err:?}");
465                }
466            }
467        } else {
468            debug!("accept(): Timed out waiting for connection");
469        }
470    }
471    tasks
472}

522#[allow(clippy::too_many_arguments)]
523async fn setup_connection<Q, C>(
524    connecting: Connecting,
525    rate_limiter: Arc<ConnectionRateLimiter>,
526    overall_connection_rate_limiter: Arc<TotalConnectionRateLimiter>,
527    client_connection_tracker: ClientConnectionTracker,
528    packet_sender: Sender<PacketAccumulator>,
529    stats: Arc<StreamerStats>,
530    server_params: Arc<QuicStreamerConfig>,
531    qos: Arc<Q>,
532    tasks: TaskTracker,
533) where
534    Q: QosController<C> + Send + Sync + 'static,
535    C: ConnectionContext + Send + Sync + 'static,
536{
537    let from = connecting.remote_address();
538    let res = timeout(QUIC_CONNECTION_HANDSHAKE_TIMEOUT, connecting).await;
539    stats
540        .outstanding_incoming_connection_attempts
541        .fetch_sub(1, Ordering::Relaxed);
542    if let Ok(connecting_result) = res {
543        match connecting_result {
544            Ok(new_connection) => {
545                debug!("Got a connection {from:?}");
546                if !rate_limiter.is_allowed(&from.ip()) {
547                    debug!("Reject connection from {from:?} -- rate limiting exceeded");
548                    stats
549                        .connection_rate_limited_per_ipaddr
550                        .fetch_add(1, Ordering::Relaxed);
551                    new_connection.close(
552                        CONNECTION_CLOSE_CODE_DISALLOWED.into(),
553                        CONNECTION_CLOSE_REASON_DISALLOWED,
554                    );
555                    return;
556                }
557
558                if !overall_connection_rate_limiter.is_allowed() {
559                    debug!(
560                        "Reject connection from {:?} -- total rate limiting exceeded",
561                        from.ip()
562                    );
563                    stats
564                        .connection_rate_limited_across_all
565                        .fetch_add(1, Ordering::Relaxed);
566                    new_connection.close(
567                        CONNECTION_CLOSE_CODE_DISALLOWED.into(),
568                        CONNECTION_CLOSE_REASON_DISALLOWED,
569                    );
570                    return;
571                }
572
573                stats.total_new_connections.fetch_add(1, Ordering::Relaxed);
574
575                let mut conn_context = qos.build_connection_context(&new_connection);
576                if let Some(cancel_connection) = qos
577                    .try_add_connection(
578                        client_connection_tracker,
579                        &new_connection,
580                        &mut conn_context,
581                    )
582                    .await
583                {
584                    tasks.spawn(handle_connection(
585                        packet_sender.clone(),
586                        new_connection,
587                        stats,
588                        server_params.wait_for_chunk_timeout,
589                        conn_context.clone(),
590                        qos,
591                        cancel_connection,
592                    ));
593                }
594            }
595            Err(e) => {
596                handle_connection_error(e, &stats, from);
597            }
598        }
599    } else {
600        stats
601            .connection_setup_timeout
602            .fetch_add(1, Ordering::Relaxed);
603    }
604}

644// Holder(s) of the Sender<PacketAccumulator> on the other end should not
645// wait for this function to exit
646fn run_packet_batch_sender(
647    packet_sender: Sender<PacketBatch>,
648    packet_receiver: Receiver<PacketAccumulator>,
649    stats: Arc<StreamerStats>,
650    cancel: CancellationToken,
651) {
652    let mut channel_disconnected = false;
653    trace!("enter packet_batch_sender");
654    loop {
655        let mut packet_perf_measure: Vec<([u8; 64], Instant)> = Vec::default();
656        let mut packet_batch = BytesPacketBatch::with_capacity(PACKETS_PER_BATCH);
657        let mut total_bytes: usize = 0;
658
659        stats
660            .total_packet_batches_allocated
661            .fetch_add(1, Ordering::Relaxed);
662        stats
663            .total_packets_allocated
664            .fetch_add(PACKETS_PER_BATCH, Ordering::Relaxed);
665
666        loop {
667            if cancel.is_cancelled() || channel_disconnected {
668                return;
669            }
670            if !packet_batch.is_empty() {
671                let len = packet_batch.len();
672                track_streamer_fetch_packet_performance(&packet_perf_measure, &stats);
673
674                if let Err(e) = packet_sender.try_send(packet_batch.into()) {
675                    stats
676                        .total_packet_batch_send_err
677                        .fetch_add(1, Ordering::Relaxed);
678                    trace!("Send error: {e}");
679
680                    // The downstream channel is disconnected, this error is not recoverable.
681                    if matches!(e, TrySendError::Disconnected(_)) {
682                        cancel.cancel();
683                        return;
684                    }
685                } else {
686                    stats
687                        .total_packet_batches_sent
688                        .fetch_add(1, Ordering::Relaxed);
689
690                    stats
691                        .total_packets_sent_to_consumer
692                        .fetch_add(len, Ordering::Relaxed);
693
694                    stats
695                        .total_bytes_sent_to_consumer
696                        .fetch_add(total_bytes, Ordering::Relaxed);
697
698                    trace!("Sent {len} packet batch");
699                }
700                break;
701            }
702
703            // On the first receive, we block on recv not to use excessive CPU when the channel is idle.
704            // This will not block the exit as the channel will be dropped on the sender's side.
705            //
706            // On subsequent receives, we call try_recv, so that we do not get blocked waiting for packets
707            // when we already have something in the batch.
708            //
709            // For setting channel_disconnected, we can ignore TryRecvError::Disconnected on try_recv and
710            // set it on the next iteration (if we don't exit early anyway from cancel token)
711            let mut first = true;
712            let mut recv = || {
713                if first {
714                    first = false;
715                    // recv is only an error if empty and disconnected
716                    packet_receiver.recv().map_err(|_| {
717                        channel_disconnected = true;
718                        TryRecvError::Disconnected
719                    })
720                } else {
721                    packet_receiver.try_recv()
722                }
723            };
724            while let Ok(mut packet_accumulator) = recv() {
725                // 86% of transactions/packets come in one chunk. In that case,
726                // we can just move the chunk to the `Packet` and no copy is
727                // made.
728                // 14% of them come in multiple chunks. In that case, we copy
729                // them into one `Bytes` buffer. We make a copy once, with
730                // intention to not do it again.
731                let num_chunks = packet_accumulator.chunks.len();
732                let mut packet = if packet_accumulator.chunks.len() == 1 {
733                    BytesPacket::new(
734                        packet_accumulator.chunks.pop().expect("expected one chunk"),
735                        packet_accumulator.meta,
736                    )
737                } else {
738                    let size: usize = packet_accumulator.chunks.iter().map(Bytes::len).sum();
739                    let mut buf = BytesMut::with_capacity(size);
740                    for chunk in packet_accumulator.chunks {
741                        buf.put_slice(&chunk);
742                    }
743                    BytesPacket::new(buf.freeze(), packet_accumulator.meta)
744                };
745
746                total_bytes += packet.meta().size;
747
748                if let Some(signature) = signature_if_should_track_packet(&packet).ok().flatten() {
749                    packet_perf_measure.push((*signature, packet_accumulator.start_time));
750                    // we set the PERF_TRACK_PACKET on
751                    packet.meta_mut().set_track_performance(true);
752                }
753                packet_batch.push(packet);
754                stats
755                    .total_chunks_processed_by_batcher
756                    .fetch_add(num_chunks, Ordering::Relaxed);
757
758                // prevent getting stuck in loop
759                if packet_batch.len() >= PACKETS_PER_BATCH {
760                    break;
761                }
762            }
763        }
764    }
765}

767fn track_streamer_fetch_packet_performance(
768    packet_perf_measure: &[([u8; 64], Instant)],
769    stats: &StreamerStats,
770) {
771    if packet_perf_measure.is_empty() {
772        return;
773    }
774    let mut measure = Measure::start("track_perf");
775    let mut process_sampled_packets_us_hist = stats.process_sampled_packets_us_hist.lock().unwrap();
776
777    let now = Instant::now();
778    for (signature, start_time) in packet_perf_measure {
779        let duration = now.duration_since(*start_time);
780        debug!(
781            "QUIC streamer fetch stage took {duration:?} for transaction {:?}",
782            Signature::from(*signature)
783        );
784        process_sampled_packets_us_hist
785            .increment(duration.as_micros() as u64)
786            .unwrap();
787    }
788
789    drop(process_sampled_packets_us_hist);
790    measure.stop();
791    stats
792        .perf_track_overhead_us
793        .fetch_add(measure.as_us(), Ordering::Relaxed);
794}

796async fn handle_connection<Q, C>(
797    packet_sender: Sender<PacketAccumulator>,
798    connection: Connection,
799    stats: Arc<StreamerStats>,
800    wait_for_chunk_timeout: Duration,
801    context: C,
802    qos: Arc<Q>,
803    cancel: CancellationToken,
804) where
805    Q: QosController<C> + Send + Sync + 'static,
806    C: ConnectionContext + Send + Sync + 'static,
807{
808    let peer_type = context.peer_type();
809    let remote_addr = connection.remote_address();
810    debug!(
811        "quic new connection {} streams: {} connections: {}",
812        remote_addr,
813        stats.active_streams.load(Ordering::Relaxed),
814        stats.total_connections.load(Ordering::Relaxed),
815    );
816    stats.total_connections.fetch_add(1, Ordering::Relaxed);
817
818    'conn: loop {
819        // Wait for new streams. If the peer is disconnected we get a cancellation signal and stop
820        // the connection task.
821        let mut stream = select! {
822            stream = connection.accept_uni() => match stream {
823                Ok(stream) => stream,
824                Err(e) => {
825                    debug!("stream error: {e:?}");
826                    break;
827                }
828            },
829            _ = cancel.cancelled() => break,
830        };
831
832        qos.on_new_stream(&context).await;
833        qos.on_stream_accepted(&context);
834        stats.active_streams.fetch_add(1, Ordering::Relaxed);
835        stats.total_new_streams.fetch_add(1, Ordering::Relaxed);
836
837        let mut meta = Meta::default();
838        meta.set_socket_addr(&remote_addr);
839        meta.set_from_staked_node(matches!(peer_type, ConnectionPeerType::Staked(_)));
840        let mut accum = PacketAccumulator::new(meta);
841
842        // Virtually all small transactions will fit in 1 chunk. Larger transactions will fit in 1
843        // or 2 chunks if the first chunk starts towards the end of a datagram. A small number of
844        // transaction will have other protocol frames inserted in the middle. Empirically it's been
845        // observed that 4 is the maximum number of chunks txs get split into.
846        //
847        // Bytes values are small, so overall the array takes only 128 bytes, and the "cost" of
848        // overallocating a few bytes is negligible compared to the cost of having to do multiple
849        // read_chunks() calls.
850        let mut chunks: [Bytes; 4] = array::from_fn(|_| Bytes::new());
851
852        loop {
853            // Read the next chunks, waiting up to `wait_for_chunk_timeout`. If we don't get chunks
854            // before then, we assume the stream is dead. This can only happen if there's severe
855            // packet loss or the peer stops sending for whatever reason.
856            let n_chunks = match tokio::select! {
857                chunk = tokio::time::timeout(
858                    wait_for_chunk_timeout,
859                    stream.read_chunks(&mut chunks)) => chunk,
860
861                // If the peer gets disconnected stop the task right away.
862                _ = cancel.cancelled() => break,
863            } {
864                // read_chunk returned success
865                Ok(Ok(chunk)) => chunk.unwrap_or(0),
866                // read_chunk returned error
867                Ok(Err(e)) => {
868                    debug!("Received stream error: {e:?}");
869                    stats
870                        .total_stream_read_errors
871                        .fetch_add(1, Ordering::Relaxed);
872                    break;
873                }
874                // timeout elapsed
875                Err(_) => {
876                    debug!("Timeout in receiving on stream");
877                    stats
878                        .total_stream_read_timeouts
879                        .fetch_add(1, Ordering::Relaxed);
880                    break;
881                }
882            };
883
884            match handle_chunks(
885                // Bytes::clone() is a cheap atomic inc
886                chunks.iter().take(n_chunks).cloned(),
887                &mut accum,
888                &packet_sender,
889                &stats,
890                peer_type,
891            ) {
892                // The stream is finished, break out of the loop and close the stream.
893                Ok(StreamState::Finished) => {
894                    qos.on_stream_finished(&context);
895                    break;
896                }
897                // The stream is still active, continue reading.
898                Ok(StreamState::Receiving) => {}
899                Err(_) => {
900                    // Disconnect peers that send invalid streams.
901                    connection.close(
902                        CONNECTION_CLOSE_CODE_INVALID_STREAM.into(),
903                        CONNECTION_CLOSE_REASON_INVALID_STREAM,
904                    );
905                    stats.active_streams.fetch_sub(1, Ordering::Relaxed);
906                    qos.on_stream_error(&context);
907                    break 'conn;
908                }
909            }
910        }
911
912        stats.active_streams.fetch_sub(1, Ordering::Relaxed);
913        qos.on_stream_closed(&context);
914    }
915
916    let removed_connection_count = qos.remove_connection(&context, connection).await;
917    if removed_connection_count > 0 {
918        stats
919            .connection_removed
920            .fetch_add(removed_connection_count, Ordering::Relaxed);
921    } else {
922        stats
923            .connection_remove_failed
924            .fetch_add(1, Ordering::Relaxed);
925    }
926    stats.total_connections.fetch_sub(1, Ordering::Relaxed);
927}

929enum StreamState {
930    // Stream is not finished, keep receiving chunks
931    Receiving,
932    // Stream is finished
933    Finished,
934}
935
936// Handle the chunks received from the stream. If the stream is finished, send the packet to the
937// packet sender.
938//
939// Returns Err(()) if the stream is invalid.
940fn handle_chunks(
941    chunks: impl ExactSizeIterator<Item = Bytes>,
942    accum: &mut PacketAccumulator,
943    packet_sender: &Sender<PacketAccumulator>,
944    stats: &StreamerStats,
945    peer_type: ConnectionPeerType,
946) -> Result<StreamState, ()> {
947    let n_chunks = chunks.len();
948    for chunk in chunks {
949        accum.meta.size += chunk.len();
950        if accum.meta.size > PACKET_DATA_SIZE {
951            // The stream window size is set to PACKET_DATA_SIZE, so one individual chunk can
952            // never exceed this size. A peer can send two chunks that together exceed the size
953            // tho, in which case we report the error.
954            stats.invalid_stream_size.fetch_add(1, Ordering::Relaxed);
955            debug!("invalid stream size {}", accum.meta.size);
956            return Err(());
957        }
958        accum.chunks.push(chunk);
959        if peer_type.is_staked() {
960            stats
961                .total_staked_chunks_received
962                .fetch_add(1, Ordering::Relaxed);
963        } else {
964            stats
965                .total_unstaked_chunks_received
966                .fetch_add(1, Ordering::Relaxed);
967        }
968    }
969
970    // n_chunks == 0 marks the end of a stream
971    if n_chunks != 0 {
972        return Ok(StreamState::Receiving);
973    }
974
975    if accum.chunks.is_empty() {
976        debug!("stream is empty");
977        stats
978            .total_packet_batches_none
979            .fetch_add(1, Ordering::Relaxed);
980        return Err(());
981    }
982
983    // done receiving chunks
984    let bytes_sent = accum.meta.size;
985    let chunks_sent = accum.chunks.len();
986
987    if let Err(err) = packet_sender.try_send(accum.clone()) {
988        stats
989            .total_handle_chunk_to_packet_batcher_send_err
990            .fetch_add(1, Ordering::Relaxed);
991        match err {
992            TrySendError::Full(_) => {
993                stats
994                    .total_handle_chunk_to_packet_batcher_send_full_err
995                    .fetch_add(1, Ordering::Relaxed);
996            }
997            TrySendError::Disconnected(_) => {
998                stats
999                    .total_handle_chunk_to_packet_batcher_send_disconnected_err
1000                    .fetch_add(1, Ordering::Relaxed);
1001            }
1002        }
1003        trace!("packet batch send error {err:?}");
1004    } else {
1005        stats
1006            .total_packets_sent_for_batching
1007            .fetch_add(1, Ordering::Relaxed);
1008        stats
1009            .total_bytes_sent_for_batching
1010            .fetch_add(bytes_sent, Ordering::Relaxed);
1011        stats
1012            .total_chunks_sent_for_batching
1013            .fetch_add(chunks_sent, Ordering::Relaxed);
1014
1015        match peer_type {
1016            ConnectionPeerType::Unstaked => {
1017                stats
1018                    .total_unstaked_packets_sent_for_batching
1019                    .fetch_add(1, Ordering::Relaxed);
1020            }
1021            ConnectionPeerType::Staked(_) => {
1022                stats
1023                    .total_staked_packets_sent_for_batching
1024                    .fetch_add(1, Ordering::Relaxed);
1025            }
1026        }
1027
1028        trace!("sent {bytes_sent} byte packet for batching");
1029    }
1030
1031    Ok(StreamState::Finished)
1032}

1118/// Prune the connection which has the oldest update
1119///
1120/// Return number pruned
1121impl ConnectionTable {
1122    pub(crate) fn new(table_type: ConnectionTableType, cancel: CancellationToken) -> Self {
1123        Self {
1124            table: IndexMap::default(),
1125            total_size: 0,
1126            table_type,
1127            cancel,
1128        }
1129    }
1130
1131    fn table_size(&self) -> usize {
1132        self.total_size
1133    }
1134
1135    fn is_staked(&self) -> bool {
1136        matches!(self.table_type, ConnectionTableType::Staked)
1137    }
1138
1139    pub(crate) fn prune_oldest(&mut self, max_size: usize) -> usize {
1140        let mut num_pruned = 0;
1141        let key = |(_, connections): &(_, &Vec<_>)| {
1142            connections.iter().map(ConnectionEntry::last_update).min()
1143        };
1144        while self.total_size.saturating_sub(num_pruned) > max_size {
1145            match self.table.values().enumerate().min_by_key(key) {
1146                None => break,
1147                Some((index, connections)) => {
1148                    num_pruned += connections.len();
1149                    self.table.swap_remove_index(index);
1150                }
1151            }
1152        }
1153        self.total_size = self.total_size.saturating_sub(num_pruned);
1154        num_pruned
1155    }
1156
1157    // Randomly selects sample_size many connections, evicts the one with the
1158    // lowest stake, and returns the number of pruned connections.
1159    // If the stakes of all the sampled connections are higher than the
1160    // threshold_stake, rejects the pruning attempt, and returns 0.
1161    pub(crate) fn prune_random(&mut self, sample_size: usize, threshold_stake: u64) -> usize {
1162        let num_pruned = std::iter::once(self.table.len())
1163            .filter(|&size| size > 0)
1164            .flat_map(|size| {
1165                let mut rng = thread_rng();
1166                repeat_with(move || rng.gen_range(0..size))
1167            })
1168            .map(|index| {
1169                let connection = self.table[index].first();
1170                let stake = connection.map(|connection: &ConnectionEntry| connection.stake());
1171                (index, stake)
1172            })
1173            .take(sample_size)
1174            .min_by_key(|&(_, stake)| stake)
1175            .filter(|&(_, stake)| stake < Some(threshold_stake))
1176            .and_then(|(index, _)| self.table.swap_remove_index(index))
1177            .map(|(_, connections)| connections.len())
1178            .unwrap_or_default();
1179        self.total_size = self.total_size.saturating_sub(num_pruned);
1180        num_pruned
1181    }

1183    pub(crate) fn try_add_connection(
1184        &mut self,
1185        key: ConnectionTableKey,
1186        port: u16,
1187        client_connection_tracker: ClientConnectionTracker,
1188        connection: Option<Connection>,
1189        peer_type: ConnectionPeerType,
1190        last_update: Arc<AtomicU64>,
1191        max_connections_per_peer: usize,
1192    ) -> Option<(
1193        Arc<AtomicU64>,
1194        CancellationToken,
1195        Arc<ConnectionStreamCounter>,
1196    )> {
1197        let connection_entry = self.table.entry(key).or_default();
1198        let has_connection_capacity = connection_entry
1199            .len()
1200            .checked_add(1)
1201            .map(|c| c <= max_connections_per_peer)
1202            .unwrap_or(false);
1203        if has_connection_capacity {
1204            let cancel = self.cancel.child_token();
1205            let stream_counter = connection_entry
1206                .first()
1207                .map(|entry| entry.stream_counter.clone())
1208                .unwrap_or(Arc::new(ConnectionStreamCounter::new()));
1209            connection_entry.push(ConnectionEntry::new(
1210                cancel.clone(),
1211                peer_type,
1212                last_update.clone(),
1213                port,
1214                client_connection_tracker,
1215                connection,
1216                stream_counter.clone(),
1217            ));
1218            self.total_size += 1;
1219            Some((last_update, cancel, stream_counter))
1220        } else {
1221            if let Some(connection) = connection {
1222                connection.close(
1223                    CONNECTION_CLOSE_CODE_TOO_MANY.into(),
1224                    CONNECTION_CLOSE_REASON_TOO_MANY,
1225                );
1226            }
1227            None
1228        }
1229    }

1231    // Returns number of connections that were removed
1232    pub(crate) fn remove_connection(
1233        &mut self,
1234        key: ConnectionTableKey,
1235        port: u16,
1236        stable_id: usize,
1237    ) -> usize {
1238        if let Entry::Occupied(mut e) = self.table.entry(key) {
1239            let e_ref = e.get_mut();
1240            let old_size = e_ref.len();
1241
1242            e_ref.retain(|connection_entry| {
1243                // Retain the connection entry if the port is different, or if the connection's
1244                // stable_id doesn't match the provided stable_id.
1245                // (Some unit tests do not fill in a valid connection in the table. To support that,
1246                // if the connection is none, the stable_id check is ignored. i.e. if the port matches,
1247                // the connection gets removed)
1248                connection_entry.port != port
1249                    || connection_entry
1250                        .connection
1251                        .as_ref()
1252                        .and_then(|connection| (connection.stable_id() != stable_id).then_some(0))
1253                        .is_some()
1254            });
1255            let new_size = e_ref.len();
1256            if e_ref.is_empty() {
1257                e.swap_remove_entry();
1258            }
1259            let connections_removed = old_size.saturating_sub(new_size);
1260            self.total_size = self.total_size.saturating_sub(connections_removed);
1261            connections_removed
1262        } else {
1263            0
1264        }
1265    }
