12/// Write messages using the Thrift compact protocol.
13#[derive(Debug)]
14pub struct TCompactOutputProtocol<T>
15where
16    T: Write,
17{
18    // Identifier of the last field serialized for a struct.
19    last_write_field_id: i16,
20    // Stack of the last written field ids (new entry added each time a nested struct is written).
21    write_field_id_stack: Vec<i16>,
22    // Field identifier of the boolean field to be written.
23    // Saved because boolean fields and their value are encoded in a single byte
24    pending_write_bool_field_identifier: Option<TFieldIdentifier>,
25    // Underlying transport used for byte-level operations.
26    transport: T,
27}
28
29impl<T> TCompactOutputProtocol<T>
30where
31    T: Write,
32{
33    /// Create a `TCompactOutputProtocol` that writes bytes to `transport`.
34    pub fn new(transport: T) -> TCompactOutputProtocol<T> {
35        TCompactOutputProtocol {
36            last_write_field_id: 0,
37            write_field_id_stack: Vec::new(),
38            pending_write_bool_field_identifier: None,
39            transport,
40        }
41    }
42
43    // FIXME: field_type as unconstrained u8 is bad
44    fn write_field_header(&mut self, field_type: u8, field_id: i16) -> Result<usize> {
45        let mut written = 0;
46
47        let field_delta = field_id - self.last_write_field_id;
48        if field_delta > 0 && field_delta < 15 {
49            written += self.write_byte(((field_delta as u8) << 4) | field_type)?;
50        } else {
51            written += self.write_byte(field_type)?;
52            written += self.write_i16(field_id)?;
53        }
54        self.last_write_field_id = field_id;
55        Ok(written)
56    }
57
58    fn write_list_set_begin(&mut self, element_type: TType, element_count: u32) -> Result<usize> {
59        let mut written = 0;
60
61        let elem_identifier = collection_type_to_u8(element_type);
62        if element_count <= 14 {
63            let header = (element_count as u8) << 4 | elem_identifier;
64            written += self.write_byte(header)?;
65        } else {
66            let header = 0xF0 | elem_identifier;
67            written += self.write_byte(header)?;
68            written += self.transport.write_varint(element_count)?;
69        }
70        Ok(written)
71    }
72
73    fn assert_no_pending_bool_write(&self) {
74        if let Some(ref f) = self.pending_write_bool_field_identifier {
75            panic!("pending bool field {:?} not written", f)
76        }
77    }
78}

80impl<T> TOutputProtocol for TCompactOutputProtocol<T>
81where
82    T: Write,
83{
84    fn write_message_begin(&mut self, identifier: &TMessageIdentifier) -> Result<usize> {
85        let mut written = 0;
86        written += self.write_byte(COMPACT_PROTOCOL_ID)?;
87        written += self.write_byte((u8::from(identifier.message_type) << 5) | COMPACT_VERSION)?;
88        written += self.transport.write_varint(identifier.sequence_number)?;
89        written += self.write_string(&identifier.name)?;
90        Ok(written)
91    }
92
93    fn write_message_end(&mut self) -> Result<usize> {
94        self.assert_no_pending_bool_write();
95        Ok(0)
96    }
97
98    fn write_struct_begin(&mut self, _: &TStructIdentifier) -> Result<usize> {
99        self.write_field_id_stack.push(self.last_write_field_id);
100        self.last_write_field_id = 0;
101        Ok(0)
102    }
103
104    fn write_struct_end(&mut self) -> Result<usize> {
105        self.assert_no_pending_bool_write();
106        self.last_write_field_id = self
107            .write_field_id_stack
108            .pop()
109            .expect("should have previous field ids");
110        Ok(0)
111    }
112
113    fn write_field_begin(&mut self, identifier: &TFieldIdentifier) -> Result<usize> {
114        match identifier.field_type {
115            TType::Bool => {
116                if self.pending_write_bool_field_identifier.is_some() {
117                    panic!(
118                        "should not have a pending bool while writing another bool with id: \
119                         {:?}",
120                        identifier
121                    )
122                }
123                self.pending_write_bool_field_identifier = Some(identifier.clone());
124                Ok(0)
125            }
126            _ => {
127                let field_type = type_to_u8(identifier.field_type);
128                let field_id = identifier.id.expect("non-stop field should have field id");
129                self.write_field_header(field_type, field_id)
130            }
131        }
132    }
133
134    fn write_field_end(&mut self) -> Result<usize> {
135        self.assert_no_pending_bool_write();
136        Ok(0)
137    }
138
139    fn write_field_stop(&mut self) -> Result<usize> {
140        self.assert_no_pending_bool_write();
141        self.write_byte(type_to_u8(TType::Stop))
142    }
143
144    fn write_bool(&mut self, b: bool) -> Result<usize> {
145        match self.pending_write_bool_field_identifier.take() {
146            Some(pending) => {
147                let field_id = pending.id.expect("bool field should have a field id");
148                let field_type_as_u8 = if b { 0x01 } else { 0x02 };
149                self.write_field_header(field_type_as_u8, field_id)
150            }
151            None => {
152                if b {
153                    self.write_byte(0x01)
154                } else {
155                    self.write_byte(0x02)
156                }
157            }
158        }
159    }
160
161    fn write_bytes(&mut self, b: &[u8]) -> Result<usize> {
162        let mut written = 0;
163        written += self.transport.write_varint::<u32>(b.len().try_into()?)?;
164        self.transport.write_all(b)?;
165        written += b.len();
166        Ok(written)
167    }
168
169    fn write_i8(&mut self, i: i8) -> Result<usize> {
170        self.write_byte(i as u8)
171    }
172
173    fn write_i16(&mut self, i: i16) -> Result<usize> {
174        self.transport.write_varint(i).map_err(From::from)
175    }
176
177    fn write_i32(&mut self, i: i32) -> Result<usize> {
178        self.transport.write_varint(i).map_err(From::from)
179    }
180
181    fn write_i64(&mut self, i: i64) -> Result<usize> {
182        self.transport.write_varint(i).map_err(From::from)
183    }
184
185    fn write_double(&mut self, d: f64) -> Result<usize> {
186        let bytes = d.to_le_bytes();
187        self.transport.write_all(&bytes)?;
188        Ok(8)
189    }
190
191    fn write_string(&mut self, s: &str) -> Result<usize> {
192        self.write_bytes(s.as_bytes())
193    }
194
195    fn write_list_begin(&mut self, identifier: &TListIdentifier) -> Result<usize> {
196        self.write_list_set_begin(identifier.element_type, identifier.size)
197    }
198
199    fn write_list_end(&mut self) -> Result<usize> {
200        Ok(0)
201    }
202
203    fn write_set_begin(&mut self, identifier: &TSetIdentifier) -> Result<usize> {
204        self.write_list_set_begin(identifier.element_type, identifier.size)
205    }
206
207    fn write_set_end(&mut self) -> Result<usize> {
208        Ok(0)
209    }
210
211    fn write_map_begin(&mut self, identifier: &TMapIdentifier) -> Result<usize> {
212        if identifier.size == 0 {
213            self.write_byte(0)
214        } else {
215            let mut written = 0;
216            written += self.transport.write_varint(identifier.size)?;
217
218            let key_type = identifier
219                .key_type
220                .expect("map identifier to write should contain key type");
221            let key_type_byte = collection_type_to_u8(key_type) << 4;
222
223            let val_type = identifier
224                .value_type
225                .expect("map identifier to write should contain value type");
226            let val_type_byte = collection_type_to_u8(val_type);
227
228            let map_type_header = key_type_byte | val_type_byte;
229            written += self.write_byte(map_type_header)?;
230            Ok(written)
231        }
232    }
233
234    fn write_map_end(&mut self) -> Result<usize> {
235        Ok(0)
236    }
237
238    fn flush(&mut self) -> Result<()> {
239        self.transport.flush().map_err(From::from)
240    }
241
242    // utility
243    //
244
245    fn write_byte(&mut self, b: u8) -> Result<usize> {
246        self.transport.write(&[b]).map_err(From::from)
247    }
248}

250pub(super) fn collection_type_to_u8(field_type: TType) -> u8 {
251    match field_type {
252        TType::Bool => 0x01,
253        f => type_to_u8(f),
254    }
255}
256
257pub(super) fn type_to_u8(field_type: TType) -> u8 {
258    match field_type {
259        TType::Stop => 0x00,
260        TType::I08 => 0x03, // equivalent to TType::Byte
261        TType::I16 => 0x04,
262        TType::I32 => 0x05,
263        TType::I64 => 0x06,
264        TType::Double => 0x07,
265        TType::String => 0x08,
266        TType::List => 0x09,
267        TType::Set => 0x0A,
268        TType::Map => 0x0B,
269        TType::Struct => 0x0C,
270        _ => panic!("should not have attempted to convert {} to u8", field_type),
271    }
