102    /// Deserializes the input `VoteStateVersions` buffer directly into the provided `VoteStateV4`.
103    ///
104    /// In a SBPF context, V0_23_5 is not supported, but in non-SBPF, all versions are supported for
105    /// compatibility with `bincode::deserialize`.
106    ///
107    /// On success, `vote_state` reflects the state of the input data. On failure, `vote_state` is
108    /// reset to `VoteStateV4::default()`.
109    #[cfg(any(target_os = "solana", feature = "bincode"))]
110    pub fn deserialize_into(
111        input: &[u8],
112        vote_state: &mut VoteStateV4,
113        vote_pubkey: &Pubkey,
114    ) -> Result<(), InstructionError> {
115        use super::vote_state_deserialize;
116        vote_state_deserialize::deserialize_into(input, vote_state, |input, vote_state| {
117            Self::deserialize_into_ptr(input, vote_state, vote_pubkey)
118        })
119    }

121    /// Deserializes the input `VoteStateVersions` buffer directly into the provided
122    /// `MaybeUninit<VoteStateV4>`.
123    ///
124    /// In a SBPF context, V0_23_5 is not supported, but in non-SBPF, all versions are supported for
125    /// compatibility with `bincode::deserialize`.
126    ///
127    /// On success, `vote_state` is fully initialized and can be converted to
128    /// `VoteStateV4` using
129    /// [`MaybeUninit::assume_init`](https://doc.rust-lang.org/std/mem/union.MaybeUninit.html#method.assume_init).
130    /// On failure, `vote_state` may still be uninitialized and must not be
131    /// converted to `VoteStateV4`.
132    #[cfg(any(target_os = "solana", feature = "bincode"))]
133    pub fn deserialize_into_uninit(
134        input: &[u8],
135        vote_state: &mut std::mem::MaybeUninit<VoteStateV4>,
136        vote_pubkey: &Pubkey,
137    ) -> Result<(), InstructionError> {
138        Self::deserialize_into_ptr(input, vote_state.as_mut_ptr(), vote_pubkey)
139    }

141    #[cfg(any(target_os = "solana", feature = "bincode"))]
142    fn deserialize_into_ptr(
143        input: &[u8],
144        vote_state: *mut VoteStateV4,
145        vote_pubkey: &Pubkey,
146    ) -> Result<(), InstructionError> {
147        use super::vote_state_deserialize::{deserialize_vote_state_into_v4, SourceVersion};
148
149        let mut cursor = std::io::Cursor::new(input);
150
151        let variant = solana_serialize_utils::cursor::read_u32(&mut cursor)?;
152        match variant {
153            // V0_23_5. not supported for bpf targets; these should not exist on mainnet
154            // supported for non-bpf targets for backwards compatibility.
155            // **Same pattern as v3 for this variant**.
156            0 => {
157                #[cfg(not(target_os = "solana"))]
158                {
159                    // Safety: vote_state is valid as it comes from `&mut MaybeUninit<VoteStateV4>` or
160                    // `&mut VoteStateV4`. In the first case, the value is uninitialized so we write()
161                    // to avoid dropping invalid data; in the latter case, we `drop_in_place()`
162                    // before writing so the value has already been dropped and we just write a new
163                    // one in place.
164                    unsafe {
165                        vote_state.write(
166                            bincode::deserialize::<VoteStateVersions>(input)
167                                .map_err(|_| InstructionError::InvalidAccountData)
168                                .and_then(|versioned| versioned.try_convert_to_v4(vote_pubkey))?,
169                        );
170                    }
171                    Ok(())
172                }
173                #[cfg(target_os = "solana")]
174                Err(InstructionError::InvalidAccountData)
175            }
176            // V1_14_11
177            1 => deserialize_vote_state_into_v4(
178                &mut cursor,
179                vote_state,
180                SourceVersion::V1_14_11 { vote_pubkey },
181            ),
182            // V3
183            2 => deserialize_vote_state_into_v4(
184                &mut cursor,
185                vote_state,
186                SourceVersion::V3 { vote_pubkey },
187            ),
188            // V4
189            3 => deserialize_vote_state_into_v4(&mut cursor, vote_state, SourceVersion::V4),
190            _ => Err(InstructionError::InvalidAccountData),
191        }?;
192
193        Ok(())
194    }

196    #[cfg(feature = "bincode")]
197    pub fn serialize(
198        versioned: &VoteStateVersions,
199        output: &mut [u8],
200    ) -> Result<(), InstructionError> {
201        bincode::serialize_into(output, versioned).map_err(|err| match *err {
202            bincode::ErrorKind::SizeLimit => InstructionError::AccountDataTooSmall,
203            _ => InstructionError::GenericError,
204        })
205    }

207    pub fn is_correct_size_and_initialized(data: &[u8]) -> bool {
208        data.len() == VoteStateV4::size_of() && data[..4] == [3, 0, 0, 0] // little-endian 3u32
209                                                                          // Always initialized
210    }

212    /// Number of credits owed to this account.
213    pub fn credits(&self) -> u64 {
214        if self.epoch_credits.is_empty() {
215            0
216        } else {
217            self.epoch_credits.last().unwrap().1
218        }
219    }

231        }
232
233        Self {
234            votes: VecDeque::from(vec![LandedVote::default(); MAX_LOCKOUT_HISTORY]),
235            root_slot: Some(u64::MAX),
236            epoch_credits: vec![(0, 0, 0); MAX_EPOCH_CREDITS_HISTORY],
