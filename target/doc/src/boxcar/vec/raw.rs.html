13/// A lock-free, append-only vector.
14pub struct Vec<T> {
15    /// A counter used to retrieve a unique index to push to.
16    ///
17    /// Stores a `buckets::Index` that may be out-of-bounds on
18    /// the positive side, but never overflows.
19    ///
20    /// This value may be more than the true length as it will
21    /// be incremented before values are actually stored.
22    inflight: AtomicUsize,
23
24    /// Buckets of length 32, 64 .. 2^62.
25    buckets: Buckets<Entry<T>, BUCKETS>,
26
27    /// The number of initialized elements in this vector.
28    count: AtomicUsize,
29}
30
31impl<T> Vec<T> {
32    /// Create an empty vector.
33    #[cfg(not(loom))]
34    pub const fn new() -> Vec<T> {
35        let Some(zero) = <buckets::Index<BUCKETS>>::new(0) else {
36            unreachable!();
37        };
38
39        Vec {
40            inflight: AtomicUsize::new(zero.into_raw().get()),
41            buckets: Buckets::new(),
42            count: AtomicUsize::new(0),
43        }
44    }
45
46    /// Create an empty vector.
47    #[cfg(loom)]
48    pub fn new() -> Vec<T> {
49        Vec {
50            inflight: AtomicUsize::new(<buckets::Index<BUCKETS>>::new(0).unwrap().into_raw().get()),
51            buckets: Buckets::new(),
52            count: AtomicUsize::new(0),
53        }
54    }
55
56    /// Constructs a new, empty `Vec<T>` with the specified capacity.
57    #[inline]
58    pub fn with_capacity(capacity: usize) -> Vec<T> {
59        let mut this = Self::new();
60
61        if let Some(highest_index) = capacity.checked_sub(1) {
62            this.buckets
63                .reserve_mut(buckets::Index::new_saturating(highest_index));
64        }
65
66        this
67    }
68
69    /// Returns the number of elements in the vector.
70    #[inline]
71    pub fn count(&self) -> usize {
72        // The `Acquire` here synchronizes with the `Release` increment
73        // when an entry is added to the vector.
74        self.count.load(Ordering::Acquire)
75    }
76
77    /// Returns a reference to the element at the given index.
78    #[inline]
79    pub fn get(&self, index: usize) -> Option<&T> {
80        // If the index is out-of-bounds, or the bucket is not allocated, return `None`.
81        let entry = self.buckets.get(buckets::Index::new(index)?)?;
82
83        if entry.active.load(Ordering::Acquire) {
84            // Safety: The entry is active.
85            unsafe { return Some(entry.value_unchecked()) }
86        }
87
88        // The entry is uninitialized.
89        None
90    }
91
92    /// Returns a reference to the element at the given index.
93    ///
94    /// # Safety
95    ///
96    /// The entry at `index` must be initialized.
97    #[inline]
98    pub unsafe fn get_unchecked(&self, index: usize) -> &T {
99        // Safety: Caller guarantees the bucket and entry are initialized.
100        unsafe {
101            let entry = self
102                .buckets
103                .get_unchecked(buckets::Index::new_unchecked(index));
104            (*entry).value_unchecked()
105        }
106    }
107
108    /// Returns a mutable reference to the element at the given index.
109    #[inline]
110    pub fn get_mut(&mut self, index: usize) -> Option<&mut T> {
111        // If the index is out-of-bounds, or the bucket is not allocated, return `None`.
112        let entry = self.buckets.get_mut(buckets::Index::new(index)?)?;
113
114        if entry.active.read_mut() {
115            // Safety: The entry is active.
116            unsafe { return Some(entry.value_unchecked_mut()) }
117        }
118
119        // The entry is uninitialized.
120        None
121    }
122
123    ///
124    /// # Safety
125    ///
126    /// The entry at `index` must be initialized.
127    #[inline]
128    pub unsafe fn get_unchecked_mut(&mut self, index: usize) -> &mut T {
129        // Safety: Caller guarantees the bucket and entry are initialized.
130        unsafe {
131            let entry = self
132                .buckets
133                .get_unchecked_mut(buckets::Index::new_unchecked(index));
134            (*entry).value_unchecked_mut()
135        }
136    }
137
138    /// Returns a unique index for insertion.
139    #[inline]
140    fn next_index(&self) -> buckets::Index<BUCKETS> {
141        // The inflight counter cannot exceed `isize::MAX`, allowing it to catch capacity overflow.
142        //
143        // Note that the `Relaxed` ordering here is sufficient, as we only care about
144        // the index being unique and do not use it for synchronization.
145        let index = self.inflight.fetch_add(1, Ordering::Relaxed);
146
147        // Safety: We cannot overflow.
148        let Some(index) = (unsafe { buckets::Index::from_raw_checked_above(index) }) else {
149            self.next_index_overflow();
150        };
151
152        index
153    }
154
155    #[cold]
156    #[inline(never)]
157    fn next_index_overflow(&self) -> ! {
158        // We could alternatively abort here, as `Arc` does. But we decrement and panic instead
159        // to keep in line with `Vec`'s behavior. Assuming that `isize::MAX` concurrent threads
160        // don't call this method, it is still impossible for it to overflow.
161        self.inflight.fetch_sub(1, Ordering::Relaxed);
162        panic!("capacity overflow");
163    }
164
165    /// Appends the element returned from the closure to the back of the vector
166    /// at the index represented by the `usize` passed to closure.
167    ///
168    /// This allows for use of the would-be index to be utilized within the
169    /// element.
170    #[inline]
171    pub fn push_with<F>(&self, f: F) -> usize
172    where
173        F: FnOnce(usize) -> T,
174    {
175        // Acquire a unique index to insert into.
176        let index = self.next_index();
177        let value = f(index.get());
178
179        // Safety: `next_index` is always in-bounds and unique.
180        unsafe { self.write(index, value) }
181    }
182
183    /// Appends an element to the back of the vector.
184    #[inline]
185    pub fn push(&self, value: T) -> usize {
186        // Safety: `next_index` is always in-bounds and unique.
187        unsafe { self.write(self.next_index(), value) }
188    }
189
190    /// Write an element at the given index.
191    ///
192    /// # Safety
193    ///
194    /// The index must be unique.
195    #[inline]
196    unsafe fn write(&self, index: buckets::Index<BUCKETS>, value: T) -> usize {
197        let entry = self.buckets.get_or_alloc(index);
198
199        unsafe {
200            // Safety: We have unique access to this entry (ensured by
201            // the caller).
202            //
203            // 1. It is impossible for another thread to attempt a `push`
204            // to this location as we retrieved it from `next_index`.
205            //
206            // 2. Any thread trying to `get` this entry will see `!active`
207            // and will not try to access it.
208            entry
209                .slot
210                .with_mut(|slot| slot.write(MaybeUninit::new(value)));
211
212            // Let other threads know that this entry is active.
213            //
214            // Note that this `Release` write synchronizes with the `Acquire`
215            // load in `Vec::get`.
216            entry.active.store(true, Ordering::Release);
217        }
218
219        // Increase the element count.
220        //
221        // The `Release` here is not strictly necessary, but does
222        // allow users to use `count` for some sort of synchronization
223        // in terms of the number of initialized elements.
224        self.count.fetch_add(1, Ordering::Release);
225
226        // Return the index of the entry that we initialized.
227        index.get()
228    }
229
230    /// Reserves capacity for at least `additional` more elements to be inserted in
231    /// the vector.
232    ///
233    /// The collection may reserve more space to avoid frequent reallocations.
234    pub fn reserve(&self, additional: usize) {
235        let len = self.count.load(Ordering::Acquire);
236
237        if let Some(highest_index) = len.saturating_add(additional).checked_sub(1) {
238            self.buckets
239                .reserve(buckets::Index::new_saturating(highest_index));
240        }
241    }
242
243    /// Iterates over shared references to values in the vector.
244    #[inline]
245    pub fn iter(&self) -> Iter<'_, T> {
246        Iter {
247            inner: self.buckets.iter(),
248            // Snapshot the number of potentially active entries when the iterator is
249            // created, so we know how many entries we have to check. The alternative
250            // is to always check every single entry and/or bucket. Note that yielding
251            // `self.count` entries is not enough; we may end up yielding the *wrong*
252            // `self.count` entries, as the gaps in-between the entries we want to yield
253            // may get filled concurrently.
254            // Safety: We cannot overflow.
255            after_inflight: unsafe {
256                buckets::Index::from_raw_checked_above(self.inflight.load(Ordering::Relaxed))
257            },
258        }
259    }
260
261    /// Iterates over owned values in the vector.
262    #[inline]
263    pub fn into_iter(mut self) -> IntoIter<T> {
264        IntoIter {
265            inner: self.buckets.into_iter(),
266            remaining: self.count.read_mut(),
267        }
268    }
269
270    /// Clear every element in the vector.
271    #[inline]
272    pub fn clear(&mut self) {
273        // Consume and reset every entry in the vector.
274        self.buckets
275            .iter_mut()
276            .filter_map(|(_, entry)| entry.take())
277            .take(self.count.read_mut())
278            .for_each(drop);
279
280        // Reset the count.
281        self.count.write_mut(0);
282        self.inflight
283            .write_mut(<buckets::Index<BUCKETS>>::new(0).unwrap().into_raw().get());
284    }
285}

287impl<T> Index<usize> for Vec<T> {
288    type Output = T;
289
290    #[inline]
291    fn index(&self, index: usize) -> &Self::Output {
292        #[cold]
293        #[inline(never)]
294        fn assert_failed(index: usize) -> ! {
295            panic!("index {index} is uninitialized");
296        }
297
298        match self.get(index) {
299            Some(value) => value,
300            None => assert_failed(index),
301        }
302    }
303}

305/// A possibly uninitialized entry in the vector.
306struct Entry<T> {
307    /// A flag indicating whether or not this entry is initialized.
308    active: AtomicBool,
309
310    /// The entry's value.
311    slot: UnsafeCell<MaybeUninit<T>>,
312}
313
314impl<T> Default for Entry<T> {
315    fn default() -> Self {
316        Self {
317            active: AtomicBool::new(false),
318            slot: UnsafeCell::new(MaybeUninit::uninit()),
319        }
320    }
321}
322
323// Safety: `AtomicBool` and `MaybeUninit` are all zeroable outside Loom.
324unsafe impl<T> MaybeZeroable for Entry<T> {
325    fn zeroable() -> bool {
326        // Loom's types are not zeroable; thus we fall back to `Default`.
327        cfg!(not(loom))
328    }
329}
330
331// Safety: An `Entry` is owned and owns its data, so sending an
332// entry only sends that data, hence `T: Send`.
333unsafe impl<T: Send> Send for Entry<T> {}
334
335// Safety: Sharing an `Entry` exposes shared access to the
336// data inside, hence `T: Sync`. Additionally, the `Entry`
337// may act as a channel, exposing owned access of elements
338// to other threads, hence we also require `T: Send`.
339unsafe impl<T: Send + Sync> Sync for Entry<T> {}
340
341// In an ideal world, we might require `T: UnwindSafe + RefUnwindSafe`
342// here, with similar reasoning to above. But this is here for backward
343// compatibility. Also, who ever worried about unwind safety :P
344impl<T> RefUnwindSafe for Entry<T> {}
345
346impl<T> Entry<T> {
347    /// Returns a reference to the value in this entry.
348    ///
349    /// # Safety
350    ///
351    /// The value must be initialized.
352    #[inline]
353    unsafe fn value_unchecked(&self) -> &T {
354        // Safety: Guaranteed by caller.
355        self.slot.with(|slot| unsafe { (*slot).assume_init_ref() })
356    }
357
358    /// Returns a mutable reference to the value in this entry.
359    ///
360    // # Safety
361    //
362    // The value must be initialized.
363    #[inline]
364    unsafe fn value_unchecked_mut(&mut self) -> &mut T {
365        // Safety: Guaranteed by caller.
366        self.slot
367            .with_mut(|slot| unsafe { (*slot).assume_init_mut() })
368    }
369
370    /// Takes the value in the entry, if there is one.
371    fn take(&mut self) -> Option<T> {
372        if self.active.read_mut() {
373            // Mark the entry as uninitialized so it is not accessed after this.
374            self.active.write_mut(false);
375
376            let value = mem::replace(&mut self.slot, UnsafeCell::new(MaybeUninit::uninit()));
377
378            // Safety: We just verified that the entry is initialized.
379            Some(unsafe { value.into_inner().assume_init() })
380        } else {
381            None
382        }
383    }
384}
385
386impl<T> Drop for Entry<T> {
387    fn drop(&mut self) {
388        if self.active.read_mut() {
389            // Safety: We have `&mut self` and verifid that the value is initialized.
390            unsafe { ptr::drop_in_place(self.slot.with_mut(|slot| (*slot).as_mut_ptr())) }
391        }
392    }
393}

395/// The number of buckets in a vector.
396const BUCKETS: usize = buckets_for_index_bits(usize::BITS - 1);
397
398pub struct Iter<'a, T> {
399    inner: buckets::Iter<'a, Entry<T>, BUCKETS>,
400    after_inflight: Option<buckets::Index<BUCKETS>>,
401}
402
403impl<T> Clone for Iter<'_, T> {
404    fn clone(&self) -> Self {
405        Self {
406            inner: self.inner.clone(),
407            after_inflight: self.after_inflight,
408        }
409    }
410}
411
412impl<'a, T> Iterator for Iter<'a, T> {
413    type Item = (usize, &'a T);
414
415    fn next(&mut self) -> Option<Self::Item> {
416        loop {
417            let (index, entry) = self.inner.next()?;
418
419            if self
420                .after_inflight
421                .is_some_and(|after_inflight| index >= after_inflight)
422            {
423                return None;
424            }
425
426            if entry.active.load(Ordering::Acquire) {
427                // Safety: We just verified that the entry is initialized.
428                return Some((index.get(), unsafe { entry.value_unchecked() }));
429            }
430        }
431    }
432}

434pub struct IntoIter<T> {
435    inner: buckets::IntoIter<Entry<T>, BUCKETS>,
436    remaining: usize,
437}
438
439impl<T> Iterator for IntoIter<T> {
440    type Item = T;
441
442    fn next(&mut self) -> Option<Self::Item> {
443        loop {
444            self.remaining = self.remaining.checked_sub(1)?;
445
446            let (_, mut entry) = self.inner.next()?;
447            if let Some(value) = entry.take() {
448                return Some(value);
449            }
450        }
451    }
452
453    fn size_hint(&self) -> (usize, Option<usize>) {
454        (self.remaining, Some(self.remaining))
455    }
456}

458impl<T> ExactSizeIterator for IntoIter<T> {
459    fn len(&self) -> usize {
460        self.remaining
