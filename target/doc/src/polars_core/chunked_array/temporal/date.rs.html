8pub(crate) fn naive_date_to_date(nd: NaiveDate) -> i32 {
9    let nt = NaiveTime::from_hms_opt(0, 0, 0).unwrap();
10    let ndt = NaiveDateTime::new(nd, nt);
11    naive_datetime_to_date(ndt)
12}

14impl DateChunked {
15    pub fn as_date_iter(&self) -> impl TrustedLen<Item = Option<NaiveDate>> + '_ {
16        // SAFETY: we know the iterators len
17        unsafe {
18            self.physical()
19                .downcast_iter()
20                .flat_map(|iter| {
21                    iter.into_iter()
22                        .map(|opt_v| opt_v.copied().map(date32_to_date))
23                })
24                .trust_my_length(self.len())
25        }
26    }
27
28    /// Construct a new [`DateChunked`] from an iterator over [`NaiveDate`].
29    pub fn from_naive_date<I: IntoIterator<Item = NaiveDate>>(name: PlSmallStr, v: I) -> Self {
30        let unit = v.into_iter().map(naive_date_to_date).collect::<Vec<_>>();
31        Int32Chunked::from_vec(name, unit).into_date()
32    }

34    /// Convert from Date into String with the given format.
35    /// See [chrono strftime/strptime](https://docs.rs/chrono/0.4.19/chrono/format/strftime/index.html).
36    pub fn to_string(&self, format: &str) -> PolarsResult<StringChunked> {
37        let format = if format == "iso" || format == "iso:strict" {
38            "%F"
39        } else {
40            format
41        };
42        let datefmt_f = |ndt: NaiveDate| ndt.format(format);
43        self.physical()
44            .try_apply_into_string_amortized(|val, buf| {
45                let ndt = date32_to_date(val);
46                write!(buf, "{}", datefmt_f(ndt))
47            })
48            .map_err(|_| polars_err!(ComputeError: "cannot format Date with format '{}'", format))
49    }
50
51    /// Convert from Date into String with the given format.
52    /// See [chrono strftime/strptime](https://docs.rs/chrono/0.4.19/chrono/format/strftime/index.html).
53    ///
54    /// Alias for `to_string`.
55    pub fn strftime(&self, format: &str) -> PolarsResult<StringChunked> {
56        self.to_string(format)
57    }
58
59    /// Construct a new [`DateChunked`] from an iterator over optional [`NaiveDate`].
60    pub fn from_naive_date_options<I: IntoIterator<Item = Option<NaiveDate>>>(
61        name: PlSmallStr,
62        v: I,
63    ) -> Self {
64        let unit = v.into_iter().map(|opt| opt.map(naive_date_to_date));
65        Int32Chunked::from_iter_options(name, unit).into_date()
66    }
