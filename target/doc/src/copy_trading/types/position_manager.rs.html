14/// Central coordinator for managing copy trading positions and monitoring wallet activity.
15/// Monitors swap transactions, market updates, and new pool creations, automatically opening and
16/// closing positions based on tracked wallet behavior. It maintains active positions and wallet
17/// configurations for copy trading.
18pub struct PositionManager {
19  /*
20  A list of open/active positions we currently have. The key is the MARKET address involved and the
21  value is the Position struct. The position struct will eventually be removed by the position
22  manager if it detects a market update on a closed position. For pumpfun bonding curve positions,
23  the market address is the bonding curve address.
24
25  The reason why we need to use Arc<Mutex<Position>> is because there are asynchronous operations
26  which may involve changing the position state, such as a market state update causing a stop loss
27  check and a tx confirmation coming in to set the entry price we bought at and the initial stop
28  loss threshold price. The reason why we use a Mutex is because pretty much all of these
29  operations require a first read of the position state such as its status, and then write and
30  modify it, and there isn't really a scenario where we need multiple readers in parallel.
31  */
32  pub open_positions: RwLock<HashMap<Pubkey, Arc<Mutex<Position>>>>,
33  /*
34  Configs for positions by wallet. The key is the wallet address we are copytrading, and the value
35  is the config that we use to copy trade them. This hashmap will eventually be contained inside of
36  an RwLock because the wallets that we are copytrading throughout the lifecycle will change TODO change this eventually
37  */
38  pub wallet_configs: HashMap<Pubkey, PositionConfig>,
39  pub tx_monitor_receiver: broadcast::Receiver<SwapTx>,
40  /*
41  The stream of all market data updates that the position manager processes and then sends out to
42  the positions individually
43  */
44  pub market_update_receiver: broadcast::Receiver<MarketUpdate>,
45  /*
46  An update of new pools that are created. This is relevant to us for pumpfun migrations.
47  We will check if there is a position open for a bonding curve that is in the complete state which
48  has had a pumpswap pool created for it, meaning the token has migrated to trading on pumpswap
49  */
50  pub new_pool_receiver: broadcast::Receiver<Arc<RwLock<dyn PoolTrait>>>,
51  /*
52  The public key of the solana wallet that will be opening these positions and making the trades.
53  */
54  pub wallet: Arc<Keypair>,
55  /*
56  Smart pointer to the central context
57  */
58  pub central_context: Arc<CentralContext>,
59  pub execution_context: Arc<ExecutionContext>,
60}
61
62impl PositionManager {
63  pub fn new(
64    tx_monitor_receiver: broadcast::Receiver<SwapTx>,
65    market_update_receiver: broadcast::Receiver<MarketUpdate>,
66    new_pool_receiver: broadcast::Receiver<Arc<RwLock<dyn PoolTrait>>>,
67    wallet: Arc<Keypair>,
68    central_context: Arc<CentralContext>,
69    execution_context: Arc<ExecutionContext>,
70  ) -> Self {
71    let result = Self {
72      open_positions: RwLock::new(HashMap::new()),
73      wallet_configs: HashMap::new(),
74      tx_monitor_receiver,
75      market_update_receiver,
76      new_pool_receiver,
77      wallet,
78      central_context,
79      execution_context,
80    };
81    // result.start_loop();
82    result
83  }
