1use solana_sdk::pubkey::Pubkey;

3/// Configuration for copy trading a specific wallet, defining position sizing and risk management.
4/// Each tracked wallet has its own configuration specifying buy amounts, hold times, stop-loss
5/// thresholds, and slippage tolerances.
6#[derive(Debug, Copy, Clone)]
7pub struct PositionConfig {
8  // _lp means lamports
9  pub buy_amount_lp: u64,
10  pub jito_tip_buy_lp: u64,
11  pub hold_time_ms: u32,
12  pub jito_tip_sell_lp: u64,
13  pub use_stop_loss: bool,
14  pub trailing_stop_loss: bool,
15  // for a 15% stop loss threshold from your entry price, you would do 150000000
16  pub stop_loss_threshold_lp: u64,
17  pub wallet_copying: Pubkey,
18  // 35% slippage would be 350000000 for example
19  pub slippage_lps: u64,
20}
