1//! The [`VariableLenPack`] serialization trait.
2
3use solana_program_error::ProgramError;
4
5/// Trait that mimics a lot of the functionality of
6/// `solana_program_pack::Pack` but specifically works for
7/// variable-size types.
8pub trait VariableLenPack {
9    /// Writes the serialized form of the instance into the given slice
10    fn pack_into_slice(&self, dst: &mut [u8]) -> Result<(), ProgramError>;
11
12    /// Deserializes the type from the given slice
13    fn unpack_from_slice(src: &[u8]) -> Result<Self, ProgramError>
14    where
15        Self: Sized;
16
17    /// Gets the packed length for a given instance of the type
18    fn get_packed_len(&self) -> Result<usize, ProgramError>;
19
20    /// Safely write the contents to the type into the given slice
21    fn pack(&self, dst: &mut [u8]) -> Result<(), ProgramError> {
22        if dst.len() != self.get_packed_len()? {
23            return Err(ProgramError::InvalidAccountData);
24        }
25        self.pack_into_slice(dst)
26    }
27}
