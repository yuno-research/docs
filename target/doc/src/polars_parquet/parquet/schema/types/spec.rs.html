5fn check_decimal_invariants(
6    physical_type: &PhysicalType,
7    precision: usize,
8    scale: usize,
9) -> ParquetResult<()> {
10    if precision < 1 {
11        return Err(ParquetError::oos(format!(
12            "DECIMAL precision must be larger than 0; It is {precision}",
13        )));
14    }
15    if scale > precision {
16        return Err(ParquetError::oos(format!(
17            "Invalid DECIMAL: scale ({scale}) cannot be greater than precision \
18            ({precision})"
19        )));
20    }
21
22    match physical_type {
23        PhysicalType::Int32 => {
24            if !(1..=9).contains(&precision) {
25                return Err(ParquetError::oos(format!(
26                    "Cannot represent INT32 as DECIMAL with precision {precision}"
27                )));
28            }
29        },
30        PhysicalType::Int64 => {
31            if !(1..=18).contains(&precision) {
32                return Err(ParquetError::oos(format!(
33                    "Cannot represent INT64 as DECIMAL with precision {precision}"
34                )));
35            }
36        },
37        PhysicalType::FixedLenByteArray(length) => {
38            let oos_error = || ParquetError::oos(format!("Byte Array length {length} out of spec"));
39            let max_precision = (2f64.powi(
40                (*length as i32)
41                    .checked_mul(8)
42                    .ok_or_else(oos_error)?
43                    .checked_sub(1)
44                    .ok_or_else(oos_error)?,
45            ) - 1f64)
46                .log10()
47                .floor() as usize;
48
49            if precision > max_precision {
50                return Err(ParquetError::oos(format!(
51                    "Cannot represent FIXED_LEN_BYTE_ARRAY as DECIMAL with length {length} and \
52                    precision {precision}. The max precision can only be {max_precision}"
53                )));
54            }
55        },
56        PhysicalType::ByteArray => {},
57        _ => {
58            return Err(ParquetError::oos(
59                "DECIMAL can only annotate INT32, INT64, BYTE_ARRAY and FIXED_LEN_BYTE_ARRAY",
60            ));
61        },
62    };
63    Ok(())
64}

66pub fn check_converted_invariants(
67    physical_type: &PhysicalType,
68    converted_type: &Option<PrimitiveConvertedType>,
69) -> ParquetResult<()> {
70    if converted_type.is_none() {
71        return Ok(());
72    };
73    let converted_type = converted_type.as_ref().unwrap();
74
75    use PrimitiveConvertedType::*;
76    match converted_type {
77        Utf8 | Bson | Json => {
78            if physical_type != &PhysicalType::ByteArray {
79                return Err(ParquetError::oos(format!(
80                    "{converted_type:?} can only annotate BYTE_ARRAY fields"
81                )));
82            }
83        },
84        Decimal(precision, scale) => {
85            check_decimal_invariants(physical_type, *precision, *scale)?;
86        },
87        Date | TimeMillis | Uint8 | Uint16 | Uint32 | Int8 | Int16 | Int32 => {
88            if physical_type != &PhysicalType::Int32 {
89                return Err(ParquetError::oos(format!(
90                    "{converted_type:?} can only annotate INT32"
91                )));
92            }
93        },
94        TimeMicros | TimestampMillis | TimestampMicros | Uint64 | Int64 => {
95            if physical_type != &PhysicalType::Int64 {
96                return Err(ParquetError::oos(format!(
97                    "{converted_type:?} can only annotate INT64"
98                )));
99            }
100        },
101        Interval => {
102            if physical_type != &PhysicalType::FixedLenByteArray(12) {
103                return Err(ParquetError::oos(
104                    "INTERVAL can only annotate FIXED_LEN_BYTE_ARRAY(12)",
105                ));
106            }
107        },
108        Enum => {
109            if physical_type != &PhysicalType::ByteArray {
110                return Err(ParquetError::oos(
111                    "ENUM can only annotate BYTE_ARRAY fields",
112                ));
113            }
114        },
115    };
116    Ok(())
117}

119pub fn check_logical_invariants(
120    physical_type: &PhysicalType,
121    logical_type: PrimitiveLogicalType,
122) -> ParquetResult<()> {
123    // Check that logical type and physical type are compatible
124    use PrimitiveLogicalType::*;
125    match (logical_type, physical_type) {
126        (Enum, PhysicalType::ByteArray) => {},
127        (Decimal(precision, scale), _) => {
128            check_decimal_invariants(physical_type, precision, scale)?;
129        },
130        (Date, PhysicalType::Int32) => {},
131        (
132            Time {
133                unit: TimeUnit::Milliseconds,
134                ..
135            },
136            PhysicalType::Int32,
137        ) => {},
138        (Time { unit, .. }, PhysicalType::Int64) => {
139            if unit == TimeUnit::Milliseconds {
140                return Err(ParquetError::oos(
141                    "Cannot use millisecond unit on INT64 type",
142                ));
143            }
144        },
145        (Timestamp { .. }, PhysicalType::Int64) => {},
146        (Integer(IntegerType::Int8), PhysicalType::Int32) => {},
147        (Integer(IntegerType::Int16), PhysicalType::Int32) => {},
148        (Integer(IntegerType::Int32), PhysicalType::Int32) => {},
149        (Integer(IntegerType::UInt8), PhysicalType::Int32) => {},
150        (Integer(IntegerType::UInt16), PhysicalType::Int32) => {},
151        (Integer(IntegerType::UInt32), PhysicalType::Int32) => {},
152        (Integer(IntegerType::UInt64), PhysicalType::Int64) => {},
153        (Integer(IntegerType::Int64), PhysicalType::Int64) => {},
154        // Null type
155        (Unknown, PhysicalType::Int32) => {},
156        (String | Json | Bson, PhysicalType::ByteArray) => {},
157        // https://github.com/apache/parquet-format/blob/master/LogicalTypes.md#uuid
158        (Uuid, PhysicalType::FixedLenByteArray(16)) => {},
159        (Float16, PhysicalType::FixedLenByteArray(2)) => {},
160        (a, b) => {
161            return Err(ParquetError::oos(format!(
162                "Cannot annotate {a:?} from {b:?} fields"
163            )));
164        },
165    };
166    Ok(())
167}
