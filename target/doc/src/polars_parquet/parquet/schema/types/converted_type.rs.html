9pub enum PrimitiveConvertedType {

11    /// an enum is converted into a binary field
12    Enum,
13    /// A decimal value.
14    ///
15    /// This may be used to annotate binary or fixed primitive types. The underlying byte array
16    /// stores the unscaled value encoded as two's complement using big-endian byte order (the most
17    /// significant byte is the zeroth element). The value of the decimal is the value *
18    /// 10^{-scale}.
19    ///
20    /// This must be accompanied by a (maximum) precision and a scale in the SchemaElement. The
21    /// precision specifies the number of digits in the decimal and the scale stores the location
22    /// of the decimal point. For example 1.23 would have precision 3 (3 total digits) and scale 2
23    /// (the decimal point is 2 digits over).
24    // (precision, scale)
25    Decimal(usize, usize),
26    /// A Date
27    ///
28    /// Stored as days since Unix epoch, encoded as the INT32 physical type.
29    ///
30    Date,
31    /// A time
32    ///
33    /// The total number of milliseconds since midnight.  The value is stored as an INT32 physical
34    /// type.
35    TimeMillis,
36    /// A time.
37    ///
38    /// The total number of microseconds since midnight.  The value is stored as an INT64 physical
39    /// type.
40    TimeMicros,
41    /// A date/time combination
42    ///
43    /// Date and time recorded as milliseconds since the Unix epoch.  Recorded as a physical type
44    /// of INT64.
45    TimestampMillis,
46    /// A date/time combination
47    ///
48    /// Date and time recorded as microseconds since the Unix epoch.  The value is stored as an
49    /// INT64 physical type.
50    TimestampMicros,
51    /// An unsigned integer value.
52    ///
53    /// The number describes the maximum number of meaningful data bits in the stored value. 8, 16
54    /// and 32 bit values are stored using the INT32 physical type.  64 bit values are stored using
55    /// the INT64 physical type.
56    Uint8,
57    Uint16,
58    Uint32,
59    Uint64,
60    /// A signed integer value.
61    ///
62    /// The number describes the maximum number of meainful data bits in the stored value. 8, 16
63    /// and 32 bit values are stored using the INT32 physical type.  64 bit values are stored using
64    /// the INT64 physical type.
65    ///
66    Int8,
67    Int16,
68    Int32,
69    Int64,
70    /// An embedded JSON document
71    ///
72    /// A JSON document embedded within a single UTF8 column.
73    Json,
74    /// An embedded BSON document
75    ///
76    /// A BSON document embedded within a single BINARY column.
77    Bson,
78    /// An interval of time
79    ///
80    /// This type annotates data stored as a FIXED_LEN_BYTE_ARRAY of length 12 This data is

85    /// duration of time is independent of any particular timezone or date.
86    Interval,

100impl TryFrom<(ConvertedType, Option<(i32, i32)>)> for PrimitiveConvertedType {
101    type Error = ParquetError;
102
103    fn try_from(
104        (ty, maybe_decimal): (ConvertedType, Option<(i32, i32)>),
105    ) -> Result<Self, Self::Error> {
106        use PrimitiveConvertedType::*;
107        Ok(match ty {
108            ConvertedType::UTF8 => Utf8,
109            ConvertedType::ENUM => Enum,
110            ConvertedType::DECIMAL => {
111                if let Some((precision, scale)) = maybe_decimal {
112                    Decimal(precision.try_into()?, scale.try_into()?)
113                } else {
114                    return Err(ParquetError::oos("Decimal requires a precision and scale"));
115                }
116            },
117            ConvertedType::DATE => Date,
118            ConvertedType::TIME_MILLIS => TimeMillis,
119            ConvertedType::TIME_MICROS => TimeMicros,
120            ConvertedType::TIMESTAMP_MILLIS => TimestampMillis,
121            ConvertedType::TIMESTAMP_MICROS => TimestampMicros,
122            ConvertedType::UINT_8 => Uint8,
123            ConvertedType::UINT_16 => Uint16,
124            ConvertedType::UINT_32 => Uint32,
125            ConvertedType::UINT_64 => Uint64,
126            ConvertedType::INT_8 => Int8,
127            ConvertedType::INT_16 => Int16,
128            ConvertedType::INT_32 => Int32,
129            ConvertedType::INT_64 => Int64,
130            ConvertedType::JSON => Json,
131            ConvertedType::BSON => Bson,
132            ConvertedType::INTERVAL => Interval,
133            _ => {
134                return Err(ParquetError::oos(format!(
135                    "Converted type \"{ty:?}\" cannot be applied to a primitive type"
136                )));
137            },
138        })
139    }
140}

142impl TryFrom<ConvertedType> for GroupConvertedType {
143    type Error = ParquetError;
144
145    fn try_from(type_: ConvertedType) -> Result<Self, Self::Error> {
146        Ok(match type_ {
147            ConvertedType::LIST => GroupConvertedType::List,
148            ConvertedType::MAP => GroupConvertedType::Map,
149            ConvertedType::MAP_KEY_VALUE => GroupConvertedType::MapKeyValue,
150            _ => return Err(ParquetError::oos("LogicalType value out of range")),
151        })
152    }
153}

165impl From<PrimitiveConvertedType> for (ConvertedType, Option<(i32, i32)>) {
166    fn from(ty: PrimitiveConvertedType) -> Self {
167        use PrimitiveConvertedType::*;
168        match ty {
169            Utf8 => (ConvertedType::UTF8, None),
170            Enum => (ConvertedType::ENUM, None),
171            Decimal(precision, scale) => (
172                ConvertedType::DECIMAL,
173                Some((precision as i32, scale as i32)),
174            ),
175            Date => (ConvertedType::DATE, None),
176            TimeMillis => (ConvertedType::TIME_MILLIS, None),
177            TimeMicros => (ConvertedType::TIME_MICROS, None),
178            TimestampMillis => (ConvertedType::TIMESTAMP_MILLIS, None),
179            TimestampMicros => (ConvertedType::TIMESTAMP_MICROS, None),
180            Uint8 => (ConvertedType::UINT_8, None),
181            Uint16 => (ConvertedType::UINT_16, None),
182            Uint32 => (ConvertedType::UINT_32, None),
183            Uint64 => (ConvertedType::UINT_64, None),
184            Int8 => (ConvertedType::INT_8, None),
185            Int16 => (ConvertedType::INT_16, None),
186            Int32 => (ConvertedType::INT_32, None),
187            Int64 => (ConvertedType::INT_64, None),
188            Json => (ConvertedType::JSON, None),
189            Bson => (ConvertedType::BSON, None),
190            Interval => (ConvertedType::INTERVAL, None),
191        }
192    }
193}
