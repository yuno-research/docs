25pub(crate) enum StateTranslation<'a> {
26    Plain(&'a [u8], usize),
27    Dictionary(hybrid_rle::HybridRleDecoder<'a>),
28}

30impl<'a> utils::StateTranslation<'a, BinaryDecoder> for StateTranslation<'a> {
31    type PlainDecoder = &'a [u8];
32
33    fn new(
34        decoder: &BinaryDecoder,
35        page: &'a DataPage,
36        dict: Option<&'a <BinaryDecoder as Decoder>::Dict>,
37        page_validity: Option<&Bitmap>,
38    ) -> ParquetResult<Self> {
39        match (page.encoding(), dict) {
40            (Encoding::Plain, _) => {
41                let values = split_buffer(page)?.values;
42                if values.len() % decoder.size != 0 {
43                    return Err(ParquetError::oos(format!(
44                        "Fixed size binary data length {} is not divisible by size {}",
45                        values.len(),
46                        decoder.size
47                    )));
48                }
49                Ok(Self::Plain(values, decoder.size))
50            },
51            (Encoding::PlainDictionary | Encoding::RleDictionary, Some(_)) => {
52                let values =
53                    dict_indices_decoder(page, page_validity.map_or(0, |bm| bm.unset_bits()))?;
54                Ok(Self::Dictionary(values))
55            },
56            _ => Err(utils::not_implemented(page)),
57        }
58    }
59
60    fn num_rows(&self) -> usize {
61        match self {
62            StateTranslation::Plain(v, n) => v.len() / n,
63            StateTranslation::Dictionary(i) => i.len(),
64        }
65    }
66}

83impl FSBVec {
84    pub fn new(size: usize) -> FSBVec {
85        match size {
86            1 => Self::Size1(Vec::new()),
87            2 => Self::Size2(Vec::new()),
88            4 => Self::Size4(Vec::new()),
89            8 => Self::Size8(Vec::new()),
90            12 => Self::Size12(Vec::new()),
91            16 => Self::Size16(Vec::new()),
92            32 => Self::Size32(Vec::new()),
93            _ => Self::Other(Vec::new(), size),
94        }
95    }
96
97    fn size(&self) -> usize {
98        match self {
99            FSBVec::Size1(_) => 1,
100            FSBVec::Size2(_) => 2,
101            FSBVec::Size4(_) => 4,
102            FSBVec::Size8(_) => 8,
103            FSBVec::Size12(_) => 12,
104            FSBVec::Size16(_) => 16,
105            FSBVec::Size32(_) => 32,
106            FSBVec::Other(_, size) => *size,
107        }
108    }
109
110    pub fn into_bytes_buffer(self) -> Buffer<u8> {
111        Buffer::from_storage(match self {
112            FSBVec::Size1(vec) => SharedStorage::bytes_from_pod_vec(vec),
113            FSBVec::Size2(vec) => SharedStorage::bytes_from_pod_vec(vec),
114            FSBVec::Size4(vec) => SharedStorage::bytes_from_pod_vec(vec),
115            FSBVec::Size8(vec) => SharedStorage::bytes_from_pod_vec(vec),
116            FSBVec::Size12(vec) => SharedStorage::bytes_from_pod_vec(vec),
117            FSBVec::Size16(vec) => SharedStorage::bytes_from_pod_vec(vec),
118            FSBVec::Size32(vec) => SharedStorage::bytes_from_pod_vec(vec),
119            FSBVec::Other(vec, _) => SharedStorage::from_vec(vec),
120        })
121    }
122
123    pub fn extend_from_byte_slice(&mut self, slice: &[u8]) {
124        let size = self.size();
125        if size == 0 {
126            assert_eq!(slice.len(), 0);
127            return;
128        }
129
130        assert_eq!(slice.len() % size, 0);
131
132        macro_rules! extend_from_slice {
133            ($v:expr) => {{
134                $v.reserve(slice.len() / size);
135                unsafe {
136                    std::ptr::copy_nonoverlapping(
137                        slice.as_ptr(),
138                        $v.as_mut_ptr().add($v.len()) as *mut _,
139                        slice.len(),
140                    )
141                }
142                let new_len = $v.len() + slice.len() / size;
143                unsafe { $v.set_len(new_len) };
144            }};
145        }
146
147        match self {
148            FSBVec::Size1(v) => extend_from_slice!(v),
149            FSBVec::Size2(v) => extend_from_slice!(v),
150            FSBVec::Size4(v) => extend_from_slice!(v),
151            FSBVec::Size8(v) => extend_from_slice!(v),
152            FSBVec::Size12(v) => extend_from_slice!(v),
153            FSBVec::Size16(v) => extend_from_slice!(v),
154            FSBVec::Size32(v) => extend_from_slice!(v),
155            FSBVec::Other(v, _) => v.extend_from_slice(slice),
156        }
157    }
158
159    fn len(&self) -> usize {
160        match self {
161            FSBVec::Size1(vec) => vec.len(),
162            FSBVec::Size2(vec) => vec.len(),
163            FSBVec::Size4(vec) => vec.len(),
164            FSBVec::Size8(vec) => vec.len(),
165            FSBVec::Size12(vec) => vec.len(),
166            FSBVec::Size16(vec) => vec.len(),
167            FSBVec::Size32(vec) => vec.len(),
168            FSBVec::Other(vec, size) => vec.len() / size,
169        }
170    }
171}

185            FSBVec::Size16(v) => v.resize(v.len() + n, Zeroable::zeroed()),

193#[allow(clippy::too_many_arguments)]
194fn decode_fsb_plain(
195    size: usize,
196    values: &[u8],
197    target: &mut FSBVec,
198    pred_true_mask: &mut BitmapBuilder,
199    validity: &mut BitmapBuilder,
200    is_optional: bool,
201    filter: Option<Filter>,
202    page_validity: Option<&Bitmap>,
203) -> ParquetResult<()> {
204    assert_ne!(size, 0);
205    assert_eq!(values.len() % size, 0);
206
207    macro_rules! decode_static_size {
208        ($target:ident) => {{
209            let values = ArrayChunks::new(values).ok_or_else(|| {
210                ParquetError::oos("Page content does not align with expected element size")
211            })?;
212            super::primitive::plain::decode_aligned_bytes_dispatch(
213                values,
214                is_optional,
215                page_validity,
216                filter,
217                validity,
218                $target,
219                pred_true_mask,
220            )
221        }};
222    }
223
224    use FSBVec as T;
225    match target {
226        T::Size1(target) => decode_static_size!(target),
227        T::Size2(target) => decode_static_size!(target),
228        T::Size4(target) => decode_static_size!(target),
229        T::Size8(target) => decode_static_size!(target),
230        T::Size12(target) => decode_static_size!(target),
231        T::Size16(target) => decode_static_size!(target),
232        T::Size32(target) => decode_static_size!(target),
233        T::Other(target, _) => {
234            // @NOTE: All these kernels are quite slow, but they should be very uncommon and the
235            // general case requires arbitrary length memcopies anyway.
236
237            if is_optional {
238                append_validity(
239                    page_validity,
240                    filter.as_ref(),
241                    validity,
242                    values.len() / size,
243                );
244            }
245
246            let page_validity =
247                constrain_page_validity(values.len() / size, page_validity, filter.as_ref());
248
249            match (page_validity, filter.as_ref()) {
250                (None, None) => target.extend_from_slice(values),
251                (None, Some(filter)) => match filter {
252                    Filter::Range(range) => {
253                        target.extend_from_slice(&values[range.start * size..range.end * size])
254                    },
255                    Filter::Mask(bitmap) => {
256                        let mut iter = bitmap.iter();
257                        let mut offset = 0;
258
259                        while iter.num_remaining() > 0 {
260                            let num_selected = iter.take_leading_ones();
261                            target
262                                .extend_from_slice(&values[offset * size..][..num_selected * size]);
263                            offset += num_selected;
264
265                            let num_filtered = iter.take_leading_zeros();
266                            offset += num_filtered;
267                        }
268                    },
269                    Filter::Predicate(_) => todo!(),
270                },
271                (Some(validity), None) => {
272                    let mut iter = validity.iter();
273                    let mut offset = 0;
274
275                    while iter.num_remaining() > 0 {
276                        let num_valid = iter.take_leading_ones();
277                        target.extend_from_slice(&values[offset * size..][..num_valid * size]);
278                        offset += num_valid;
279
280                        let num_filtered = iter.take_leading_zeros();
281                        target.resize(target.len() + num_filtered * size, 0);
282                    }
283                },
284                (Some(validity), Some(filter)) => match filter {
285                    Filter::Range(range) => {
286                        let (skipped, active) = validity.split_at(range.start);
287
288                        let active = active.sliced(0, range.len());
289
290                        let mut iter = active.iter();
291                        let mut offset = skipped.set_bits();
292
293                        while iter.num_remaining() > 0 {
294                            let num_valid = iter.take_leading_ones();
295                            target.extend_from_slice(&values[offset * size..][..num_valid * size]);
296                            offset += num_valid;
297
298                            let num_filtered = iter.take_leading_zeros();
299                            target.resize(target.len() + num_filtered * size, 0);
300                        }
301                    },
302                    Filter::Mask(filter) => {
303                        let mut offset = 0;
304                        for (is_selected, is_valid) in filter.iter().zip(validity.iter()) {
305                            if is_selected {
306                                if is_valid {
307                                    target.extend_from_slice(&values[offset * size..][..size]);
308                                } else {
309                                    target.resize(target.len() + size, 0);
310                                }
311                            }
312
313                            offset += usize::from(is_valid);
314                        }
315                    },
316                    Filter::Predicate(_) => todo!(),
317                },
318            }
319
320            Ok(())
321        },
322    }
323}

325#[allow(clippy::too_many_arguments)]
326fn decode_fsb_dict(
327    size: usize,
328    values: HybridRleDecoder<'_>,
329    dict: &FixedSizeBinaryArray,
330    dict_mask: Option<&Bitmap>,
331    target: &mut FSBVec,
332    pred_true_mask: &mut BitmapBuilder,
333    validity: &mut BitmapBuilder,
334    is_optional: bool,
335    filter: Option<Filter>,
336    page_validity: Option<&Bitmap>,
337) -> ParquetResult<()> {
338    assert_ne!(size, 0);
339
340    macro_rules! decode_static_size {
341        ($dict:ident, $target:ident) => {{
342            let dict = $dict.values().as_slice();
343            // @NOTE: We initialize the dict with the right alignment for this to work.
344            let dict = bytemuck::cast_slice(dict);
345
346            super::dictionary_encoded::decode_dict_dispatch(
347                values,
348                dict,
349                dict_mask,
350                is_optional,
351                page_validity,
352                filter,
353                validity,
354                $target,
355                pred_true_mask,
356            )
357        }};
358    }
359
360    use FSBVec as T;
361    match target {
362        T::Size1(target) => decode_static_size!(dict, target),
363        T::Size2(target) => decode_static_size!(dict, target),
364        T::Size4(target) => decode_static_size!(dict, target),
365        T::Size8(target) => decode_static_size!(dict, target),
366        T::Size12(target) => decode_static_size!(dict, target),
367        T::Size16(target) => decode_static_size!(dict, target),
368        T::Size32(target) => decode_static_size!(dict, target),
369        T::Other(target, _) => {
370            // @NOTE: All these kernels are quite slow, but they should be very uncommon and the
371            // general case requires arbitrary length memcopies anyway.
372
373            let dict = dict.values().as_slice();
374
375            if is_optional {
376                append_validity(
377                    page_validity,
378                    filter.as_ref(),
379                    validity,
380                    values.len() / size,
381                );
382            }
383
384            let page_validity =
385                constrain_page_validity(values.len() / size, page_validity, filter.as_ref());
386
387            let mut indexes = Vec::with_capacity(values.len());
388
389            for chunk in values.into_chunk_iter() {
390                match chunk? {
391                    HybridRleChunk::Rle(value, repeats) => {
392                        indexes.resize(indexes.len() + repeats, value)
393                    },
394                    HybridRleChunk::Bitpacked(decoder) => decoder.collect_into(&mut indexes),
395                }
396            }
397
398            match (page_validity, filter.as_ref()) {
399                (None, None) => target.extend(
400                    indexes
401                        .into_iter()
402                        .flat_map(|v| &dict[(v as usize) * size..][..size]),
403                ),
404                (None, Some(filter)) => match filter {
405                    Filter::Range(range) => target.extend(
406                        indexes[range.start..range.end]
407                            .iter()
408                            .flat_map(|v| &dict[(*v as usize) * size..][..size]),
409                    ),
410                    Filter::Mask(bitmap) => {
411                        let mut iter = bitmap.iter();
412                        let mut offset = 0;
413
414                        while iter.num_remaining() > 0 {
415                            let num_selected = iter.take_leading_ones();
416                            target.extend(
417                                indexes[offset..][..num_selected]
418                                    .iter()
419                                    .flat_map(|v| &dict[(*v as usize) * size..][..size]),
420                            );
421                            offset += num_selected;
422
423                            let num_filtered = iter.take_leading_zeros();
424                            offset += num_filtered;
425                        }
426                    },
427                    Filter::Predicate(_) => todo!(),
428                },
429                (Some(validity), None) => {
430                    let mut iter = validity.iter();
431                    let mut offset = 0;
432
433                    while iter.num_remaining() > 0 {
434                        let num_valid = iter.take_leading_ones();
435                        target.extend(
436                            indexes[offset..][..num_valid]
437                                .iter()
438                                .flat_map(|v| &dict[(*v as usize) * size..][..size]),
439                        );
440                        offset += num_valid;
441
442                        let num_filtered = iter.take_leading_zeros();
443                        target.resize(target.len() + num_filtered * size, 0);
444                    }
445                },
446                (Some(validity), Some(filter)) => match filter {
447                    Filter::Range(range) => {
448                        let (skipped, active) = validity.split_at(range.start);
449
450                        let active = active.sliced(0, range.len());
451
452                        let mut iter = active.iter();
453                        let mut offset = skipped.set_bits();
454
455                        while iter.num_remaining() > 0 {
456                            let num_valid = iter.take_leading_ones();
457                            target.extend(
458                                indexes[offset..][..num_valid]
459                                    .iter()
460                                    .flat_map(|v| &dict[(*v as usize) * size..][..size]),
461                            );
462                            offset += num_valid;
463
464                            let num_filtered = iter.take_leading_zeros();
465                            target.resize(target.len() + num_filtered * size, 0);
466                        }
467                    },
468                    Filter::Mask(filter) => {
469                        let mut offset = 0;
470                        for (is_selected, is_valid) in filter.iter().zip(validity.iter()) {
471                            if is_selected {
472                                if is_valid {
473                                    target.extend_from_slice(
474                                        &dict[(indexes[offset] as usize) * size..][..size],
475                                    );
476                                } else {
477                                    target.resize(target.len() + size, 0);
478                                }
479                            }
480
481                            offset += usize::from(is_valid);
482                        }
483                    },
484                    Filter::Predicate(_) => todo!(),
485                },
486            }
487
488            Ok(())
489        },
490    }

493impl Decoder for BinaryDecoder {
494    type Translation<'a> = StateTranslation<'a>;
495    type Dict = FixedSizeBinaryArray;
496    type DecodedState = (FSBVec, BitmapBuilder);
497    type Output = FixedSizeBinaryArray;
498
499    fn with_capacity(&self, capacity: usize) -> Self::DecodedState {
500        let size = self.size;
501
502        let values = match size {
503            1 => FSBVec::Size1(Vec::with_capacity(capacity)),
504            2 => FSBVec::Size2(Vec::with_capacity(capacity)),
505            4 => FSBVec::Size4(Vec::with_capacity(capacity)),
506            8 => FSBVec::Size8(Vec::with_capacity(capacity)),
507            12 => FSBVec::Size12(Vec::with_capacity(capacity)),
508            16 => FSBVec::Size16(Vec::with_capacity(capacity)),
509            32 => FSBVec::Size32(Vec::with_capacity(capacity)),
510            _ => FSBVec::Other(Vec::with_capacity(capacity * size), size),
511        };
512
513        (values, BitmapBuilder::with_capacity(capacity))
514    }
515
516    fn deserialize_dict(&mut self, page: DictPage) -> ParquetResult<Self::Dict> {
517        let mut target = FSBVec::new(self.size);
518        decode_fsb_plain(
519            self.size,
520            page.buffer.as_ref(),
521            &mut target,
522            &mut BitmapBuilder::new(),
523            &mut BitmapBuilder::new(),
524            false,
525            None,
526            None,
527        )?;
528
529        Ok(FixedSizeBinaryArray::new(
530            ArrowDataType::FixedSizeBinary(self.size),
531            target.into_bytes_buffer(),
532            None,
533        ))
534    }
535
536    fn has_predicate_specialization(
537        &self,
538        _state: &utils::State<'_, Self>,
539        _predicate: &PredicateFilter,
540    ) -> ParquetResult<bool> {
541        // @TODO: This can be enabled for the fast paths
542        Ok(false)
543    }
544
545    fn extend_decoded(
546        &self,
547        decoded: &mut Self::DecodedState,
548        additional: &dyn arrow::array::Array,
549        is_optional: bool,
550    ) -> ParquetResult<()> {
551        let additional = additional
552            .as_any()
553            .downcast_ref::<FixedSizeBinaryArray>()
554            .unwrap();
555        decoded
556            .0
557            .extend_from_byte_slice(additional.values().as_slice());
558        match additional.validity() {
559            Some(v) => decoded.1.extend_from_bitmap(v),
560            None if is_optional => decoded.1.extend_constant(additional.len(), true),
561            None => {},
562        }
563
564        Ok(())
565    }
566
567    fn finalize(
568        &self,
569        dtype: ArrowDataType,
570        _dict: Option<Self::Dict>,
571        (values, validity): Self::DecodedState,
572    ) -> ParquetResult<Self::Output> {
573        let validity = freeze_validity(validity);
574
575        Ok(FixedSizeBinaryArray::new(
576            dtype,
577            values.into_bytes_buffer(),
578            validity,
579        ))
580    }
581
582    fn extend_filtered_with_state(
583        &mut self,
584        state: utils::State<'_, Self>,
585        decoded: &mut Self::DecodedState,
586        pred_true_mask: &mut BitmapBuilder,
587        filter: Option<Filter>,
588    ) -> ParquetResult<()> {
589        match state.translation {
590            StateTranslation::Plain(values, size) => decode_fsb_plain(
591                size,
592                values,
593                &mut decoded.0,
594                pred_true_mask,
595                &mut decoded.1,
596                state.is_optional,
597                filter,
598                state.page_validity.as_ref(),
599            ),
600            StateTranslation::Dictionary(values) => decode_fsb_dict(
601                self.size,
602                values,
603                state.dict.unwrap(),
604                state.dict_mask,
605                &mut decoded.0,
606                pred_true_mask,
607                &mut decoded.1,
608                state.is_optional,
609                filter,
610                state.page_validity.as_ref(),
611            ),
612        }
613    }
