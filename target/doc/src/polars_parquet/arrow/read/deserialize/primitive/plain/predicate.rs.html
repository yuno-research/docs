6#[inline(never)]
7pub fn decode_equals_no_values<B: AlignedBytes>(
8    values: ArrayChunks<'_, B>,
9    needle: B,
10    pred_true_mask: &mut BitmapBuilder,
11) {
12    pred_true_mask.reserve(values.len());
13    for &v in values {
14        let is_pred_true = B::from_unaligned(v) == needle;
15
16        // SAFETY: We reserved enough before the loop.
17        unsafe {
18            pred_true_mask.push_unchecked(is_pred_true);
19        }
20    }
21}

40#[inline(never)]
41pub fn decode_is_in<B: AlignedBytes>(
42    values: ArrayChunks<'_, B>,
43    needle: &[B; 8],
44    target: &mut Vec<B>,
45    pred_true_mask: &mut BitmapBuilder,
46) {
47    target.reserve(values.len());
48    pred_true_mask.reserve(values.len());
49    let mut next_idx = target.len();
50    for &v in values {
51        let v = B::from_unaligned(v);
52        let is_pred_true = needle.contains(&v);
53
54        // SAFETY: We reserved enough before the loop.
55        unsafe {
56            target.as_mut_ptr().add(next_idx).write(v);
57            pred_true_mask.push_unchecked(is_pred_true);
58        }
59
60        next_idx += usize::from(is_pred_true);
61    }
62    unsafe {
63        target.set_len(next_idx);
64    }
