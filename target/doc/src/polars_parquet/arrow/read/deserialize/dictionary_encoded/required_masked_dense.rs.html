9#[inline(never)]
10pub fn decode<B: AlignedBytes, D: IndexMapping<Output = B>>(
11    mut values: HybridRleDecoder<'_>,
12    dict: D,
13    mut filter: Bitmap,
14    target: &mut Vec<B>,
15) -> ParquetResult<()> {
16    assert!(values.len() >= filter.len());
17
18    let mut num_rows_to_skip = filter.take_leading_zeros();
19    filter.take_trailing_zeros();
20
21    let num_rows = filter.set_bits();
22
23    values.limit_to(num_rows_to_skip + filter.len());
24
25    // Dispatch to the non-filter kernel if all rows are needed anyway.
26    if num_rows == filter.len() {
27        return super::required::decode(values, dict, target, num_rows_to_skip);
28    }
29
30    if dict.is_empty() && !filter.is_empty() {
31        return Err(oob_dict_idx());
32    }
33
34    target.reserve(num_rows);
35
36    let mut filter = BitMask::from_bitmap(&filter);
37
38    let mut values_buffer = [0u32; 128];
39    let values_buffer = &mut values_buffer;
40
41    // Skip over whole HybridRleChunks
42    required_skip_whole_chunks(&mut values, &mut num_rows_to_skip)?;
43
44    while let Some(chunk) = values.next_chunk()? {
45        debug_assert!(num_rows_to_skip < chunk.len() || chunk.len() == 0);
46
47        match chunk {
48            HybridRleChunk::Rle(value, size) => {
49                if size == 0 {
50                    continue;
51                }
52
53                // If we know that we have `size` times `value` that we can append, but there might
54                // be nulls in between those values.
55                //
56                // 1. See how many `num_rows = valid + invalid` values `size` would entail. This is
57                //    done with `num_bits_before_nth_one` on the validity mask.
58                // 2. Fill `num_rows` values into the target buffer.
59                // 3. Advance the validity mask by `num_rows` values.
60
61                let current_filter;
62
63                (current_filter, filter) = filter.split_at(size - num_rows_to_skip);
64                let num_chunk_rows = current_filter.set_bits();
65
66                if num_chunk_rows > 0 {
67                    let Some(value) = dict.get(value) else {
68                        return Err(oob_dict_idx());
69                    };
70
71                    target.resize(target.len() + num_chunk_rows, value);
72                }
73            },
74            HybridRleChunk::Bitpacked(mut decoder) => {
75                let size = decoder.len();
76                let mut chunked = decoder.chunked();
77
78                let mut buffer_part_idx = 0;
79                let mut values_offset = 0;
80                let mut num_buffered: usize = 0;
81                let mut skip_values = num_rows_to_skip;
82
83                let current_filter;
84
85                (current_filter, filter) = filter.split_at(size - num_rows_to_skip);
86
87                let mut iter = |mut f: u64, len: usize| {
88                    debug_assert!(len <= 64);
89
90                    // Skip chunk if we don't any values from here.
91                    if f == 0 {
92                        skip_values += len;
93                        return ParquetResult::Ok(());
94                    }
95
96                    // Skip over already buffered items.
97                    let num_buffered_skipped = skip_values.min(num_buffered);
98                    values_offset += num_buffered_skipped;
99                    num_buffered -= num_buffered_skipped;
100                    skip_values -= num_buffered_skipped;
101
102                    // If we skipped plenty already, just skip decoding those chunks instead of
103                    // decoding them and throwing them away.
104                    chunked.decoder.skip_chunks(skip_values / 32);
105                    // The leftovers we have to decode but we can also just skip.
106                    skip_values %= 32;
107
108                    while num_buffered < len {
109                        let buffer_part = <&mut [u32; 32]>::try_from(
110                            &mut values_buffer[buffer_part_idx * 32..][..32],
111                        )
112                        .unwrap();
113                        let num_added = chunked.next_into(buffer_part).unwrap();
114
115                        verify_dict_indices(buffer_part, dict.len())?;
116
117                        let skip_chunk_values = skip_values.min(num_added);
118
119                        values_offset += skip_chunk_values;
120                        num_buffered += num_added - skip_chunk_values;
121                        skip_values -= skip_chunk_values;
122
123                        buffer_part_idx += 1;
124                        buffer_part_idx %= 4;
125                    }
126
127                    let mut num_read = 0;
128                    let mut num_written = 0;
129                    let target_ptr = unsafe { target.as_mut_ptr().add(target.len()) };
130
131                    while f != 0 {
132                        let offset = f.trailing_zeros() as usize;
133
134                        num_read += offset;
135
136                        let idx = values_buffer[(values_offset + num_read) % 128];
137                        // SAFETY:
138                        // 1. `values_buffer` starts out as only zeros, which we know is in the
139                        //    dictionary following the original `dict.is_empty` check.
140                        // 2. Each time we write to `values_buffer`, it is followed by a
141                        //    `verify_dict_indices`.
142                        let value = unsafe { dict.get_unchecked(idx) };
143                        unsafe { target_ptr.add(num_written).write(value) };
144
145                        num_written += 1;
146                        num_read += 1;
147
148                        f >>= offset + 1; // Clear least significant bit.
149                    }
150
151                    values_offset += len;
152                    values_offset %= 128;
153                    num_buffered -= len;
154                    unsafe {
155                        target.set_len(target.len() + num_written);
156                    }
157
158                    ParquetResult::Ok(())
159                };
160
161                let mut f_iter = current_filter.fast_iter_u56();
162
163                for f in f_iter.by_ref() {
164                    iter(f, 56)?;
165                }
166
167                let (f, fl) = f_iter.remainder();
168                iter(f, fl)?;
169            },
170        }
171
172        num_rows_to_skip = 0;
173    }
174
175    Ok(())
176}

189    fn values_and_mask() -> impl Strategy<Value = (Vec<u8>, u32, Vec<u32>, Bitmap)> {
190        (bitmap(1..100), (0..300u32)).prop_flat_map(|(mask, max_idx)| {
191            let len = mask.len();
192            let hybrid_rle = hybrid_rle::proptest::hybrid_rle(max_idx, len);
193
194            (
195                hybrid_rle,
196                Just(max_idx),
197                any_with::<Vec<u32>>(size_range((max_idx + 1) as usize).lift()),
198                Just(mask),
199            )
200        })
201    }
202
203    fn _test_decode(
204        hybrid_rle: HybridRleDecoder<'_>,
205        dict: &[Bytes4Alignment4],
206        mask: &Bitmap,
207    ) -> TestCaseResult {
208        let mut result = Vec::<arrow::types::Bytes4Alignment4>::with_capacity(mask.set_bits());
209        decode(hybrid_rle.clone(), dict, mask.clone(), &mut result).unwrap();
210
211        let idxs = hybrid_rle.collect().unwrap();
212        let mut result_i = 0;
213        for (idx, is_selected) in idxs.iter().zip(mask.iter()) {
214            if is_selected {
215                prop_assert_eq!(result[result_i], dict[*idx as usize]);
216                result_i += 1;
217            }
218        }
219
220        TestCaseResult::Ok(())
221    }
222
223    proptest! {
224        #[test]
225        fn test_decode_masked_optional(
226            (ref hybrid_rle, max_idx, ref dict, ref mask) in values_and_mask()
227        ) {
228            let hybrid_rle = HybridRleDecoder::new(hybrid_rle, 32 - max_idx.leading_zeros(), mask.len());
229            let dict = bytemuck::cast_slice(dict.as_slice());
230            _test_decode(hybrid_rle, dict, mask)?
231        }
232    }
