31/// The max size of a string we can fit inline
32pub(crate) const MAX_SIZE: usize = core::mem::size_of::<String>();
33/// Used as a discriminant to identify different variants
34pub(crate) const HEAP_MASK: u8 = LastByte::Heap as u8;
35/// Used for `StaticStr` variant
36pub(crate) const STATIC_STR_MASK: u8 = LastByte::Static as u8;
37/// When our string is stored inline, we represent the length of the string in the last byte, offset
38/// by `LENGTH_MASK`
39pub(crate) const LENGTH_MASK: u8 = 0b11000000;

41const EMPTY: Repr = Repr::const_new("");

43#[repr(C)]
44pub(crate) struct Repr(
45    /// We have a pointer in the representation to properly carry provenance.
46    *const (),
47    /// Then we need two `usize`s (aka WORDs) of data, for the first we just define a `usize`...
48    usize,
49    /// ...but the second we breakup into multiple pieces...
50    #[cfg(target_pointer_width = "64")]
51    u32,
52    u16,
53    u8,
54    /// ...so that the last byte can be a [`LastByte`], which allows the compiler to see a niche
55    /// value.
56    LastByte,
57);
58static_assertions::assert_eq_size!([u8; MAX_SIZE], Repr);
59
60unsafe impl Send for Repr {}
61unsafe impl Sync for Repr {}

64    #[inline]
65    pub(crate) fn new(text: &str) -> Result<Self, ReserveError> {
66        let len = text.len();
67
68        if len == 0 {
69            Ok(EMPTY)
70        } else if len <= MAX_SIZE {
71            // SAFETY: We checked that the length of text is less than or equal to MAX_SIZE
72            let inline = unsafe { InlineBuffer::new(text) };
73            Ok(Repr::from_inline(inline))
74        } else {
75            HeapBuffer::new(text).map(Repr::from_heap)
76        }
77    }
78
79    #[inline]
80    pub(crate) const fn const_new(text: &'static str) -> Self {
81        if text.len() <= MAX_SIZE {
82            let inline = InlineBuffer::new_const(text);
83            Repr::from_inline(inline)
84        } else {
85            let repr = StaticStr::new(text);
86            Repr::from_static(repr)
87        }
88    }

90    /// Create a [`Repr`] with the provided `capacity`
91    #[inline]
92    pub(crate) fn with_capacity(capacity: usize) -> Result<Self, ReserveError> {
93        if capacity <= MAX_SIZE {
94            Ok(EMPTY)
95        } else {
96            HeapBuffer::with_capacity(capacity).map(Repr::from_heap)
97        }
98    }

102    pub(crate) fn from_utf8<B: AsRef<[u8]>>(buf: B) -> Result<Self, Utf8Error> {
103        // Get a &str from the Vec, failing if it's not valid UTF-8
104        let s = core::str::from_utf8(buf.as_ref())?;
105        // Construct a Repr from the &str
106        Ok(Self::new(s).unwrap_with_msg())

109    /// Create a [`Repr`] from a slice of bytes that is UTF-8, without validating that it is indeed
110    /// UTF-8
111    ///
112    /// # Safety
113    ///
114    /// * The caller must guarantee that `buf` is valid UTF-8.
115    #[inline]
116    pub(crate) unsafe fn from_utf8_unchecked<B: AsRef<[u8]>>(buf: B) -> Result<Self, ReserveError> {
117        let bytes = buf.as_ref();
118        let bytes_len = bytes.len();
119
120        // Create a Repr with enough capacity for the entire buffer
121        let mut repr = Repr::with_capacity(bytes_len)?;
122
123        // SAFETY: The caller is responsible for making sure the provided buffer is UTF-8. This
124        // invariant is documented in the public API
125        let slice = repr.as_mut_buf();
126        // write the chunk into the Repr
127        slice[..bytes_len].copy_from_slice(bytes);
128
129        // Set the length of the Repr
130        // SAFETY: We just wrote the entire `buf` into the Repr
131        repr.set_len(bytes_len);
132
133        Ok(repr)
134    }

136    /// Create a [`Repr`] from a [`String`], in `O(1)` time. We'll attempt to inline the string
137    /// if `should_inline` is `true`
138    ///
139    /// Note: If the provided [`String`] is >16 MB and we're on a 32-bit arch, we'll copy the
140    /// `String`.
141    #[inline]
142    pub(crate) fn from_string(s: String, should_inline: bool) -> Result<Self, ReserveError> {
143        let og_cap = s.capacity();
144        let cap = Capacity::new(og_cap);
145
146        #[cold]
147        fn capacity_on_heap(s: String) -> Result<Repr, ReserveError> {
148            HeapBuffer::new(s.as_str()).map(Repr::from_heap)
149        }
150
151        #[cold]
152        fn empty() -> Result<Repr, ReserveError> {
153            Ok(EMPTY)
154        }
155
156        if cap.is_heap() {
157            // We only hit this case if the provided String is > 16MB and we're on a 32-bit arch. We
158            // expect it to be unlikely, thus we hint that to the compiler
159            capacity_on_heap(s)
160        } else if og_cap == 0 {
161            // We don't expect converting from an empty String often, so we make this code path cold
162            empty()
163        } else if should_inline && s.len() <= MAX_SIZE {
164            // SAFETY: Checked to make sure the string would fit inline
165            let inline = unsafe { InlineBuffer::new(s.as_str()) };
166            Ok(Repr::from_inline(inline))
167        } else {
168            let mut s = mem::ManuallyDrop::new(s.into_bytes());
169            let len = s.len();
170            let raw_ptr = s.as_mut_ptr();
171
172            let ptr = ptr::NonNull::new(raw_ptr).expect("string with capacity has null ptr?");
173            let heap = HeapBuffer { ptr, len, cap };
174
175            Ok(Repr::from_heap(heap))
176        }
177    }

179    /// Converts a [`Repr`] into a [`String`], in `O(1)` time, if possible
180    #[inline]
181    pub(crate) fn into_string(self) -> String {
182        #[cold]
183        fn into_string_heap(this: HeapBuffer) -> String {
184            // SAFETY: We know pointer is valid for `length` bytes
185            let slice = unsafe { core::slice::from_raw_parts(this.ptr.as_ptr(), this.len) };
186            // SAFETY: A `Repr` contains valid UTF-8
187            let s = unsafe { core::str::from_utf8_unchecked(slice) };
188
189            String::from(s)
190        }
191
192        if self.is_heap_allocated() {
193            // SAFETY: we just checked that the discriminant indicates we're a HeapBuffer
194            let heap_buffer = unsafe { self.into_heap() };
195
196            if heap_buffer.cap.is_heap() {
197                // We don't expect capacity to be on the heap often, so we mark it as cold
198                into_string_heap(heap_buffer)
199            } else {
200                // Wrap the BoxString in a ManuallyDrop so the underlying buffer doesn't get freed
201                let this = mem::ManuallyDrop::new(heap_buffer);
202
203                // SAFETY: We checked above to make sure capacity is valid
204                let cap = unsafe { this.cap.as_usize() };
205
206                // SAFETY:
207                // * The memory in `ptr` was previously allocated by the same allocator the standard
208                //   library uses, with a required alignment of exactly 1.
209                // * `length` is less than or equal to capacity, due to internal invaraints.
210                // * `capacity` is correctly maintained internally.
211                // * `BoxString` only ever contains valid UTF-8.
212                unsafe { String::from_raw_parts(this.ptr.as_ptr(), this.len, cap) }
213            }
214        } else {
215            String::from(self.as_str())
216        }
217    }

219    /// Reserves at least `additional` bytes. If there is already enough capacity to store
220    /// `additional` bytes this is a no-op
221    #[inline]
222    pub(crate) fn reserve(&mut self, additional: usize) -> Result<(), ReserveError> {
223        let len = self.len();
224        let needed_capacity = len.checked_add(additional).ok_or(ReserveError(()))?;
225
226        if !self.is_static_str() && needed_capacity <= self.capacity() {
227            // we already have enough space, no-op
228            // If self.is_static_str() is true, then we would have to convert
229            // it to other variants since static_str variant cannot be modified.
230            Ok(())
231        } else if needed_capacity <= MAX_SIZE {
232            // It's possible to have a `Repr` that is heap allocated with a capacity less than
233            // MAX_SIZE, if that `Repr` was created From a String or Box<str>
234            //
235            // SAFETY: Our needed_capacity is >= our length, which is <= than MAX_SIZE
236            let inline = unsafe { InlineBuffer::new(self.as_str()) };
237            *self = Repr::from_inline(inline);
238            Ok(())
239        } else if !self.is_heap_allocated() {
240            // We're not heap allocated, but need to be, create a HeapBuffer
241            let heap = HeapBuffer::with_additional(self.as_str(), additional)?;
242            *self = Repr::from_heap(heap);
243            Ok(())
244        } else {
245            // We're already heap allocated, but we need more capacity
246            //
247            // SAFETY: We checked above to see if we're heap allocated
248            let heap_buffer = unsafe { self.as_mut_heap() };
249
250            // To reduce allocations, we amortize our growth
251            let amortized_capacity = heap::amortized_growth(len, additional);
252            // Attempt to grow our capacity, allocating a new HeapBuffer on failure
253            if heap_buffer.realloc(amortized_capacity).is_err() {
254                // Create a new HeapBuffer
255                let heap = HeapBuffer::with_additional(self.as_str(), additional)?;
256                *self = Repr::from_heap(heap);
257            }
258
259            Ok(())
260        }
261    }

263    pub(crate) fn shrink_to(&mut self, min_capacity: usize) {
264        // Note: We can't shrink the inline variant since it's buffer is a fixed size
265        // or the static str variant since it is just a pointer, so we only
266        // take action here if our string is heap allocated
267        if !self.is_heap_allocated() {
268            return;
269        }
270
271        // SAFETY: We just checked the discriminant to make sure we're heap allocated
272        let heap = unsafe { self.as_mut_heap() };
273
274        let old_capacity = heap.capacity();
275        let new_capacity = heap.len.max(min_capacity);
276
277        if new_capacity <= MAX_SIZE {
278            // Inline string if possible.
279
280            let mut inline = InlineBuffer::empty();
281            // SAFETY: Our src is on the heap, so it does not overlap with our new inline
282            // buffer, and the src is a `Repr` so we can assume it's valid UTF-8
283            unsafe {
284                inline
285                    .0
286                    .as_mut_ptr()
287                    .copy_from_nonoverlapping(heap.ptr.as_ptr(), heap.len)
288            };
289            // SAFETY: The src we wrote from was a `Repr` which we can assume is valid UTF-8
290            unsafe { inline.set_len(heap.len) }
291            *self = Repr::from_inline(inline);
292            return;
293        }
294
295        // Return if the string cannot be strunk.
296        if new_capacity >= old_capacity {
297            return;
298        }
299
300        // Try to shrink in-place.
301        if heap.realloc(new_capacity).is_ok() {
302            return;
303        }
304
305        // Otherwise try to allocate a new, smaller chunk.
306        // We can ignore the error. The string keeps its old capacity, but that's okay.
307        if let Ok(mut new_this) = Repr::with_capacity(new_capacity) {
308            new_this.push_str(self.as_str());
309            *self = new_this;
310        }
311    }

313    #[inline]
314    pub(crate) fn push_str(&mut self, s: &str) {
315        // If `s` is empty, then there's no reason to reserve or push anything
316        // at all.
317        if s.is_empty() {
318            return;
319        }
320
321        let len = self.len();
322        let str_len = s.len();
323
324        // Reserve at least enough space to fit `s`
325        self.reserve(str_len).unwrap_with_msg();
326
327        // SAFETY: `s` which we're appending to the buffer, is valid UTF-8
328        let slice = unsafe { self.as_mut_buf() };
329        let push_buffer = &mut slice[len..len + str_len];
330
331        debug_assert_eq!(push_buffer.len(), s.len());
332
333        // Copy the string into our buffer
334        push_buffer.copy_from_slice(s.as_bytes());
335
336        // Increment the length of our string
337        //
338        // SAFETY: We appended `s` which is valid UTF-8, and if our size became greater than
339        // MAX_SIZE, our call to reserve would make us heap allocated
340        unsafe { self.set_len(len + str_len) };
341    }

345        let ch = self.as_str().chars().next_back()?;

347        // SAFETY: We know this is is a valid length which falls on a char boundary

353    /// Returns the string content, and only the string content, as a slice of bytes.
354    #[inline]
355    pub(crate) fn as_slice(&self) -> &[u8] {
356        // initially has the value of the stack pointer, conditionally becomes the heap pointer
357        let mut pointer = self as *const Self as *const u8;
358        let heap_pointer = self.0 as *const u8;
359        if self.last_byte() >= HEAP_MASK {
360            pointer = heap_pointer;
361        }
362
363        // initially has the value of the stack length, conditionally becomes the heap length
364        let mut length = core::cmp::min(
365            self.last_byte().wrapping_sub(LENGTH_MASK) as usize,
366            MAX_SIZE,
367        );
368        let heap_length = self.1;
369        if self.last_byte() >= HEAP_MASK {
370            length = heap_length;
371        }
372
373        // SAFETY: We know the data is valid, aligned, and part of the same contiguous allocated
374        // chunk. It's also valid for the lifetime of self
375        unsafe { core::slice::from_raw_parts(pointer, length) }
376    }

384    /// Returns the length of the string that we're storing
385    #[inline]
386    pub(crate) fn len(&self) -> usize {
387        // This ugly looking code results in two conditional moves and only one comparison, without
388        // branching. The outcome of a comparison is a tristate `{lt, eq, gt}`, but the compiler
389        // won't use this optimization if you match on `len_inline.cmp(&MAX_SIZE)`, so we have to
390        // do it manually.
391
392        // Force the compiler to read the variable, so it won't put the reading in a branch.
393        let len_heap = ensure_read(self.1);
394
395        let last_byte = self.last_byte();
396
397        // Extending the variable early results in fewer instructions, because loading and
398        // extending can be done in one instruction.
399        let mut len = (last_byte as usize)
400            .wrapping_sub(LENGTH_MASK as usize)
401            .min(MAX_SIZE);
402
403        // our discriminant is stored in the last byte and denotes stack vs heap
404        //
405        // Note: We should never add an `else` statement here, keeping the conditional simple allows
406        // the compiler to optimize this to a conditional-move instead of a branch
407        if last_byte >= HEAP_MASK {
408            len = len_heap;
409        }
410
411        len
412    }

414    /// Returns `true` if the length is 0, `false` otherwise
415    #[inline]
416    pub(crate) fn is_empty(&self) -> bool {
417        let len_heap = ensure_read(self.1);
418        let last_byte = self.last_byte() as usize;
419        let mut len = last_byte.wrapping_sub(LastByte::L0 as u8 as usize);
420        if last_byte >= LastByte::Heap as u8 as usize {
421            len = len_heap;
422        }
423        len == 0
424    }

427    #[inline]
428    pub(crate) fn capacity(&self) -> usize {
429        #[cold]
430        fn heap_capacity(this: &Repr) -> usize {
431            // SAFETY: We just checked the discriminant to make sure we're heap allocated
432            let heap_buffer = unsafe { this.as_heap() };
433            heap_buffer.capacity()
434        }
435
436        if let Some(s) = self.as_static_str() {
437            s.len()
438        } else if self.is_heap_allocated() {
439            heap_capacity(self)
440        } else {
441            MAX_SIZE
442        }

480    /// Return a mutable reference to the entirely underlying buffer
481    ///
482    /// # Safety
483    /// * Callers must guarantee that any modifications made to the buffer are valid UTF-8
484    pub(crate) unsafe fn as_mut_buf(&mut self) -> &mut [u8] {
485        #[cold]
486        fn inline_static_str(this: &mut Repr) {
487            if let Some(s) = this.as_static_str() {
488                *this = Repr::new(s).unwrap_with_msg();
489            }
490        }
491
492        if self.is_static_str() {
493            inline_static_str(self);
494        }
495
496        // the last byte stores our discriminant and stack length
497        let last_byte = self.last_byte();
498
499        let (ptr, cap) = if last_byte == HEAP_MASK {
500            // SAFETY: We just checked the discriminant to make sure we're heap allocated
501            let heap_buffer = self.as_heap();
502            let ptr = heap_buffer.ptr.as_ptr();
503            let cap = heap_buffer.capacity();
504
505            (ptr, cap)
506        } else {
507            let ptr = self as *mut Self as *mut u8;
508            (ptr, MAX_SIZE)
509        };
510
511        // SAFETY: Our data is valid for `cap` bytes, and is initialized
512        core::slice::from_raw_parts_mut(ptr, cap)
513    }

515    /// Sets the length of the string that our underlying buffer contains
516    ///
517    /// # Safety
518    /// * `len` bytes in the buffer must be valid UTF-8
519    /// * If the underlying buffer is stored inline, `len` must be <= MAX_SIZE
520    pub(crate) unsafe fn set_len(&mut self, len: usize) {
521        if let Some(s) = self.as_static_variant_mut() {
522            s.set_len(len);
523        } else if self.is_heap_allocated() {
524            // SAFETY: We just checked the discriminant to make sure we're heap allocated
525            let heap_buffer = self.as_mut_heap();
526            // SAFETY: The caller guarantees that `len` bytes is valid UTF-8
527            heap_buffer.set_len(len);
528        } else {
529            // SAFETY: We just checked the discriminant to make sure we're an InlineBuffer
530            let inline_buffer = self.as_mut_inline();
531            // SAFETY: The caller guarantees that len <= MAX_SIZE, and `len` bytes is valid UTF-8
532            inline_buffer.set_len(len);
533        }
534    }

536    /// Zero out the memory backing this [`Repr`].
537    #[cfg(feature = "zeroize")]
538    pub(crate) fn zeroize(&mut self) {
539        // We can't zero out static memory so we just replace ourselves with
540        // the EMPTY variant.
541        if self.is_static_str() {
542            *self = EMPTY;
543            return;
544        }
545
546        /// Performs a volatile `memset` operation which fills a slice with a value.
547        ///
548        /// # SAFETY:
549        ///
550        /// * The memory pointed to by `dst` must be valid for `count` contiguous bytes.
551        /// * `count` must not be larger than an isize
552        /// * `dst` + `count` must not wrap around the address space.
553        ///  
554        /// Derived from: <https://github.com/RustCrypto/utils/blob/c68a5204b2e66b0f60832d845e048fca96a81211/zeroize/src/lib.rs#L766-L791>.
555        ///
556        /// TODO(parkmycar): use `volatile_set_memory` when stabilized
557        #[inline(always)]
558        unsafe fn volatile_zero(dst: *mut u8, count: usize) {
559            for i in 0..count {
560                let dst = dst.add(i);
561                ptr::write_volatile(dst, 0);
562            }
563        }
564
565        /// Uses fences to prevent the compiler from re-ordering memory accesses.
566        #[inline(always)]
567        fn atomic_fence() {
568            use core::sync::atomic;
569            atomic::compiler_fence(atomic::Ordering::SeqCst);
570        }
571
572        // The last byte stores our discriminant and stack length.
573        let last_byte = self.last_byte();
574
575        let (ptr, cap) = if last_byte == HEAP_MASK {
576            // SAFETY: We just checked the discriminant to make sure we're heap allocated.
577            let heap_buffer = unsafe { self.as_mut_heap() };
578            // SAFTEY: Setting the length to 0 is always safe because the empty string is
579            // valid UTF-8.
580            unsafe { heap_buffer.set_len(0) };
581
582            let ptr = heap_buffer.ptr.as_ptr();
583            let cap = heap_buffer.capacity();
584            (ptr, cap)
585        } else {
586            // SAFETY: We just checked the discriminant above to see if we're heap allocated.
587            let inline_buffer = unsafe { self.as_mut_inline() };
588            // SAFTEY: Setting the length to 0 is always safe because the empty string is
589            // valid UTF-8.
590            unsafe { inline_buffer.set_len(0) };
591
592            let ptr = self as *mut Self as *mut u8;
593            let cap = MAX_SIZE - 1;
594            (ptr, cap)
595        };
596
597        // SAFTEY: We know our pointer is valid for `cap` bytes because the capacity came
598        // from an already existing CompactString. Also we don't allow allocations larger
599        // then an isize.
600        unsafe { volatile_zero(ptr, cap) };
601        atomic_fence()
602    }

604    /// Returns the last byte that's on the stack.
605    ///
606    /// The last byte stores the discriminant that indicates whether the string is on the stack or
607    /// on the heap. When the string is on the stack the last byte also stores the length
608    #[inline(always)]
609    const fn last_byte(&self) -> u8 {
610        cfg_if::cfg_if! {
611            if #[cfg(target_pointer_width = "64")] {
612                let last_byte = self.5;
613            } else if #[cfg(target_pointer_width = "32")] {
614                let last_byte = self.4;
615            } else {
616                compile_error!("Unsupported target_pointer_width");
617            }
618        };
619        last_byte as u8
620    }

622    /// Reinterprets an [`InlineBuffer`] into a [`Repr`]
623    ///
624    /// Note: This is safe because [`InlineBuffer`] and [`Repr`] are the same size. We used to

733impl Clone for Repr {
734    #[inline]
735    fn clone(&self) -> Self {
736        #[inline(never)]
737        fn clone_heap(this: &Repr) -> Repr {
738            Repr::new(this.as_str()).unwrap_with_msg()
739        }
740
741        // There are only two cases we need to care about: If the string is allocated on the heap
742        // or not. If it is, then the data must be cloned properly, otherwise we can simply copy
743        // the `Repr`.
744        if self.is_heap_allocated() {
745            clone_heap(self)
746        } else {
747            // SAFETY: We just checked that `self` can be copied because it is an inline string or
748            // a reference to a `&'static str`.
749            unsafe { core::ptr::read(self) }
750        }
751    }
752
753    #[inline]
754    fn clone_from(&mut self, source: &Self) {
755        #[inline(never)]
756        fn clone_from_heap(this: &mut Repr, source: &Repr) {
757            unsafe { this.set_len(0) };
758            this.push_str(source.as_str());
759        }
760
761        // There are only two cases we need to care about: If the string is allocated on the heap
762        // or not. If it is, then the data must be cloned proberly, otherwise we can simply copy
763        // the `Repr`.
764        if source.is_heap_allocated() {
765            clone_from_heap(self, source)
766        } else {
767            // SAFETY: We just checked that `source` can be copied because it is an inline string or
768            // a reference to a `&'static str`.
769            *self = unsafe { core::ptr::read(source) }
770        }
771    }
772}

774impl Drop for Repr {
775    #[inline]
776    fn drop(&mut self) {
777        // By "outlining" the actual Drop code and only calling it if we're a heap variant, it
778        // allows dropping an inline variant to be as cheap as possible.
779        if self.is_heap_allocated() {
780            outlined_drop(self)
781        }
782
783        #[cold]
784        fn outlined_drop(this: &mut Repr) {
785            // SAFETY: We just checked the discriminant to make sure we're heap allocated
786            let heap_buffer = unsafe { this.as_mut_heap() };
787            heap_buffer.dealloc();
788        }
789    }
790}

792impl Extend<char> for Repr {
793    #[inline]
794    fn extend<T: IntoIterator<Item = char>>(&mut self, iter: T) {
795        let iter = iter.into_iter();
796
797        let (lower_bound, _) = iter.size_hint();
798        if lower_bound > 0 {
799            // Ignore the error and hope that the lower_bound is incorrect.
800            let _: Result<(), ReserveError> = self.reserve(lower_bound);
801        }
802
803        for c in iter {
804            self.push_str(c.encode_utf8(&mut [0; 4]));
805        }
806    }
807}

839/// Returns the supplied value, and ensures that the value is eagerly loaded into a register.
840#[inline(always)]
841fn ensure_read(value: usize) -> usize {
842    // SAFETY: This assembly instruction is a noop that only affects the instruction ordering.
843    //
844    // TODO(parkmycar): Re-add loongarch and riscv once we have CI coverage for them.
845    #[cfg(all(
846        not(miri),
847        any(
848            target_arch = "x86",
849            target_arch = "x86_64",
850            target_arch = "arm",
851            target_arch = "aarch64",
852        )
853    ))]
854    unsafe {
855        core::arch::asm!(
856            "/* {value} */",
857            value = in(reg) value,
858            options(nomem, nostack),
859        );
860    };
861
862    value
863}
