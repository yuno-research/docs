11    fn compute_pow5_128(q: i32) -> (u64, u64) {
12        let mut c = if q < 0 {
13            let pow5 = BigUint::from(5_u8).pow((-q) as u32);
14            let mut z = 0_u16;
15            while (BigUint::from(1_u8) << z) < pow5 {
16                z += 1;
17            }
18            let b = if q < -27 {
19                2 * z + 128
20            } else {
21                z + 127
22            };
23            (BigUint::from(1_u8) << b) / pow5 + BigUint::from(1_u8)
24        } else {
25            BigUint::from(5_u8).pow(q as u32)
26        };
27        while c < (BigUint::from(1_u8) << 127) {
28            c <<= 1;
29        }
30        while c >= (BigUint::from(1_u8) << 128) {
31            c >>= 1;
32        }
33        let mut digits = c.to_u32_digits();
34        while digits.len() < 4 {
35            digits.push(0);
36        }
37        assert_eq!(digits.len(), 4);
38        let lo = (digits[0] as u64) + (digits[1] as u64 * (1_u64 << 32));
39        let hi = (digits[2] as u64) + (digits[3] as u64 * (1_u64 << 32));
40        (hi, lo)
41    }
42
43    #[test]
44    #[cfg_attr(miri, ignore)]
45    fn test_pow5_table() {
46        for q in SMALLEST_POWER_OF_FIVE..=LARGEST_POWER_OF_FIVE {
47            let (hi, lo) = compute_pow5_128(q);
48            let expected = POWER_OF_FIVE_128[(q - SMALLEST_POWER_OF_FIVE) as usize];
49            assert_eq!(hi, expected.0);
50            assert_eq!(lo, expected.1);
51        }
52    }
53}
