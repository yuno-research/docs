51impl Decimal {
52    pub const MAX_DIGITS: usize = 768;
53    pub const MAX_DIGITS_WITHOUT_OVERFLOW: usize = 19;
54    pub const DECIMAL_POINT_RANGE: i32 = 2047;
55
56    #[inline]
57    pub fn try_add_digit(&mut self, digit: u8) {
58        if self.num_digits < Self::MAX_DIGITS {
59            self.digits[self.num_digits] = digit;
60        }
61        self.num_digits += 1;
62    }
63
64    #[inline]
65    pub fn trim(&mut self) {
66        while self.num_digits != 0 && self.digits[self.num_digits - 1] == 0 {
67            self.num_digits -= 1;
68        }
69    }
70
71    #[inline]
72    pub fn round(&self) -> u64 {
73        if self.num_digits == 0 || self.decimal_point < 0 {
74            return 0;
75        } else if self.decimal_point > 18 {
76            return 0xFFFF_FFFF_FFFF_FFFF_u64;
77        }
78        let dp = self.decimal_point as usize;
79        let mut n = 0_u64;
80        for i in 0..dp {
81            n *= 10;
82            if i < self.num_digits {
83                n += self.digits[i] as u64;
84            }
85        }
86        let mut round_up = false;
87        if dp < self.num_digits {
88            round_up = self.digits[dp] >= 5;
89            if self.digits[dp] == 5 && dp + 1 == self.num_digits {
90                round_up = self.truncated || ((dp != 0) && (1 & self.digits[dp - 1] != 0));
91            }
92        }
93        if round_up {
94            n += 1;
95        }
96        n
97    }
98
99    #[inline]
100    pub fn left_shift(&mut self, shift: usize) {
101        if self.num_digits == 0 {
102            return;
103        }
104        let num_new_digits = number_of_digits_decimal_left_shift(self, shift);
105        let mut read_index = self.num_digits;
106        let mut write_index = self.num_digits + num_new_digits;
107        let mut n = 0_u64;
108        while read_index != 0 {
109            read_index -= 1;
110            write_index -= 1;
111            n += (self.digits[read_index] as u64) << shift;
112            let quotient = n / 10;
113            let remainder = n - (10 * quotient);
114            if write_index < Self::MAX_DIGITS {
115                self.digits[write_index] = remainder as u8;
116            } else if remainder > 0 {
117                self.truncated = true;
118            }
119            n = quotient;
120        }
121        while n > 0 {
122            write_index -= 1;
123            let quotient = n / 10;
124            let remainder = n - (10 * quotient);
125            if write_index < Self::MAX_DIGITS {
126                self.digits[write_index] = remainder as u8;
127            } else if remainder > 0 {
128                self.truncated = true;
129            }
130            n = quotient;
131        }
132        self.num_digits += num_new_digits;
133        if self.num_digits > Self::MAX_DIGITS {
134            self.num_digits = Self::MAX_DIGITS;
135        }
136        self.decimal_point += num_new_digits as i32;
137        self.trim();
138    }
139
140    #[inline]
141    pub fn right_shift(&mut self, shift: usize) {
142        let mut read_index = 0;
143        let mut write_index = 0;
144        let mut n = 0_u64;
145        while (n >> shift) == 0 {
146            if read_index < self.num_digits {
147                n = (10 * n) + self.digits[read_index] as u64;
148                read_index += 1;
149            } else if n == 0 {
150                return;
151            } else {
152                while (n >> shift) == 0 {
153                    n *= 10;
154                    read_index += 1;
155                }
156                break;
157            }
158        }
159        self.decimal_point -= read_index as i32 - 1;
160        if self.decimal_point < -Self::DECIMAL_POINT_RANGE {
161            self.num_digits = 0;
162            self.decimal_point = 0;
163            self.negative = false;
164            self.truncated = false;
165            return;
166        }
167        let mask = (1_u64 << shift) - 1;
168        while read_index < self.num_digits {
169            let new_digit = (n >> shift) as u8;
170            n = (10 * (n & mask)) + self.digits[read_index] as u64;
171            read_index += 1;
172            self.digits[write_index] = new_digit;
173            write_index += 1;
174        }
175        while n > 0 {
176            let new_digit = (n >> shift) as u8;
177            n = 10 * (n & mask);
178            if write_index < Self::MAX_DIGITS {
179                self.digits[write_index] = new_digit;
180                write_index += 1;
181            } else if new_digit > 0 {
182                self.truncated = true;
183            }
184        }
185        self.num_digits = write_index;
186        self.trim();
187    }
188}

190#[inline]
191pub fn parse_decimal(mut s: &[u8]) -> Decimal {
192    // can't fail since it follows a call to parse_number
193    assert!(!s.is_empty(), "the buffer cannot be empty since it follows a call to parse_number");
194    let mut d = Decimal::default();
195    let start = s;
196
197    let c = s[0];
198    d.negative = c == b'-';
199    if c == b'-' || c == b'+' {
200        s = s.advance(1);
201    }
202    s = s.skip_chars(b'0');
203    parse_digits(&mut s, |digit| d.try_add_digit(digit));
204    if s.check_first(b'.') {
205        s = s.advance(1);
206        let first = s;
207        if d.num_digits == 0 {
208            s = s.skip_chars(b'0');
209        }
210        while s.len() >= 8 && d.num_digits + 8 < Decimal::MAX_DIGITS {
211            // SAFETY: Safe since `s.len() >= 8`
212            let v = unsafe { s.read_u64() };
213            if !is_8digits(v) {
214                break;
215            }
216            // SAFETY: Safe since `num_digits + 8 < Decimal::MAX_DIGITS`
217            unsafe { d.digits[d.num_digits..].write_u64(v - 0x3030_3030_3030_3030) };
218            d.num_digits += 8;
219            s = s.advance(8);
220        }
221        parse_digits(&mut s, |digit| d.try_add_digit(digit));
222        d.decimal_point = s.len() as i32 - first.len() as i32;
223    }
224    if d.num_digits != 0 {
225        // Ignore the trailing zeros if there are any
226        let mut n_trailing_zeros = 0;
227        for &c in start[..(start.len() - s.len())].iter().rev() {
228            if c == b'0' {
229                n_trailing_zeros += 1;
230            } else if c != b'.' {
231                break;
232            }
233        }
234        d.decimal_point += n_trailing_zeros as i32;
235        d.num_digits -= n_trailing_zeros;
236        d.decimal_point += d.num_digits as i32;
237        if d.num_digits > Decimal::MAX_DIGITS {
238            d.truncated = true;
239            d.num_digits = Decimal::MAX_DIGITS;
240        }
241    }
242    if s.check_first2(b'e', b'E') {
243        s = s.advance(1);
244        let mut neg_exp = false;
245        if s.check_first(b'-') {
246            neg_exp = true;
247            s = s.advance(1);
248        } else if s.check_first(b'+') {
249            s = s.advance(1);
250        }
251        let mut exp_num = 0_i32;
252        parse_digits(&mut s, |digit| {
253            if exp_num < 0x10000 {
254                exp_num = 10 * exp_num + digit as i32;
255            }
256        });
257        d.decimal_point += if neg_exp {
258            -exp_num
259        } else {
260            exp_num
261        };
262    }
263    for i in d.num_digits..Decimal::MAX_DIGITS_WITHOUT_OVERFLOW {
264        d.digits[i] = 0;
265    }
266    d
267}
