38/// Error definitions
39#[derive(Debug, thiserror::Error, PartialEq, Eq)]
40pub enum ElfError {
41    /// Failed to parse ELF file
42    #[error("Failed to parse ELF file: {0}")]
43    FailedToParse(String),
44    /// Entrypoint out of bounds
45    #[error("Entrypoint out of bounds")]
46    EntrypointOutOfBounds,
47    /// Invalid entrypoint
48    #[error("Invalid entrypoint")]
49    InvalidEntrypoint,
50    /// Failed to get section
51    #[error("Failed to get section {0}")]
52    FailedToGetSection(String),
53    /// Unresolved symbol
54    #[error("Unresolved symbol ({0}) at instruction #{1:?} (ELF file offset {2:#x})")]
55    UnresolvedSymbol(String, usize, usize),
56    /// Section not found
57    #[error("Section not found: {0}")]
58    SectionNotFound(String),
59    /// Relative jump out of bounds
60    #[error("Relative jump out of bounds at instruction #{0}")]
61    RelativeJumpOutOfBounds(usize),
62    /// Symbol hash collision
63    #[error("Symbol hash collision {0:#x}")]
64    SymbolHashCollision(u32),
65    /// Incompatible ELF: wrong endianess
66    #[error("Incompatible ELF: wrong endianess")]
67    WrongEndianess,
68    /// Incompatible ELF: wrong ABI
69    #[error("Incompatible ELF: wrong ABI")]
70    WrongAbi,
71    /// Incompatible ELF: wrong machine
72    #[error("Incompatible ELF: wrong machine")]
73    WrongMachine,
74    /// Incompatible ELF: wrong class
75    #[error("Incompatible ELF: wrong class")]
76    WrongClass,
77    /// Not one text section
78    #[error("Multiple or no text sections, consider removing llc option: -function-sections")]
79    NotOneTextSection,
80    /// Read-write data not supported
81    #[error("Found writable section ({0}) in ELF, read-write data not supported")]
82    WritableSectionNotSupported(String),
83    /// Relocation failed, no loadable section contains virtual address
84    #[error("Relocation failed, no loadable section contains virtual address {0:#x}")]
85    AddressOutsideLoadableSection(u64),
86    /// Relocation failed, invalid referenced virtual address
87    #[error("Relocation failed, invalid referenced virtual address {0:#x}")]
88    InvalidVirtualAddress(u64),
89    /// Relocation failed, unknown type
90    #[error("Relocation failed, unknown type {0:?}")]
91    UnknownRelocation(u32),
92    /// Failed to read relocation info
93    #[error("Failed to read relocation info")]
94    FailedToReadRelocationInfo,
95    /// Incompatible ELF: wrong type
96    #[error("Incompatible ELF: wrong type")]
97    WrongType,
98    /// Unknown symbol
99    #[error("Unknown symbol with index {0}")]
100    UnknownSymbol(usize),
101    /// Offset or value is out of bounds
102    #[error("Offset or value is out of bounds")]
103    ValueOutOfBounds,
104    /// Detected sbpf_version required by the executable which are not enabled
105    #[error("Detected sbpf_version required by the executable which are not enabled")]
106    UnsupportedSBPFVersion,
107    /// Invalid program header
108    #[error("Invalid ELF program header")]
109    InvalidProgramHeader,
110}

112impl From<ElfParserError> for ElfError {
113    fn from(err: ElfParserError) -> Self {
114        match err {
115            ElfParserError::InvalidSectionHeader
116            | ElfParserError::InvalidString
117            | ElfParserError::InvalidSize
118            | ElfParserError::Overlap
119            | ElfParserError::SectionNotInOrder
120            | ElfParserError::NoSectionNameStringTable
121            | ElfParserError::InvalidDynamicSectionTable
122            | ElfParserError::InvalidRelocationTable
123            | ElfParserError::InvalidAlignment
124            | ElfParserError::NoStringTable
125            | ElfParserError::NoDynamicStringTable
126            | ElfParserError::InvalidFileHeader
127            | ElfParserError::StringTooLong(_, _) => ElfError::FailedToParse(err.to_string()),
128            ElfParserError::InvalidProgramHeader => ElfError::InvalidProgramHeader,
129            ElfParserError::OutOfBounds => ElfError::ValueOutOfBounds,
130        }
131    }
132}

134fn get_section(elf: &Elf64, name: &[u8]) -> Result<Elf64Shdr, ElfError> {
135    for section_header in elf.section_header_table() {
136        if elf.section_name(section_header.sh_name)? == name {
137            return Ok(section_header.clone());
138        }
139    }
140
141    Err(ElfError::SectionNotFound(
142        std::str::from_utf8(name)
143            .unwrap_or("UTF-8 error")
144            .to_string(),
145    ))
146}

148// For more information on the BPF instruction set:
149// https://github.com/iovisor/bpf-docs/blob/master/eBPF.md
150
151// msb                                                        lsb
152// +------------------------+----------------+----+----+--------+
153// |immediate               |offset          |src |dst |opcode  |
154// +------------------------+----------------+----+----+--------+
155
156// From least significant to most significant bit:
157//   8 bit opcode
158//   4 bit destination register (dst)
159//   4 bit source register (src)
160//   16 bit offset
161//   32 bit immediate (imm)
162
163/// Byte offset of the immediate field in the instruction
164const BYTE_OFFSET_IMMEDIATE: usize = 4;
165/// Byte length of the immediate field
166const BYTE_LENGTH_IMMEDIATE: usize = 4;
167
168/// BPF relocation types.
169#[allow(non_camel_case_types)]
170#[derive(Debug, PartialEq, Copy, Clone)]
171enum BpfRelocationType {
172    /// No relocation, placeholder
173    R_Bpf_None = 0,
174    /// R_BPF_64_64 relocation type is used for ld_imm64 instruction.
175    /// The actual to-be-relocated data (0 or section offset) is
176    /// stored at r_offset + 4 and the read/write data bitsize is 32
177    /// (4 bytes). The relocation can be resolved with the symbol
178    /// value plus implicit addend.
179    R_Bpf_64_64 = 1,
180    /// 64 bit relocation of a ldxdw instruction.  The ldxdw
181    /// instruction occupies two instruction slots. The 64-bit address
182    /// to load from is split into the 32-bit imm field of each
183    /// slot. The first slot's pre-relocation imm field contains the
184    /// virtual address (typically same as the file offset) of the
185    /// location to load. Relocation involves calculating the
186    /// post-load 64-bit physical address referenced by the imm field
187    /// and writing that physical address back into the imm fields of
188    /// the ldxdw instruction.
189    R_Bpf_64_Relative = 8,
190    /// Relocation of a call instruction.  The existing imm field
191    /// contains either an offset of the instruction to jump to (think
192    /// local function call) or a special value of "-1".  If -1 the
193    /// symbol must be looked up in the symbol table.  The relocation
194    /// entry contains the symbol number to call.  In order to support
195    /// both local jumps and calling external symbols a 32-bit hash is
196    /// computed and stored in the the call instruction's 32-bit imm
197    /// field.  The hash is used later to look up the 64-bit address
198    /// to jump to.  In the case of a local jump the hash is
199    /// calculated using the current program counter and in the case
200    /// of a symbol the hash is calculated using the name of the
201    /// symbol.
202    R_Bpf_64_32 = 10,
203}
204impl BpfRelocationType {
205    fn from_x86_relocation_type(from: u32) -> Option<BpfRelocationType> {
206        match from {
207            R_X86_64_NONE => Some(BpfRelocationType::R_Bpf_None),
208            R_X86_64_64 => Some(BpfRelocationType::R_Bpf_64_64),
209            R_X86_64_RELATIVE => Some(BpfRelocationType::R_Bpf_64_Relative),
210            R_X86_64_32 => Some(BpfRelocationType::R_Bpf_64_32),
211            _ => None,
212        }
213    }
214}

216/// ELF section
217#[derive(Debug, PartialEq)]
218pub enum Section {
219    /// Owned section data.
220    ///
221    /// The first field is virtual address of the section.
222    /// The second field is the actual section data.
223    Owned(usize, Vec<u8>),
224    /// Borrowed section data.
225    ///
226    /// The first field is virtual address of the section.
227    /// The second field can be used to index the input ELF buffer to
228    /// retrieve the section data.
229    Borrowed(usize, Range<usize>),
230}

337    /// Create from raw text section bytes (list of instructions)
338    pub fn new_from_text_bytes(
339        text_bytes: &[u8],
340        loader: Arc<BuiltinProgram<C>>,
341        sbpf_version: SBPFVersion,
342        mut function_registry: FunctionRegistry<usize>,
343    ) -> Result<Self, ElfError> {
344        let elf_bytes = AlignedMemory::from_slice(text_bytes);
345        let entry_pc = if let Some((_name, pc)) = function_registry.lookup_by_name(b"entrypoint") {
346            pc
347        } else {
348            function_registry.register_function_hashed_legacy(
349                &loader,
350                !sbpf_version.static_syscalls(),
351                *b"entrypoint",
352                0,
353            )?;
354            0
355        };
356        Ok(Self {
357            elf_bytes,
358            sbpf_version,
359            ro_section: Section::Borrowed(ebpf::MM_RODATA_START as usize, 0..text_bytes.len()),
360            text_section_vaddr: if sbpf_version.enable_lower_bytecode_vaddr() {
361                ebpf::MM_BYTECODE_START
362            } else {
363                ebpf::MM_RODATA_START
364            },
365            text_section_range: 0..text_bytes.len(),
366            entry_pc,
367            function_registry,
368            loader,
369            #[cfg(all(feature = "jit", not(target_os = "windows"), target_arch = "x86_64"))]
370            compiled_program: None,
371        })
372    }

374    /// Fully loads an ELF
375    pub fn load(bytes: &[u8], loader: Arc<BuiltinProgram<C>>) -> Result<Self, ElfError> {
376        const E_FLAGS_OFFSET: usize = 48;
377        let e_flags = LittleEndian::read_u32(
378            bytes
379                .get(E_FLAGS_OFFSET..E_FLAGS_OFFSET.saturating_add(std::mem::size_of::<u32>()))
380                .ok_or(ElfParserError::OutOfBounds)?,
381        );
382        let config = loader.get_config();
383        let sbpf_version = match e_flags {
384            0 => SBPFVersion::V0,
385            1 => SBPFVersion::V1,
386            2 => SBPFVersion::V2,
387            3 => SBPFVersion::V3,
388            4 => SBPFVersion::V4,
389            _ => SBPFVersion::Reserved,
390        };
391        if !config.enabled_sbpf_versions.contains(&sbpf_version) {
392            return Err(ElfError::UnsupportedSBPFVersion);
393        }
394
395        let mut executable = if sbpf_version.enable_stricter_elf_headers() {
396            Self::load_with_strict_parser(bytes, loader)?
397        } else {
398            Self::load_with_lenient_parser(bytes, loader)?
399        };
400        executable.sbpf_version = sbpf_version;
401        Ok(executable)
402    }

404    /// Loads an ELF without relocation
405    pub fn load_with_strict_parser(
406        bytes: &[u8],
407        loader: Arc<BuiltinProgram<C>>,
408    ) -> Result<Self, ElfParserError> {
409        use crate::elf_parser::{
410            consts::{ELFMAG, EV_CURRENT, PF_R, PF_W, PF_X, PT_LOAD, SHN_UNDEF, STT_FUNC},
411            types::{Elf64Ehdr, Elf64Shdr, Elf64Sym},
412        };
413
414        let aligned_memory = AlignedMemory::<{ HOST_ALIGN }>::from_slice(bytes);
415        let elf_bytes = aligned_memory.as_slice();
416
417        let (file_header_range, file_header) = Elf64::parse_file_header(elf_bytes)?;
418        let program_header_table_range = mem::size_of::<Elf64Ehdr>()
419            ..mem::size_of::<Elf64Phdr>()
420                .saturating_mul(file_header.e_phnum as usize)
421                .saturating_add(mem::size_of::<Elf64Ehdr>());
422        if file_header.e_ident.ei_mag != ELFMAG
423            || file_header.e_ident.ei_class != ELFCLASS64
424            || file_header.e_ident.ei_data != ELFDATA2LSB
425            || file_header.e_ident.ei_version != EV_CURRENT as u8
426            || file_header.e_ident.ei_osabi != ELFOSABI_NONE
427            || file_header.e_ident.ei_abiversion != 0x00
428            || file_header.e_ident.ei_pad != [0x00; 7]
429            || file_header.e_type != ET_DYN
430            || file_header.e_machine != EM_SBPF
431            || file_header.e_version != EV_CURRENT
432            // file_header.e_entry
433            || file_header.e_phoff != mem::size_of::<Elf64Ehdr>() as u64
434            // file_header.e_shoff
435            // file_header.e_flags
436            || file_header.e_ehsize != mem::size_of::<Elf64Ehdr>() as u16
437            || file_header.e_phentsize != mem::size_of::<Elf64Phdr>() as u16
438            || file_header.e_phnum < EXPECTED_PROGRAM_HEADERS.len() as u16
439            || program_header_table_range.end >= elf_bytes.len()
440            || file_header.e_shentsize != mem::size_of::<Elf64Shdr>() as u16
441            // file_header.e_shnum
442            || file_header.e_shstrndx >= file_header.e_shnum
443        {
444            return Err(ElfParserError::InvalidFileHeader);
445        }
446
447        const EXPECTED_PROGRAM_HEADERS: [(u32, u64); 4] = [
448            (PF_X, ebpf::MM_BYTECODE_START),     // byte code
449            (PF_R, ebpf::MM_RODATA_START),       // read only data
450            (PF_R | PF_W, ebpf::MM_STACK_START), // stack
451            (PF_R | PF_W, ebpf::MM_HEAP_START),  // heap
452        ];
453        let program_header_table =
454            Elf64::slice_from_bytes::<Elf64Phdr>(elf_bytes, program_header_table_range.clone())?;
455        for (program_header, (p_flags, p_vaddr)) in program_header_table
456            .iter()
457            .zip(EXPECTED_PROGRAM_HEADERS.iter())
458        {
459            let p_filesz = if (*p_flags & PF_W) != 0 {
460                0
461            } else {
462                program_header.p_memsz
463            };
464            if program_header.p_type != PT_LOAD
465                || program_header.p_flags != *p_flags
466                || program_header.p_offset < program_header_table_range.end as u64
467                || program_header.p_offset >= elf_bytes.len() as u64
468                || program_header.p_offset.checked_rem(ebpf::INSN_SIZE as u64) != Some(0)
469                || program_header.p_vaddr != *p_vaddr
470                || program_header.p_paddr != *p_vaddr
471                || program_header.p_filesz != p_filesz
472                || program_header.p_filesz
473                    > (elf_bytes.len() as u64).saturating_sub(program_header.p_offset)
474                || program_header.p_filesz.checked_rem(ebpf::INSN_SIZE as u64) != Some(0)
475                || program_header.p_memsz >= ebpf::MM_REGION_SIZE
476            {
477                return Err(ElfParserError::InvalidProgramHeader);
478            }
479        }
480
481        let bytecode_header = &program_header_table[0];
482        let rodata_header = &program_header_table[1];
483        let text_section_vaddr = bytecode_header.p_vaddr;
484        let text_section_range = bytecode_header.file_range().unwrap_or_default();
485        let ro_section = Section::Borrowed(
486            rodata_header.p_vaddr as usize,
487            rodata_header.file_range().unwrap_or_default(),
488        );
489
490        if !bytecode_header.vm_range().contains(
491            &file_header
492                .e_entry
493                .saturating_add(ebpf::INSN_SIZE as u64)
494                .saturating_sub(1),
495        ) || file_header.e_entry.checked_rem(ebpf::INSN_SIZE as u64) != Some(0)
496        {
497            return Err(ElfParserError::InvalidFileHeader);
498        }
499        let entry_pc = file_header
500            .e_entry
501            .saturating_sub(bytecode_header.p_vaddr)
502            .checked_div(ebpf::INSN_SIZE as u64)
503            .unwrap_or_default() as usize;
504        let entry_insn = ebpf::get_insn(&elf_bytes[text_section_range.clone()], entry_pc);
505        if !entry_insn.is_function_start_marker() {
506            return Err(ElfParserError::InvalidFileHeader);
507        }
508
509        let mut function_registry = FunctionRegistry::<usize>::default();
510        let config = loader.get_config();
511        if config.enable_symbol_and_section_labels {
512            let (_section_header_table_range, section_header_table) =
513                Elf64::parse_section_header_table(
514                    elf_bytes,
515                    file_header_range.clone(),
516                    file_header,
517                    program_header_table_range.clone(),
518                )
519                .unwrap();
520            let section_names_section_header = (file_header.e_shstrndx != SHN_UNDEF)
521                .then(|| {
522                    section_header_table
523                        .get(file_header.e_shstrndx as usize)
524                        .ok_or(ElfParserError::OutOfBounds)
525                })
526                .transpose()?
527                .unwrap();
528            let mut symbol_names_section_header = None;
529            let mut symbol_table_section_header = None;
530            for section_header in section_header_table.iter() {
531                let section_name = Elf64::get_string_in_section(
532                    elf_bytes,
533                    section_names_section_header,
534                    section_header.sh_name,
535                    64,
536                )
537                .unwrap();
538                if section_name == b".strtab" {
539                    symbol_names_section_header = Some(section_header);
540                }
541                if section_name == b".symtab" {
542                    symbol_table_section_header = Some(section_header);
543                }
544            }
545            let symbol_names_section_header = symbol_names_section_header.unwrap();
546            let symbol_table: &[Elf64Sym] =
547                Elf64::slice_from_section_header(elf_bytes, symbol_table_section_header.unwrap())
548                    .unwrap();
549            for symbol in symbol_table {
550                if symbol.st_info & STT_FUNC == 0 {
551                    continue;
552                }
553                let target_pc = symbol
554                    .st_value
555                    .saturating_sub(bytecode_header.p_vaddr)
556                    .checked_div(ebpf::INSN_SIZE as u64)
557                    .unwrap_or_default() as usize;
558                let name = Elf64::get_string_in_section(
559                    elf_bytes,
560                    symbol_names_section_header,
561                    symbol.st_name as Elf64Word,
562                    u8::MAX as usize,
563                )
564                .unwrap();
565                function_registry
566                    .register_function(target_pc as u32, name, target_pc)
567                    .unwrap();
568            }
569        }
570
571        Ok(Self {
572            elf_bytes: aligned_memory,
573            sbpf_version: SBPFVersion::Reserved, // Is set in Self::load()
574            ro_section,
575            text_section_vaddr,
576            text_section_range,
577            entry_pc,
578            function_registry,
579            loader,
580            #[cfg(all(feature = "jit", not(target_os = "windows"), target_arch = "x86_64"))]
581            compiled_program: None,
582        })
583    }

585    /// Loads an ELF with relocation
586    fn load_with_lenient_parser(
587        bytes: &[u8],
588        loader: Arc<BuiltinProgram<C>>,
589    ) -> Result<Self, ElfError> {
590        // We always need one memory copy to take ownership and for relocations
591        let aligned_memory = AlignedMemory::<{ HOST_ALIGN }>::from_slice(bytes);
592        let (mut elf_bytes, unrelocated_elf_bytes) =
593            if is_memory_aligned(bytes.as_ptr() as usize, HOST_ALIGN) {
594                (aligned_memory, bytes)
595            } else {
596                // We might need another memory copy to ensure alignment
597                (aligned_memory.clone(), aligned_memory.as_slice())
598            };
599        let elf = Elf64::parse(unrelocated_elf_bytes)?;
600
601        let config = loader.get_config();
602        let header = elf.file_header();
603
604        Self::validate(&elf, elf_bytes.as_slice())?;
605
606        // calculate the text section info
607        let text_section = get_section(&elf, b".text")?;
608        let text_section_vaddr = text_section.sh_addr.saturating_add(ebpf::MM_RODATA_START);
609        if (config.reject_broken_elfs && text_section.sh_addr != text_section.sh_offset)
610            || text_section_vaddr > ebpf::MM_STACK_START
611        {
612            return Err(ElfError::ValueOutOfBounds);
613        }
614
615        // relocate symbols
616        let mut function_registry = FunctionRegistry::default();
617        Self::relocate(
618            &mut function_registry,
619            &loader,
620            &elf,
621            elf_bytes.as_slice_mut(),
622        )?;
623
624        // calculate entrypoint offset into the text section
625        let offset = header.e_entry.saturating_sub(text_section.sh_addr);
626        if offset.checked_rem(ebpf::INSN_SIZE as u64) != Some(0) {
627            return Err(ElfError::InvalidEntrypoint);
628        }
629        let entry_pc = if let Some(entry_pc) = (offset as usize).checked_div(ebpf::INSN_SIZE) {
630            function_registry.unregister_function(ebpf::hash_symbol_name(b"entrypoint"));
631            function_registry.register_function_hashed_legacy(
632                &loader,
633                true,
634                *b"entrypoint",
635                entry_pc,
636            )?;
637            entry_pc
638        } else {
639            return Err(ElfError::InvalidEntrypoint);
640        };
641
642        let ro_section = Self::parse_ro_sections(
643            config,
644            elf.section_header_table()
645                .iter()
646                .map(|s| (elf.section_name(s.sh_name).ok(), s)),
647            elf_bytes.as_slice(),
648        )?;
649
650        Ok(Self {
651            elf_bytes,
652            sbpf_version: SBPFVersion::Reserved, // Is set in Self::load()
653            ro_section,
654            text_section_vaddr,
655            text_section_range: text_section.file_range().unwrap_or_default(),
656            entry_pc,
657            function_registry,
658            loader,
659            #[cfg(all(feature = "jit", not(target_os = "windows"), target_arch = "x86_64"))]
660            compiled_program: None,
661        })
662    }

664    /// Calculate the total memory size of the executable
665    #[rustfmt::skip]
666    #[allow(clippy::size_of_ref)]
667    pub fn mem_size(&self) -> usize {
668        let mut total = mem::size_of::<Self>();
669        total = total
670            // elf bytes
671            .saturating_add(self.elf_bytes.mem_size())
672            // ro section
673            .saturating_add(match &self.ro_section {
674                Section::Owned(_, data) => data.capacity(),
675                Section::Borrowed(_, _) => 0,
676            })
677            // bpf functions
678            .saturating_add(self.function_registry.mem_size());
679
680        #[cfg(all(feature = "jit", not(target_os = "windows"), target_arch = "x86_64"))]
681        {
682            // compiled programs
683            total = total.saturating_add(self.compiled_program.as_ref().map_or(0, |program| program.mem_size()));
684        }
685
686        total
687    }

691    /// Validates the ELF
692    pub fn validate(elf: &Elf64, elf_bytes: &[u8]) -> Result<(), ElfError> {
693        let header = elf.file_header();
694        if header.e_ident.ei_class != ELFCLASS64 {
695            return Err(ElfError::WrongClass);
696        }
697        if header.e_ident.ei_data != ELFDATA2LSB {
698            return Err(ElfError::WrongEndianess);
699        }
700        if header.e_ident.ei_osabi != ELFOSABI_NONE {
701            return Err(ElfError::WrongAbi);
702        }
703        if header.e_machine != EM_BPF && header.e_machine != EM_SBPF {
704            return Err(ElfError::WrongMachine);
705        }
706        if header.e_type != ET_DYN {
707            return Err(ElfError::WrongType);
708        }
709
710        let num_text_sections =
711            elf.section_header_table()
712                .iter()
713                .fold(0, |count: usize, section_header| {
714                    if let Ok(this_name) = elf.section_name(section_header.sh_name) {
715                        if this_name == b".text" {
716                            return count.saturating_add(1);
717                        }
718                    }
719                    count
720                });
721        if 1 != num_text_sections {
722            return Err(ElfError::NotOneTextSection);
723        }
724
725        for section_header in elf.section_header_table().iter() {
726            if let Ok(name) = elf.section_name(section_header.sh_name) {
727                if name.starts_with(b".bss")
728                    || (section_header.is_writable()
729                        && (name.starts_with(b".data") && !name.starts_with(b".data.rel")))
730                {
731                    return Err(ElfError::WritableSectionNotSupported(
732                        String::from_utf8_lossy(name).to_string(),
733                    ));
734                }
735            }
736        }
737
738        for section_header in elf.section_header_table().iter() {
739            let start = section_header.sh_offset as usize;
740            let end = section_header
741                .sh_offset
742                .checked_add(section_header.sh_size)
743                .ok_or(ElfError::ValueOutOfBounds)? as usize;
744            let _ = elf_bytes
745                .get(start..end)
746                .ok_or(ElfError::ValueOutOfBounds)?;
747        }
748        let text_section = get_section(elf, b".text")?;
749        if !text_section.vm_range().contains(&header.e_entry) {
750            return Err(ElfError::EntrypointOutOfBounds);
751        }
752
753        Ok(())
754    }

756    /// Parses and concatenates the readonly data sections
757    pub fn parse_ro_sections<'a, S: IntoIterator<Item = (Option<&'a [u8]>, &'a Elf64Shdr)>>(
758        config: &Config,
759        sections: S,
760        elf_bytes: &[u8],
761    ) -> Result<Section, ElfError> {
762        // the lowest section address
763        let mut lowest_addr = usize::MAX;
764        // the highest section address
765        let mut highest_addr = 0;
766        // the aggregated section length, not including gaps between sections
767        let mut ro_fill_length = 0usize;
768        let mut invalid_offsets = false;
769
770        // keep track of where ro sections are so we can tell whether they're
771        // contiguous
772        let mut first_ro_section = 0;
773        let mut last_ro_section = 0;
774        let mut n_ro_sections = 0usize;
775
776        let mut ro_slices = vec![];
777        for (i, (name, section_header)) in sections.into_iter().enumerate() {
778            match name {
779                Some(name)
780                    if name == b".text"
781                        || name == b".rodata"
782                        || name == b".data.rel.ro"
783                        || name == b".eh_frame" => {}
784                _ => continue,
785            }
786
787            if n_ro_sections == 0 {
788                first_ro_section = i;
789            }
790            last_ro_section = i;
791            n_ro_sections = n_ro_sections.saturating_add(1);
792
793            let section_addr = section_header.sh_addr;
794
795            // sh_offset handling:
796            // section_addr must match sh_offset
797            if !invalid_offsets && section_addr != section_header.sh_offset {
798                invalid_offsets = true;
799            }
800
801            let vaddr_end = section_addr.saturating_add(ebpf::MM_RODATA_START);
802            if (config.reject_broken_elfs && invalid_offsets) || vaddr_end > ebpf::MM_STACK_START {
803                return Err(ElfError::ValueOutOfBounds);
804            }
805
806            let section_data = elf_bytes
807                .get(section_header.file_range().unwrap_or_default())
808                .ok_or(ElfError::ValueOutOfBounds)?;
809
810            let section_addr = section_addr as usize;
811            lowest_addr = lowest_addr.min(section_addr);
812            highest_addr = highest_addr.max(section_addr.saturating_add(section_data.len()));
813            ro_fill_length = ro_fill_length.saturating_add(section_data.len());
814
815            ro_slices.push((section_addr, section_data));
816        }
817
818        if config.reject_broken_elfs && lowest_addr.saturating_add(ro_fill_length) > highest_addr {
819            return Err(ElfError::ValueOutOfBounds);
820        }
821
822        let can_borrow = !invalid_offsets
823            && last_ro_section
824                .saturating_add(1)
825                .saturating_sub(first_ro_section)
826                == n_ro_sections;
827        let ro_section = if config.optimize_rodata && can_borrow {
828            // Read only sections are grouped together with no intermixed non-ro
829            // sections. We can borrow.
830
831            let addr_offset = if lowest_addr >= ebpf::MM_RODATA_START as usize {
832                // The first field of Section::Borrowed is an offset from
833                // ebpf::MM_RODATA_START so if the linker has already put the
834                // sections within ebpf::MM_RODATA_START, we need to subtract
835                // it now.
836                lowest_addr
837            } else {
838                lowest_addr.saturating_add(ebpf::MM_RODATA_START as usize)
839            };
840
841            Section::Borrowed(addr_offset, lowest_addr..highest_addr)
842        } else {
843            // Read only and other non-ro sections are mixed. Zero the non-ro
844            // sections and and copy the ro ones at their intended offsets.
845
846            if config.optimize_rodata {
847                // The rodata region starts at MM_RODATA_START + offset,
848                // [MM_RODATA_START, MM_RODATA_START + offset) is not
849                // mappable. We only need to allocate highest_addr - lowest_addr
850                // bytes.
851                highest_addr = highest_addr.saturating_sub(lowest_addr);
852            } else {
853                // For backwards compatibility, the whole [MM_RODATA_START,
854                // MM_RODATA_START + highest_addr) range is mappable. We need
855                // to allocate the whole address range.
856                lowest_addr = 0;
857            };
858
859            let buf_len = highest_addr;
860            if buf_len > elf_bytes.len() {
861                return Err(ElfError::ValueOutOfBounds);
862            }
863
864            let mut ro_section = vec![0; buf_len];
865            for (section_addr, slice) in ro_slices.iter() {
866                let buf_offset_start = section_addr.saturating_sub(lowest_addr);
867                ro_section[buf_offset_start..buf_offset_start.saturating_add(slice.len())]
868                    .copy_from_slice(slice);
869            }
870
871            let addr_offset = if lowest_addr >= ebpf::MM_RODATA_START as usize {
872                lowest_addr
873            } else {
874                lowest_addr.saturating_add(ebpf::MM_RODATA_START as usize)
875            };
876            Section::Owned(addr_offset, ro_section)
877        };
878
879        Ok(ro_section)
880    }

882    /// Relocates the ELF in-place
883    fn relocate(
884        function_registry: &mut FunctionRegistry<usize>,
885        loader: &BuiltinProgram<C>,
886        elf: &Elf64,
887        elf_bytes: &mut [u8],
888    ) -> Result<(), ElfError> {
889        let mut syscall_cache = BTreeMap::new();
890        let text_section = get_section(elf, b".text")?;
891
892        // Fixup all program counter relative call instructions
893        let config = loader.get_config();
894        let text_bytes = elf_bytes
895            .get_mut(text_section.file_range().unwrap_or_default())
896            .ok_or(ElfError::ValueOutOfBounds)?;
897        let instruction_count = text_bytes
898            .len()
899            .checked_div(ebpf::INSN_SIZE)
900            .ok_or(ElfError::ValueOutOfBounds)?;
901        for i in 0..instruction_count {
902            let insn = ebpf::get_insn(text_bytes, i);
903            if insn.opc == ebpf::CALL_IMM && insn.imm != -1 {
904                let target_pc = (i as isize)
905                    .saturating_add(1)
906                    .saturating_add(insn.imm as isize);
907                if target_pc < 0 || target_pc >= instruction_count as isize {
908                    return Err(ElfError::RelativeJumpOutOfBounds(i));
909                }
910                let name = if config.enable_symbol_and_section_labels {
911                    format!("function_{target_pc}")
912                } else {
913                    String::default()
914                };
915                let key = function_registry.register_function_hashed_legacy(
916                    loader,
917                    true,
918                    name.as_bytes(),
919                    target_pc as usize,
920                )?;
921                let offset = i.saturating_mul(ebpf::INSN_SIZE).saturating_add(4);
922                let checked_slice = text_bytes
923                    .get_mut(offset..offset.saturating_add(4))
924                    .ok_or(ElfError::ValueOutOfBounds)?;
925                LittleEndian::write_u32(checked_slice, key);
926            }
927        }
928
929        // Fixup all the relocations in the relocation section if exists
930        for relocation in elf.dynamic_relocations_table().unwrap_or_default().iter() {
931            let r_offset = relocation.r_offset as usize;
932
933            match BpfRelocationType::from_x86_relocation_type(relocation.r_type()) {
934                Some(BpfRelocationType::R_Bpf_64_64) => {
935                    // Offset of the immediate field
936                    let imm_offset = r_offset.saturating_add(BYTE_OFFSET_IMMEDIATE);
937
938                    // Read the instruction's immediate field which contains virtual
939                    // address to convert to physical
940                    let checked_slice = elf_bytes
941                        .get(imm_offset..imm_offset.saturating_add(BYTE_LENGTH_IMMEDIATE))
942                        .ok_or(ElfError::ValueOutOfBounds)?;
943                    let refd_addr = LittleEndian::read_u32(checked_slice) as u64;
944
945                    let symbol = elf
946                        .dynamic_symbol_table()
947                        .and_then(|table| table.get(relocation.r_sym() as usize).cloned())
948                        .ok_or_else(|| ElfError::UnknownSymbol(relocation.r_sym() as usize))?;
949
950                    // The relocated address is relative to the address of the
951                    // symbol at index `r_sym`
952                    let mut addr = symbol.st_value.saturating_add(refd_addr);
953
954                    // The "physical address" from the VM's perspective is rooted
955                    // at `MM_RODATA_START`. If the linker hasn't already put
956                    // the symbol within `MM_RODATA_START`, we need to do so
957                    // now.
958                    if addr < ebpf::MM_RODATA_START {
959                        addr = ebpf::MM_RODATA_START.saturating_add(addr);
960                    }
961
962                    let imm_low_offset = imm_offset;
963                    let imm_high_offset = imm_low_offset.saturating_add(INSN_SIZE);
964
965                    // Write the low side of the relocate address
966                    let imm_slice = elf_bytes
967                        .get_mut(
968                            imm_low_offset..imm_low_offset.saturating_add(BYTE_LENGTH_IMMEDIATE),
969                        )
970                        .ok_or(ElfError::ValueOutOfBounds)?;
971                    LittleEndian::write_u32(imm_slice, (addr & 0xFFFFFFFF) as u32);
972
973                    // Write the high side of the relocate address
974                    let imm_slice = elf_bytes
975                        .get_mut(
976                            imm_high_offset..imm_high_offset.saturating_add(BYTE_LENGTH_IMMEDIATE),
977                        )
978                        .ok_or(ElfError::ValueOutOfBounds)?;
979                    LittleEndian::write_u32(
980                        imm_slice,
981                        addr.checked_shr(32).unwrap_or_default() as u32,
982                    );
983                }
984                Some(BpfRelocationType::R_Bpf_64_Relative) => {
985                    // Relocation between different sections, where the target
986                    // memory is not associated to a symbol (eg some compiler
987                    // generated rodata that doesn't have an explicit symbol).
988
989                    // Offset of the immediate field
990                    let imm_offset = r_offset.saturating_add(BYTE_OFFSET_IMMEDIATE);
991
992                    if text_section
993                        .file_range()
994                        .unwrap_or_default()
995                        .contains(&r_offset)
996                    {
997                        // We're relocating a lddw instruction, which spans two
998                        // instruction slots. The address to be relocated is
999                        // split in two halves in the two imms of the
1000                        // instruction slots.
1001                        let imm_low_offset = imm_offset;
1002                        let imm_high_offset = r_offset
1003                            .saturating_add(INSN_SIZE)
1004                            .saturating_add(BYTE_OFFSET_IMMEDIATE);
1005
1006                        // Read the low side of the address
1007                        let imm_slice = elf_bytes
1008                            .get(
1009                                imm_low_offset
1010                                    ..imm_low_offset.saturating_add(BYTE_LENGTH_IMMEDIATE),
1011                            )
1012                            .ok_or(ElfError::ValueOutOfBounds)?;
1013                        let va_low = LittleEndian::read_u32(imm_slice) as u64;
1014
1015                        // Read the high side of the address
1016                        let imm_slice = elf_bytes
1017                            .get(
1018                                imm_high_offset
1019                                    ..imm_high_offset.saturating_add(BYTE_LENGTH_IMMEDIATE),
1020                            )
1021                            .ok_or(ElfError::ValueOutOfBounds)?;
1022                        let va_high = LittleEndian::read_u32(imm_slice) as u64;
1023
1024                        // Put the address back together
1025                        let mut refd_addr = va_high.checked_shl(32).unwrap_or_default() | va_low;
1026
1027                        if refd_addr == 0 {
1028                            return Err(ElfError::InvalidVirtualAddress(refd_addr));
1029                        }
1030
1031                        if refd_addr < ebpf::MM_RODATA_START {
1032                            // The linker hasn't already placed rodata within
1033                            // MM_RODATA_START, so we do so now
1034                            refd_addr = ebpf::MM_RODATA_START.saturating_add(refd_addr);
1035                        }
1036
1037                        // Write back the low half
1038                        let imm_slice = elf_bytes
1039                            .get_mut(
1040                                imm_low_offset
1041                                    ..imm_low_offset.saturating_add(BYTE_LENGTH_IMMEDIATE),
1042                            )
1043                            .ok_or(ElfError::ValueOutOfBounds)?;
1044                        LittleEndian::write_u32(imm_slice, (refd_addr & 0xFFFFFFFF) as u32);
1045
1046                        // Write back the high half
1047                        let imm_slice = elf_bytes
1048                            .get_mut(
1049                                imm_high_offset
1050                                    ..imm_high_offset.saturating_add(BYTE_LENGTH_IMMEDIATE),
1051                            )
1052                            .ok_or(ElfError::ValueOutOfBounds)?;
1053                        LittleEndian::write_u32(
1054                            imm_slice,
1055                            refd_addr.checked_shr(32).unwrap_or_default() as u32,
1056                        );
1057                    } else {
1058                        // There used to be a bug in toolchains before
1059                        // https://github.com/solana-labs/llvm-project/pull/35 where for 64 bit
1060                        // relocations we were encoding only the low 32 bits, shifted 32 bits to
1061                        // the left. Our relocation code used to be compatible with that, so we
1062                        // need to keep supporting this case for backwards compatibility.
1063                        let addr_slice = elf_bytes
1064                            .get(imm_offset..imm_offset.saturating_add(BYTE_LENGTH_IMMEDIATE))
1065                            .ok_or(ElfError::ValueOutOfBounds)?;
1066                        let mut refd_addr = LittleEndian::read_u32(addr_slice) as u64;
1067                        refd_addr = ebpf::MM_RODATA_START.saturating_add(refd_addr);
1068
1069                        let addr_slice = elf_bytes
1070                            .get_mut(r_offset..r_offset.saturating_add(mem::size_of::<u64>()))
1071                            .ok_or(ElfError::ValueOutOfBounds)?;
1072                        LittleEndian::write_u64(addr_slice, refd_addr);
1073                    }
1074                }
1075                Some(BpfRelocationType::R_Bpf_64_32) => {
1076                    // The .text section has an unresolved call to symbol instruction
1077                    // Hash the symbol name and stick it into the call instruction's imm
1078                    // field.  Later that hash will be used to look up the function location.
1079
1080                    // Offset of the immediate field
1081                    let imm_offset = r_offset.saturating_add(BYTE_OFFSET_IMMEDIATE);
1082
1083                    let symbol = elf
1084                        .dynamic_symbol_table()
1085                        .and_then(|table| table.get(relocation.r_sym() as usize).cloned())
1086                        .ok_or_else(|| ElfError::UnknownSymbol(relocation.r_sym() as usize))?;
1087
1088                    let name = elf
1089                        .dynamic_symbol_name(symbol.st_name as Elf64Word)
1090                        .map_err(|_| ElfError::UnknownSymbol(symbol.st_name as usize))?;
1091
1092                    // If the symbol is defined, this is a bpf-to-bpf call
1093                    let key = if symbol.is_function() && symbol.st_value != 0 {
1094                        if !text_section.vm_range().contains(&symbol.st_value) {
1095                            return Err(ElfError::ValueOutOfBounds);
1096                        }
1097                        let target_pc = (symbol.st_value.saturating_sub(text_section.sh_addr)
1098                            as usize)
1099                            .checked_div(ebpf::INSN_SIZE)
1100                            .unwrap_or_default();
1101                        function_registry
1102                            .register_function_hashed_legacy(loader, true, name, target_pc)?
1103                    } else {
1104                        // Else it's a syscall
1105                        let hash = *syscall_cache
1106                            .entry(symbol.st_name)
1107                            .or_insert_with(|| ebpf::hash_symbol_name(name));
1108                        if config.reject_broken_elfs
1109                            && loader.get_function_registry().lookup_by_key(hash).is_none()
1110                        {
1111                            return Err(ElfError::UnresolvedSymbol(
1112                                String::from_utf8_lossy(name).to_string(),
1113                                r_offset.checked_div(ebpf::INSN_SIZE).unwrap_or(0),
1114                                r_offset,
1115                            ));
1116                        }
1117                        hash
1118                    };
1119
1120                    let checked_slice = elf_bytes
1121                        .get_mut(imm_offset..imm_offset.saturating_add(BYTE_LENGTH_IMMEDIATE))
1122                        .ok_or(ElfError::ValueOutOfBounds)?;
1123                    LittleEndian::write_u32(checked_slice, key);
1124                }
1125                _ => return Err(ElfError::UnknownRelocation(relocation.r_type())),
1126            }
1127        }
1128
1129        if config.enable_symbol_and_section_labels {
1130            // Register all known function names from the symbol table
1131            for symbol in elf.symbol_table().ok().flatten().unwrap_or_default().iter() {
1132                if symbol.st_info & 0xEF != 0x02 {
1133                    continue;
1134                }
1135                if !text_section.vm_range().contains(&symbol.st_value) {
1136                    return Err(ElfError::ValueOutOfBounds);
1137                }
1138                let target_pc = (symbol.st_value.saturating_sub(text_section.sh_addr) as usize)
1139                    .checked_div(ebpf::INSN_SIZE)
1140                    .unwrap_or_default();
1141                let name = elf
1142                    .symbol_name(symbol.st_name as Elf64Word)
1143                    .map_err(|_| ElfError::UnknownSymbol(symbol.st_name as usize))?;
1144                function_registry.register_function_hashed_legacy(loader, true, name, target_pc)?;
1145            }
1146        }
1147
1148        Ok(())
1149    }

1151    #[allow(dead_code)]
1152    fn dump_data(name: &str, prog: &[u8]) {
1153        let mut eight_bytes: Vec<u8> = Vec::new();
1154        println!("{name}");
1155        for i in prog.iter() {
1156            if eight_bytes.len() >= 7 {
1157                println!("{eight_bytes:02X?}");
1158                eight_bytes.clear();
1159            } else {
1160                eight_bytes.push(*i);
1161            }
1162        }
1163    }
1164}

1166/// Creates a [MemoryRegion] for the given [Section]
1167pub fn get_ro_region(ro_section: &Section, elf: &[u8]) -> MemoryRegion {
1168    let (offset, ro_data) = match ro_section {
1169        Section::Owned(offset, data) => (*offset, data.as_slice()),
1170        Section::Borrowed(offset, byte_range) => (*offset, &elf[byte_range.clone()]),
1171    };
1172
1173    // If offset > 0, the region will start at MM_RODATA_START + the offset of
1174    // the first read only byte. [MM_RODATA_START, MM_RODATA_START + offset)
1175    // will be unmappable, see MemoryRegion::vm_to_host.
1176    MemoryRegion::new_readonly(ro_data, offset as u64)
1177}
