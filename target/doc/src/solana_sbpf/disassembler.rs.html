8//! Functions in this module are used to handle eBPF programs with a higher level representation,

19fn resolve_label(cfg_nodes: &BTreeMap<usize, CfgNode>, pc: usize) -> &str {
20    cfg_nodes
21        .get(&pc)
22        .map(|cfg_node| cfg_node.label.as_str())
23        .unwrap_or("[invalid]")
24}

26#[inline]
27fn alu_imm_str(name: &str, insn: &ebpf::Insn) -> String {
28    format!("{} r{}, {}", name, insn.dst, insn.imm)
29}
30
31#[inline]
32fn alu_reg_str(name: &str, insn: &ebpf::Insn) -> String {
33    format!("{} r{}, r{}", name, insn.dst, insn.src)
34}
35
36#[inline]
37fn byteswap_str(name: &str, insn: &ebpf::Insn) -> String {
38    match insn.imm {
39        16 | 32 | 64 => {}
40        _ => println!("[Disassembler] Warning: Invalid offset value for {name} insn"),
41    }
42    format!("{}{} r{}", name, insn.imm, insn.dst)
43}
44
45#[inline]
46fn signed_off_str(value: i16) -> String {
47    if value < 0 {
48        format!("-{:#x}", -(value as isize))
49    } else {
50        format!("+{value:#x}")
51    }
52}
53
54#[inline]
55fn ld_st_imm_str(name: &str, insn: &ebpf::Insn) -> String {
56    format!(
57        "{} [r{}{}], {}",
58        name,
59        insn.dst,
60        signed_off_str(insn.off),
61        insn.imm
62    )
63}
64
65#[inline]
66fn ld_reg_str(name: &str, insn: &ebpf::Insn) -> String {
67    format!(
68        "{} r{}, [r{}{}]",
69        name,
70        insn.dst,
71        insn.src,
72        signed_off_str(insn.off)
73    )
74}
75
76#[inline]
77fn st_reg_str(name: &str, insn: &ebpf::Insn) -> String {
78    format!(
79        "{} [r{}{}], r{}",
80        name,
81        insn.dst,
82        signed_off_str(insn.off),
83        insn.src
84    )
85}
86
87#[inline]
88fn jmp_imm_str(name: &str, insn: &ebpf::Insn, cfg_nodes: &BTreeMap<usize, CfgNode>) -> String {
89    let target_pc = (insn.ptr as isize + insn.off as isize + 1) as usize;
90    format!(
91        "{} r{}, {}, {}",
92        name,
93        insn.dst,
94        insn.imm,
95        resolve_label(cfg_nodes, target_pc)
96    )
97}
98
99#[inline]
100fn jmp_reg_str(name: &str, insn: &ebpf::Insn, cfg_nodes: &BTreeMap<usize, CfgNode>) -> String {
101    let target_pc = (insn.ptr as isize + insn.off as isize + 1) as usize;
102    format!(
103        "{} r{}, r{}, {}",
104        name,
105        insn.dst,
106        insn.src,
107        resolve_label(cfg_nodes, target_pc)
108    )
109}

111/// Disassemble an eBPF instruction

121    let name;

127        // BPF_LDX class

130        ebpf::LD_W_REG  if !sbpf_version.move_memory_instruction_classes() => { name = "ldxw";  desc = ld_reg_str(name, insn); },
131        ebpf::LD_DW_REG if !sbpf_version.move_memory_instruction_classes() => { name = "ldxdw"; desc = ld_reg_str(name, insn); },
132
133        // BPF_ST class
134        ebpf::ST_B_IMM  if !sbpf_version.move_memory_instruction_classes() => { name = "stb";   desc = ld_st_imm_str(name, insn); },
135        ebpf::ST_H_IMM  if !sbpf_version.move_memory_instruction_classes() => { name = "sth";   desc = ld_st_imm_str(name, insn); },
136        ebpf::ST_W_IMM  if !sbpf_version.move_memory_instruction_classes() => { name = "stw";   desc = ld_st_imm_str(name, insn); },
137        ebpf::ST_DW_IMM if !sbpf_version.move_memory_instruction_classes() => { name = "stdw";  desc = ld_st_imm_str(name, insn); },
138
139        // BPF_STX class

142        ebpf::ST_W_REG  if !sbpf_version.move_memory_instruction_classes() => { name = "stxw";  desc = st_reg_str(name, insn); },

149        ebpf::SUB32_REG  => { name = "sub32";  desc = alu_reg_str(name, insn);  },
150        ebpf::MUL32_IMM  if !sbpf_version.enable_pqr() => { name = "mul32";  desc = alu_imm_str(name, insn);  },
151        ebpf::MUL32_REG  if !sbpf_version.enable_pqr() => { name = "mul32";  desc = alu_reg_str(name, insn);  },
152        ebpf::LD_1B_REG  if sbpf_version.move_memory_instruction_classes() => { name = "ldxb";  desc = ld_reg_str(name, insn); },
153        ebpf::DIV32_IMM  if !sbpf_version.enable_pqr() => { name = "div32";  desc = alu_imm_str(name, insn);  },
154        ebpf::DIV32_REG  if !sbpf_version.enable_pqr() => { name = "div32";  desc = alu_reg_str(name, insn);  },
155        ebpf::LD_2B_REG  if sbpf_version.move_memory_instruction_classes() => { name = "ldxh";  desc = ld_reg_str(name, insn); },
156        ebpf::OR32_IMM   => { name = "or32";   desc = alu_imm_str(name, insn);  },
157        ebpf::OR32_REG   => { name = "or32";   desc = alu_reg_str(name, insn);  },
158        ebpf::AND32_IMM  => { name = "and32";  desc = alu_imm_str(name, insn);  },
159        ebpf::AND32_REG  => { name = "and32";  desc = alu_reg_str(name, insn);  },
160        ebpf::LSH32_IMM  => { name = "lsh32";  desc = alu_imm_str(name, insn);  },
161        ebpf::LSH32_REG  => { name = "lsh32";  desc = alu_reg_str(name, insn);  },
162        ebpf::RSH32_IMM  => { name = "rsh32";  desc = alu_imm_str(name, insn);  },

172        ebpf::MOV32_REG  => { name = "mov32";  desc = alu_reg_str(name, insn);  },
173        ebpf::ARSH32_IMM => { name = "arsh32"; desc = alu_imm_str(name, insn);  },
174        ebpf::ARSH32_REG => { name = "arsh32"; desc = alu_reg_str(name, insn);  },
175        ebpf::LE         => { name = "le";     desc = byteswap_str(name, insn); },
176        ebpf::BE         => { name = "be";     desc = byteswap_str(name, insn); },
177
178        // BPF_ALU64_STORE class
179        ebpf::ADD64_IMM  => { name = "add64";  desc = alu_imm_str(name, insn); },
180        ebpf::ADD64_REG  => { name = "add64";  desc = alu_reg_str(name, insn); },
181        ebpf::SUB64_IMM  => { name = "sub64";  desc = alu_imm_str(name, insn); },
182        ebpf::SUB64_REG  => { name = "sub64";  desc = alu_reg_str(name, insn); },
183        ebpf::MUL64_IMM  if !sbpf_version.enable_pqr() => { name = "mul64";  desc = alu_imm_str(name, insn); },
184        ebpf::ST_1B_IMM  if sbpf_version.move_memory_instruction_classes() => { name = "stb";   desc = ld_st_imm_str(name, insn); },
185        ebpf::MUL64_REG  if !sbpf_version.enable_pqr() => { name = "mul64";  desc = alu_reg_str(name, insn); },
186        ebpf::ST_1B_REG  if sbpf_version.move_memory_instruction_classes() => { name = "stxb";  desc = st_reg_str(name, insn); },
187        ebpf::DIV64_IMM  if !sbpf_version.enable_pqr() => { name = "div64";  desc = alu_imm_str(name, insn); },
188        ebpf::ST_2B_IMM  if sbpf_version.move_memory_instruction_classes() => { name = "sth";   desc = ld_st_imm_str(name, insn); },
189        ebpf::DIV64_REG  if !sbpf_version.enable_pqr() => { name = "div64";  desc = alu_reg_str(name, insn); },
190        ebpf::ST_2B_REG  if sbpf_version.move_memory_instruction_classes() => { name = "stxh";  desc = st_reg_str(name, insn); },
191        ebpf::OR64_IMM   => { name = "or64";   desc = alu_imm_str(name, insn); },
192        ebpf::OR64_REG   => { name = "or64";   desc = alu_reg_str(name, insn); },
193        ebpf::AND64_IMM  => { name = "and64";  desc = alu_imm_str(name, insn); },
194        ebpf::AND64_REG  => { name = "and64";  desc = alu_reg_str(name, insn); },
195        ebpf::LSH64_IMM  => { name = "lsh64";  desc = alu_imm_str(name, insn); },
196        ebpf::LSH64_REG  => { name = "lsh64";  desc = alu_reg_str(name, insn); },
197        ebpf::RSH64_IMM  => { name = "rsh64";  desc = alu_imm_str(name, insn); },
198        ebpf::RSH64_REG  => { name = "rsh64";  desc = alu_reg_str(name, insn); },
199        ebpf::ST_4B_IMM  if sbpf_version.move_memory_instruction_classes() => { name = "stw";   desc = ld_st_imm_str(name, insn); },
200        ebpf::NEG64      if !sbpf_version.disable_neg() => { name = "neg64";  desc = format!("{} r{}", name, insn.dst); },
201        ebpf::ST_4B_REG  if sbpf_version.move_memory_instruction_classes() => { name = "stxw";  desc = st_reg_str(name, insn); },
202        ebpf::MOD64_IMM  if !sbpf_version.enable_pqr() => { name = "mod64";  desc = alu_imm_str(name, insn); },
203        ebpf::ST_8B_IMM  if sbpf_version.move_memory_instruction_classes() => { name = "stdw";  desc = ld_st_imm_str(name, insn); },
204        ebpf::MOD64_REG  if !sbpf_version.enable_pqr() => { name = "mod64";  desc = alu_reg_str(name, insn); },
205        ebpf::ST_8B_REG  if sbpf_version.move_memory_instruction_classes() => { name = "stxdw"; desc = st_reg_str(name, insn); },
206        ebpf::XOR64_IMM  => { name = "xor64";  desc = alu_imm_str(name, insn); },
207        ebpf::XOR64_REG  => { name = "xor64";  desc = alu_reg_str(name, insn); },
208        ebpf::MOV64_IMM  => { name = "mov64";  desc = alu_imm_str(name, insn); },
209        ebpf::MOV64_REG  => { name = "mov64";  desc = alu_reg_str(name, insn); },
210        ebpf::ARSH64_IMM => { name = "arsh64"; desc = alu_imm_str(name, insn); },
211        ebpf::ARSH64_REG => { name = "arsh64"; desc = alu_reg_str(name, insn); },
212        ebpf::HOR64_IMM  => { name = "hor64"; desc = alu_imm_str(name, insn); },
213
214        // BPF_PQR class
215        ebpf::LMUL32_IMM  if sbpf_version.enable_pqr() => { name = "lmul32"; desc = alu_imm_str(name, insn); },
216        ebpf::LMUL32_REG  if sbpf_version.enable_pqr() => { name = "lmul32"; desc = alu_reg_str(name, insn); },
217        ebpf::LMUL64_IMM  if sbpf_version.enable_pqr() => { name = "lmul64"; desc = alu_imm_str(name, insn); },
218        ebpf::LMUL64_REG  if sbpf_version.enable_pqr() => { name = "lmul64"; desc = alu_reg_str(name, insn); },
219        ebpf::UHMUL64_IMM if sbpf_version.enable_pqr() => { name = "uhmul64"; desc = alu_imm_str(name, insn); },
220        ebpf::UHMUL64_REG if sbpf_version.enable_pqr() => { name = "uhmul64"; desc = alu_reg_str(name, insn); },
221        ebpf::SHMUL64_IMM if sbpf_version.enable_pqr() => { name = "shmul64"; desc = alu_imm_str(name, insn); },
222        ebpf::SHMUL64_REG if sbpf_version.enable_pqr() => { name = "shmul64"; desc = alu_reg_str(name, insn); },
223        ebpf::UDIV32_IMM  if sbpf_version.enable_pqr() => { name = "udiv32"; desc = alu_imm_str(name, insn); },
224        ebpf::UDIV32_REG  if sbpf_version.enable_pqr() => { name = "udiv32"; desc = alu_reg_str(name, insn); },
225        ebpf::UDIV64_IMM  if sbpf_version.enable_pqr() => { name = "udiv64"; desc = alu_imm_str(name, insn); },
226        ebpf::UDIV64_REG  if sbpf_version.enable_pqr() => { name = "udiv64"; desc = alu_reg_str(name, insn); },
227        ebpf::UREM32_IMM  if sbpf_version.enable_pqr() => { name = "urem32"; desc = alu_imm_str(name, insn); },

231        ebpf::SDIV32_IMM  if sbpf_version.enable_pqr() => { name = "sdiv32"; desc = alu_imm_str(name, insn); },
232        ebpf::SDIV32_REG  if sbpf_version.enable_pqr() => { name = "sdiv32"; desc = alu_reg_str(name, insn); },

239
240        // BPF_JMP class
241        ebpf::JA         => {
242            name = "ja";
243            let target_pc = (insn.ptr as isize + insn.off as isize + 1) as usize;
244            desc = format!("{} {}", name, resolve_label(cfg_nodes, target_pc));
245        },
246        ebpf::JEQ_IMM    => { name = "jeq";  desc = jmp_imm_str(name, insn, cfg_nodes); },
247        ebpf::JEQ_REG    => { name = "jeq";  desc = jmp_reg_str(name, insn, cfg_nodes); },
248        ebpf::JGT_IMM    => { name = "jgt";  desc = jmp_imm_str(name, insn, cfg_nodes); },
249        ebpf::JGT_REG    => { name = "jgt";  desc = jmp_reg_str(name, insn, cfg_nodes); },
250        ebpf::JGE_IMM    => { name = "jge";  desc = jmp_imm_str(name, insn, cfg_nodes); },
251        ebpf::JGE_REG    => { name = "jge";  desc = jmp_reg_str(name, insn, cfg_nodes); },
252        ebpf::JLT_IMM    => { name = "jlt";  desc = jmp_imm_str(name, insn, cfg_nodes); },
253        ebpf::JLT_REG    => { name = "jlt";  desc = jmp_reg_str(name, insn, cfg_nodes); },
254        ebpf::JLE_IMM    => { name = "jle";  desc = jmp_imm_str(name, insn, cfg_nodes); },
255        ebpf::JLE_REG    => { name = "jle";  desc = jmp_reg_str(name, insn, cfg_nodes); },
256        ebpf::JSET_IMM   => { name = "jset"; desc = jmp_imm_str(name, insn, cfg_nodes); },
257        ebpf::JSET_REG   => { name = "jset"; desc = jmp_reg_str(name, insn, cfg_nodes); },
258        ebpf::JNE_IMM    => { name = "jne";  desc = jmp_imm_str(name, insn, cfg_nodes); },
259        ebpf::JNE_REG    => { name = "jne";  desc = jmp_reg_str(name, insn, cfg_nodes); },
260        ebpf::JSGT_IMM   => { name = "jsgt"; desc = jmp_imm_str(name, insn, cfg_nodes); },
261        ebpf::JSGT_REG   => { name = "jsgt"; desc = jmp_reg_str(name, insn, cfg_nodes); },
262        ebpf::JSGE_IMM   => { name = "jsge"; desc = jmp_imm_str(name, insn, cfg_nodes); },
263        ebpf::JSGE_REG   => { name = "jsge"; desc = jmp_reg_str(name, insn, cfg_nodes); },
264        ebpf::JSLT_IMM   => { name = "jslt"; desc = jmp_imm_str(name, insn, cfg_nodes); },
265        ebpf::JSLT_REG   => { name = "jslt"; desc = jmp_reg_str(name, insn, cfg_nodes); },
266        ebpf::JSLE_IMM   => { name = "jsle"; desc = jmp_imm_str(name, insn, cfg_nodes); },
267        ebpf::JSLE_REG   => { name = "jsle"; desc = jmp_reg_str(name, insn, cfg_nodes); },
268        ebpf::CALL_IMM   => {
269            let key = sbpf_version.calculate_call_imm_target_pc(pc, insn.imm);
270            let mut name = "call";
271            let mut function_name = function_registry.lookup_by_key(key).map(|(function_name, _)| String::from_utf8_lossy(function_name).to_string());
272            if !sbpf_version.static_syscalls() && function_name.is_none() {
273                name = "syscall";
274                function_name = loader.get_function_registry().lookup_by_key(insn.imm as u32).map(|(function_name, _)| String::from_utf8_lossy(function_name).to_string());
275            }
276            desc = format!("{} {}", name, function_name.unwrap_or_else(|| "[invalid]".to_string()));
277        },
278        ebpf::CALL_REG   => { name = "callx"; desc = format!("{} r{}", name, if sbpf_version.callx_uses_src_reg() { insn.src } else { insn.imm as u8 }); },
279        ebpf::EXIT     if !sbpf_version.static_syscalls() => { name = "exit"; desc = name.to_string(); },
280        ebpf::RETURN   if sbpf_version.static_syscalls() =>  { name = "return"; desc = name.to_string(); },
281        ebpf::SYSCALL  if sbpf_version.static_syscalls() =>  { desc = format!("syscall {}", insn.imm); },
282
283        _                => { name = "unknown"; desc = format!("{} opcode={:#x}", name, insn.opc); },
284    };
285    desc
