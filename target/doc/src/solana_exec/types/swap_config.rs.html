1use solana_central::PoolTrait;
2use solana_central::SwapDirection;
3use solana_sdk::pubkey::Pubkey;
4use std::sync::{Arc, RwLock};

6/// Configuration for a single swap operation. Contains all information needed to execute a swap
7/// instruction. Multiple swap configs can be combined to build multi-hop swap transactions.
8#[derive(Clone)]
9pub struct SwapConfig {
10  /// The pool to execute the swap on
11  pub pool: Arc<RwLock<dyn PoolTrait>>,
12  /// Direction of the swap (A to B or B to A)
13  pub direction: SwapDirection,
14  /// Amount of input token to swap, in token units. For example, if swapping token B for token A,
15  /// this is the amount of token B to send to the pool in exchange for token A..
16  pub amount_in: u64,
17  /// Slippage tolerance in lamports (10^9 lamports = 1). Set to $10^9$ for infinite slippage. The
18  /// slippage calculation for AMMs: `min_amount_out = (b_reserves / a_reserves) * a_amount_in *
19  /// (1 - swap_fee) * (1 - slippage_lp / 10^9)`
20  pub slippage_lp: u64,
21  /// Whether to open the output token account before executing the swap. Set to `true` if buying a
22  /// token for the first time and the token account on the wallet hasn't been opened yet.
23  pub open_token_account: bool,
24  /// Whether to close the input token account after the swap. Set to `true` when swapping all
25  /// remaining balance and no longer need the account and would like to redeem the rent paid for
26  /// it.
27  pub close_token_account: bool,
28  /// Public key of the wallet executing the swap
29  pub wallet: Pubkey,
30}
