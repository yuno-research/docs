1use solana_central::constants::PUMP_CONSTANTS;
2use solana_central::Instruction;
3
4/// Determine if a Solana instruction is a Pumpfun bonding curve creation instruction. Checks data
5/// length, account count, discriminator, and program ID to identify Pumpfun bonding curve creation
6/// instructions.
7pub fn is_pf_bonding_curve_create_instruction(instruction: &Instruction) -> bool {
8  /*
9  Data can be of any unbounded size due to string in it, but the minimum size will be 40 since it
10  has to contain the creator address pubkey and the instruction discriminator
11  */
12  if instruction.data.len() < 40 {
13    return false;
14  }
15  // Length of accounts is minimum 14 for this instruction
16  if instruction.accounts.len() < 14 {
17    return false;
18  }
19  // check data discriminator
20  for i in 0..PUMP_CONSTANTS.bonding_curve_create_instruction_discriminator.len() {
21    if instruction.data[i] != PUMP_CONSTANTS.bonding_curve_create_instruction_discriminator[i] {
22      return false;
23    }
24  }
25  // check program id matches
26  if instruction.tx_account_keys[instruction.program_id_index as usize] != PUMP_CONSTANTS.bonding_curve_program {
27    return false;
28  }
29  // All checks for bonding curve create instruction passed
30  true
31}
