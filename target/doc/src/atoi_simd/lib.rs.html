1//! # Fast `&[u8]` to integer parser
2//!
3//! SIMD (fast) parsing is supported on x86_64 (SSE4.1, AVX2) and on Arm64 (aarch64, Neon),
4//! but this library works even if you don't have a SIMD supported cpu (and it will be still faster than str::parse).
5//!
6//! Supports negative values and validates the input.
7//!
8//! Supported output types: u8, i8, u16, i16, u32, i32, u64, i64, u128, i128, usize, isize.
9//!
10//! Has good test coverage, and can be considered safe.
11//!
12//! To enable SIMD it needs the `target-feature` or `target-cpu` flags set, or it will fallback to non-SIMD functions.
13//! You can copy the `./.cargo/config.toml` to your project, or use one of the following environment variables:
14//!
15//! -   `RUSTFLAGS="-C target-feature=+sse2,+sse3,+sse4.1,+ssse3,+avx,+avx2"` for x86_64
16//!
17//! -   `RUSTFLAGS="-C target-feature=+neon"` for Arm64
18//!
19//! -   `RUSTFLAGS="-C target-cpu=native"` will optimize for your current cpu
20//!
21//! If you have `&str` then use `.as_bytes()`
22//!
23//! Supports `no_std` with `--no-default-features`
24//!
25//! # Examples
26//!
27//! ```
28//! let val: u64 = atoi_simd::parse(b"1234").unwrap();
29//! assert_eq!(val, 1234_u64);
30//!
31//! assert_eq!(atoi_simd::parse::<i64>(b"-2345"), Ok(-2345_i64));
32//!
33//! assert_eq!(atoi_simd::parse_any::<u64>(b"123something_else"), Ok((123_u64, 3)));
34//!
35//! // a drop-in replacement for `str::parse`
36//! assert_eq!(atoi_simd::parse_skipped::<u64>(b"+000000000000000000001234"), Ok(1234_u64));
37//! ```

73/// Parses a slice of digits, and checks the first '-' char for signed integers.
74#[inline]
75pub fn parse<T: Parse>(s: &[u8]) -> Result<T, AtoiSimdError> {
76    T::atoi_simd_parse(s)
77}
78
79/// Parses a positive integer.
80#[inline]
81pub fn parse_pos<T: Parse>(s: &[u8]) -> Result<T, AtoiSimdError> {
82    T::atoi_simd_parse_pos(s)
83}
84
85/// Parses a negative integer. Slice must not contain '-' sign.
86#[inline]
87pub fn parse_neg<T: ParseNeg>(s: &[u8]) -> Result<T, AtoiSimdError> {
88    T::atoi_simd_parse_neg(s)
89}
90
91/// Parses a slice of digits until it reaches invalid character, and checks the first '-' char for signed integers.
92/// Returns the parsed value and the parsed size of the slice.
93#[inline]
94pub fn parse_any<T: Parse>(s: &[u8]) -> Result<(T, usize), AtoiSimdError> {
95    T::atoi_simd_parse_any(s)
96}
97
98/// Parses a positive integer until it reaches an invalid character.
99/// Returns the parsed value and the parsed size of the slice.
100#[inline]
101pub fn parse_any_pos<T: Parse>(s: &[u8]) -> Result<(T, usize), AtoiSimdError> {
102    T::atoi_simd_parse_any_pos(s)
103}
104
105/// Parses a negative integer until it reaches an invalid character. Slice must not contain '-' sign.
106/// Returns the parsed value and the parsed size of the slice.
107#[inline]
108pub fn parse_any_neg<T: ParseNeg>(s: &[u8]) -> Result<(T, usize), AtoiSimdError> {
109    T::atoi_simd_parse_any_neg(s)
110}

112/// Parses a slice of digits. Has been made to be used as a drop-in replacement for `str::parse`.
113/// Checks the first '-' char for signed integers.
114/// Skips the '+' char and extra zeroes at the beginning.
115/// It's slower than `parse()`.
116#[inline]
117pub fn parse_skipped<T: Parse>(s: &[u8]) -> Result<T, AtoiSimdError> {
118    T::atoi_simd_parse_skipped(s)
119}
120
121#[deprecated(since = "0.16.0", note = "Use `parse_any` instead")]
122#[inline]
123pub fn parse_until_invalid<T: Parse>(s: &[u8]) -> Result<(T, usize), AtoiSimdError> {
124    parse_any(s)
125}

127#[deprecated(since = "0.16.0", note = "Use `parse_any_pos` instead")]
128#[inline]
129pub fn parse_until_invalid_pos<T: Parse>(s: &[u8]) -> Result<(T, usize), AtoiSimdError> {
130    parse_any_pos(s)
131}
