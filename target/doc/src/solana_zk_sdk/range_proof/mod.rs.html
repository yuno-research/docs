4//! The Bulletproofs range-proof implementation over Curve25519 Ristretto points.
5//!
6//! The implementation is based on the dalek-cryptography bulletproofs
7//! [implementation](https://github.com/dalek-cryptography/bulletproofs). Compared to the original
8//! implementation by dalek-cryptography:
9//! - This implementation focuses on the range proof implementation, while the dalek-cryptography
10//!   crate additionally implements the general bulletproofs implementation for languages that can be
11//!   represented by arithmetic circuits as well as MPC.
12//! - This implementation implements a non-interactive range proof aggregation that is specified in
13//!   the original Bulletproofs [paper](https://eprint.iacr.org/2017/1066) (Section 4.3).
14

76/// A Bulletproofs range proof.
77#[allow(non_snake_case)]
78#[cfg(not(target_os = "solana"))]
79#[derive(Clone)]
80pub struct RangeProof {
81    /// A commitment to the bit-vectors `a_L` and `a_R`.
82    pub(crate) A: CompressedRistretto, // 32 bytes
83    /// A commitment to the blinding vectors `s_L` and `s_R`.
84    pub(crate) S: CompressedRistretto, // 32 bytes
85    /// A commitment to the `t_1` coefficient of the polynomial `t(x)`.
86    pub(crate) T_1: CompressedRistretto, // 32 bytes
87    /// A commitment to the `t_2` coefficient of the polynomial `t(x)`.
88    pub(crate) T_2: CompressedRistretto, // 32 bytes
89    /// The evaluation of the polynomial `t(x)` at the challenge point `x`.
90    pub(crate) t_x: Scalar, // 32 bytes
91    /// The blinding factor for the `t_x` value.
92    pub(crate) t_x_blinding: Scalar, // 32 bytes
93    /// The blinding factor for the synthetic commitment to `l(x)` and `r(x)`.
94    pub(crate) e_blinding: Scalar, // 32 bytes
95    /// The inner product proof.
96    pub(crate) ipp_proof: InnerProductProof, // 448 bytes for withdraw; 512 for transfer
97}

99#[allow(non_snake_case)]
100#[cfg(not(target_os = "solana"))]
101impl RangeProof {
102    /// Creates an aggregated range proof for a set of values that are committed to a set of
103    /// Pedersen commitments.
104    ///
105    /// This function implements the aggregated proof generation logic from Section 4.3
106    /// of the Bulletproofs paper. It allows proving that multiple values are in their
107    /// respective ranges, creating one proof that is much smaller than the sum of
108    /// individual proofs.
109    ///
110    /// # Panics
111    /// This function will panic if the `openings` vector does not contain the same number
112    /// of elements as the `amounts` and `bit_lengths` vectors.
113    #[allow(clippy::many_single_char_names)]
114    #[cfg(not(target_os = "solana"))]
115    pub fn new(
116        amounts: Vec<u64>,
117        bit_lengths: Vec<usize>,
118        openings: Vec<&PedersenOpening>,
119        transcript: &mut Transcript,
120    ) -> Result<Self, RangeProofGenerationError> {
121        // 1. Validate inputs
122        let m = amounts.len();
123        if bit_lengths.len() != m || openings.len() != m {
124            return Err(RangeProofGenerationError::VectorLengthMismatch);
125        }
126
127        // each bit length must be greater than 0 for the proof to make sense
128        if bit_lengths
129            .iter()
130            .any(|bit_length| *bit_length == 0 || *bit_length > u64::BITS as usize)
131        {
132            return Err(RangeProofGenerationError::InvalidBitSize);
133        }
134
135        // total vector dimension to compute the ultimate inner product proof for
136        let nm: usize = bit_lengths.iter().sum();
137        if !nm.is_power_of_two() {
138            return Err(RangeProofGenerationError::VectorLengthMismatch);
139        }
140
141        let bp_gens = RangeProofGens::new(nm)
142            .map_err(|_| RangeProofGenerationError::MaximumGeneratorLengthExceeded)?;
143
144        transcript.range_proof_domain_separator(nm as u64);
145
146        // 2. Create commitments A and S.
147        // A is a commitment to the bit-vectors a_L and a_R
148        let a_blinding = Scalar::random(&mut OsRng);
149        let mut A = a_blinding * &(*H);
150
151        let mut gens_iter = bp_gens.G(nm).zip(bp_gens.H(nm));
152        for (amount_i, n_i) in amounts.iter().zip(bit_lengths.iter()) {
153            for j in 0..(*n_i) {
154                let (G_ij, H_ij) = gens_iter.next().unwrap();
155
156                // `j` is guaranteed to be at most `u64::BITS` (a 6-bit number) and therefore,
157                // casting is lossless and right shift can be safely unwrapped
158                let v_ij = Choice::from((amount_i.checked_shr(j as u32).unwrap() & 1) as u8);
159                let mut point = -H_ij;
160                // Add G_ij if bit is 1, else do nothing (since a_R = a_L - 1)
161                point.conditional_assign(G_ij, v_ij);
162                A += point;
163            }
164        }
165        let A = A.compress();
166
167        // generate blinding factors and generate their Pedersen vector commitment
168        let s_L: Vec<Scalar> = (0..nm).map(|_| Scalar::random(&mut OsRng)).collect();
169        let s_R: Vec<Scalar> = (0..nm).map(|_| Scalar::random(&mut OsRng)).collect();
170
171        // generate blinding factor for Pedersen commitment; `s_blinding` should not to be confused
172        // with blinding factors for the actual inner product vector
173        let s_blinding = Scalar::random(&mut OsRng);
174
175        let S = RistrettoPoint::multiscalar_mul(
176            iter::once(&s_blinding).chain(s_L.iter()).chain(s_R.iter()),
177            iter::once(&(*H)).chain(bp_gens.G(nm)).chain(bp_gens.H(nm)),
178        )
179        .compress();
180
181        // 3. Derive challenges y and z.
182        transcript.append_point(b"A", &A);
183        transcript.append_point(b"S", &S);
184
185        let y = transcript.challenge_scalar(b"y");
186        let z = transcript.challenge_scalar(b"z");
187
188        // 4. Construct the blinded vector polynomials l(x) and r(x).
189        //    l(x) = (a_L - z*1) + s_L*x
190        //    r(x) = y^nm o (a_R + z*1 + s_R*x) + (z^2*2^n_1 || ... || z^{m+1}*2^n_m)
191        //    where `o` is the Hadamard product and `||` is vector concatenation.
192        let mut l_poly = util::VecPoly1::zero(nm);
193        let mut r_poly = util::VecPoly1::zero(nm);
194
195        let mut i = 0;
196        let mut exp_z = z * z;
197        let mut exp_y = Scalar::ONE;
198
199        for (amount_i, n_i) in amounts.iter().zip(bit_lengths.iter()) {
200            let mut exp_2 = Scalar::ONE;
201
202            for j in 0..(*n_i) {
203                // `j` is guaranteed to be at most `u64::BITS` (a 6-bit number) and therefore,
204                // casting is lossless and right shift can be safely unwrapped
205                let a_L_j = Scalar::from(amount_i.checked_shr(j as u32).unwrap() & 1);
206                let a_R_j = a_L_j - Scalar::ONE;
207
208                l_poly.0[i] = a_L_j - z;
209                l_poly.1[i] = s_L[i];
210                r_poly.0[i] = exp_y * (a_R_j + z) + exp_z * exp_2;
211                r_poly.1[i] = exp_y * s_R[i];
212
213                exp_y *= y;
214                exp_2 = exp_2 + exp_2;
215
216                // `i` is capped by the sum of vectors in `bit_lengths`
217                i = i.checked_add(1).unwrap();
218            }
219            exp_z *= z;
220        }
221
222        // 5. Compute the inner product polynomial t(x) = <l(x), r(x)>.
223        let t_poly = l_poly
224            .inner_product(&r_poly)
225            .ok_or(RangeProofGenerationError::InnerProductLengthMismatch)?;
226
227        // 6. Commit to the t_1 and t_2 coefficients of t(x).
228        let (T_1, t_1_blinding) = Pedersen::new(t_poly.1);
229        let (T_2, t_2_blinding) = Pedersen::new(t_poly.2);
230
231        let T_1 = T_1.get_point().compress();
232        let T_2 = T_2.get_point().compress();
233
234        transcript.append_point(b"T_1", &T_1);
235        transcript.append_point(b"T_2", &T_2);
236
237        // 7. Derive challenge x and compute openings.
238        let x = transcript.challenge_scalar(b"x");
239
240        // Compute the aggregated blinding factor for all value commitments.
241        let mut agg_opening = Scalar::ZERO;
242        let mut exp_z = z;
243        for opening in openings {
244            exp_z *= z;
245            agg_opening += exp_z * opening.get_scalar();
246        }
247
248        let t_blinding_poly = util::Poly2(
249            agg_opening,
250            *t_1_blinding.get_scalar(),
251            *t_2_blinding.get_scalar(),
252        );
253
254        let t_x = t_poly.eval(x);
255        let t_x_blinding = t_blinding_poly.eval(x);
256
257        transcript.append_scalar(b"t_x", &t_x);
258        transcript.append_scalar(b"t_x_blinding", &t_x_blinding);
259
260        // homomorphically compuate the openings for A + x*S
261        let e_blinding = a_blinding + s_blinding * x;
262
263        // 8. Finally, create the inner product proof.
264        let l_vec = l_poly.eval(x);
265        let r_vec = r_poly.eval(x);
266
267        transcript.append_scalar(b"e_blinding", &e_blinding);
268
269        // compute the inner product argument on the commitment:
270        // P = <l(x), G> + <r(x), H'> + <l(x), r(x)>*Q
271        let w = transcript.challenge_scalar(b"w");
272        let Q = w * &G;
273
274        let G_factors: Vec<Scalar> = iter::repeat_n(Scalar::ONE, nm).collect();
275        let H_factors: Vec<Scalar> = util::exp_iter(y.invert()).take(nm).collect();
276
277        // this variable exists for backwards compatibility
278        let _c = transcript.challenge_scalar(b"c");
279
280        let ipp_proof = InnerProductProof::new(
281            &Q,
282            &G_factors,
283            &H_factors,
284            bp_gens.G(nm).cloned().collect(),
285            bp_gens.H(nm).cloned().collect(),
286            l_vec,
287            r_vec,
288            transcript,
289        )?;
290
291        // compute challenge `d` for consistency with the verifier
292        transcript.append_scalar(b"ipp_a", &ipp_proof.a);
293        transcript.append_scalar(b"ipp_b", &ipp_proof.b);
294        let _d = transcript.challenge_scalar(b"d");
295
296        Ok(RangeProof {
297            A,
298            S,
299            T_1,
300            T_2,
301            t_x,
302            t_x_blinding,
303            e_blinding,
304            ipp_proof,
305        })
306    }

308    /// Verifies an aggregated range proof for a set of commitments.
309    ///
310    /// This function implements the verifier's logic, which is optimized into a
311    /// single large multiscalar multiplication (`mega_check`) for efficiency. This
312    /// check simultaneously verifies all aspects of the proof.
313    #[allow(clippy::many_single_char_names)]
314    pub fn verify(
315        &self,
316        comms: Vec<&PedersenCommitment>,
317        bit_lengths: Vec<usize>,
318        transcript: &mut Transcript,
319    ) -> Result<(), RangeProofVerificationError> {
320        // 1. Validate inputs and reconstruct challenges from the transcript.
321        if comms.len() != bit_lengths.len() {
322            return Err(RangeProofVerificationError::VectorLengthMismatch);
323        }
324
325        let m = bit_lengths.len();
326        let nm: usize = bit_lengths.iter().sum();
327        let bp_gens = RangeProofGens::new(nm)
328            .map_err(|_| RangeProofVerificationError::MaximumGeneratorLengthExceeded)?;
329
330        if !nm.is_power_of_two() {
331            return Err(RangeProofVerificationError::InvalidBitSize);
332        }
333
334        transcript.range_proof_domain_separator(nm as u64);
335
336        // append proof data to transcript and derive appropriate challenge scalars
337        transcript.validate_and_append_point(b"A", &self.A)?;
338        transcript.validate_and_append_point(b"S", &self.S)?;
339
340        let y = transcript.challenge_scalar(b"y");
341        let z = transcript.challenge_scalar(b"z");
342
343        let zz = z * z;
344        let minus_z = -z;
345
346        transcript.validate_and_append_point(b"T_1", &self.T_1)?;
347        transcript.validate_and_append_point(b"T_2", &self.T_2)?;
348
349        let x = transcript.challenge_scalar(b"x");
350
351        transcript.append_scalar(b"t_x", &self.t_x);
352        transcript.append_scalar(b"t_x_blinding", &self.t_x_blinding);
353        transcript.append_scalar(b"e_blinding", &self.e_blinding);
354
355        let w = transcript.challenge_scalar(b"w");
356
357        // this variable exists for backwards compatibility
358        let _c = transcript.challenge_scalar(b"c");
359
360        // 2. Compute the scalars for the verification equation.
361        let (x_sq, x_inv_sq, s) = self.ipp_proof.verification_scalars(nm, transcript)?;
362        let s_inv = s.iter().rev();
363
364        let a = self.ipp_proof.a;
365        let b = self.ipp_proof.b;
366
367        transcript.append_scalar(b"ipp_a", &a);
368        transcript.append_scalar(b"ipp_b", &b);
369
370        // Challenge for batching the main algebraic relation checks
371        let d = transcript.challenge_scalar(b"d");
372
373        // 3. Construct the scalars for the single large multiscalar multiplication.
374
375        // This vector is used in the `h` terms of the final check.
376        // It's a concatenation of powers-of-2 vectors, each scaled by a power of z.
377        // Formula: z^0*2^n_0 || z^1*2^n_1 || ... || z^{m-1}*2^n_{m-1}
378        let concat_z_and_2: Vec<Scalar> = util::exp_iter(z)
379            .zip(bit_lengths.iter())
380            .flat_map(|(exp_z, n_i)| {
381                util::exp_iter(Scalar::from(2u64))
382                    .take(*n_i)
383                    .map(move |exp_2| exp_2 * exp_z)
384            })
385            .collect();
386
387        let gs = s.iter().map(|s_i| minus_z - a * s_i);
388        let hs = s_inv
389            .zip(util::exp_iter(y.invert()))
390            .zip(concat_z_and_2.iter())
391            .map(|((s_i_inv, exp_y_inv), z_and_2)| z + exp_y_inv * (zz * z_and_2 - b * s_i_inv));
392
393        let basepoint_scalar =
394            w * (self.t_x - a * b) + d * (delta(&bit_lengths, &y, &z) - self.t_x);
395        let value_commitment_scalars = util::exp_iter(z).take(m).map(|z_exp| d * zz * z_exp);
396
397        // 4. Perform the final "mega-check"
398        // This single multiscalar multiplication verifies all relations simultaneously.
399        let mega_check = RistrettoPoint::optional_multiscalar_mul(
400            iter::once(Scalar::ONE)
401                .chain(iter::once(x))
402                .chain(iter::once(d * x))
403                .chain(iter::once(d * x * x))
404                .chain(iter::once(-self.e_blinding - d * self.t_x_blinding))
405                .chain(iter::once(basepoint_scalar))
406                .chain(x_sq.iter().cloned())
407                .chain(x_inv_sq.iter().cloned())
408                .chain(gs)
409                .chain(hs)
410                .chain(value_commitment_scalars),
411            iter::once(self.A.decompress())
412                .chain(iter::once(self.S.decompress()))
413                .chain(iter::once(self.T_1.decompress()))
414                .chain(iter::once(self.T_2.decompress()))
415                .chain(iter::once(Some(*H)))
416                .chain(iter::once(Some(G)))
417                .chain(self.ipp_proof.L_vec.iter().map(|L| L.decompress()))
418                .chain(self.ipp_proof.R_vec.iter().map(|R| R.decompress()))
419                .chain(bp_gens.G(nm).map(|&x| Some(x)))
420                .chain(bp_gens.H(nm).map(|&x| Some(x)))
421                .chain(comms.iter().map(|V| Some(*V.get_point()))),
422        )
423        .ok_or(RangeProofVerificationError::MultiscalarMul)?;
424
425        if mega_check.is_identity() {
426            Ok(())
427        } else {
428            Err(RangeProofVerificationError::AlgebraicRelation)
429        }
430    }

432    // Following the dalek rangeproof library signature for now. The exact method signature can be

434    pub fn to_bytes(&self) -> Vec<u8> {
435        let mut buf = Vec::with_capacity(7 * 32 + self.ipp_proof.serialized_size());
436        buf.extend_from_slice(self.A.as_bytes());
437        buf.extend_from_slice(self.S.as_bytes());
438        buf.extend_from_slice(self.T_1.as_bytes());
439        buf.extend_from_slice(self.T_2.as_bytes());
440        buf.extend_from_slice(self.t_x.as_bytes());
441        buf.extend_from_slice(self.t_x_blinding.as_bytes());
442        buf.extend_from_slice(self.e_blinding.as_bytes());
443        buf.extend_from_slice(&self.ipp_proof.to_bytes());
444        buf
445    }

447    // Following the dalek rangeproof library signature for now. The exact method signature can be
448    // changed.
449    pub fn from_bytes(slice: &[u8]) -> Result<RangeProof, RangeProofVerificationError> {
450        if slice.len() % 32 != 0 {
451            return Err(RangeProofVerificationError::Deserialization);
452        }
453        if slice.len() < 7 * 32 {
454            return Err(RangeProofVerificationError::Deserialization);
455        }
456
457        let A = CompressedRistretto(util::read32(&slice[0..]));
458        let S = CompressedRistretto(util::read32(&slice[32..]));
459        let T_1 = CompressedRistretto(util::read32(&slice[2 * 32..]));
460        let T_2 = CompressedRistretto(util::read32(&slice[3 * 32..]));
461
462        let t_x = Scalar::from_canonical_bytes(util::read32(&slice[4 * 32..]))
463            .into_option()
464            .ok_or(RangeProofVerificationError::Deserialization)?;
465        let t_x_blinding = Scalar::from_canonical_bytes(util::read32(&slice[5 * 32..]))
466            .into_option()
467            .ok_or(RangeProofVerificationError::Deserialization)?;
468        let e_blinding = Scalar::from_canonical_bytes(util::read32(&slice[6 * 32..]))
469            .into_option()
470            .ok_or(RangeProofVerificationError::Deserialization)?;
471
472        let ipp_proof = InnerProductProof::from_bytes(&slice[7 * 32..])?;
473
474        Ok(RangeProof {
475            A,
476            S,
477            T_1,
478            T_2,
479            t_x,
480            t_x_blinding,
481            e_blinding,
482            ipp_proof,
483        })
484    }

487/// Computes the `delta(y,z)` term for the verification equation.
488///
489/// This term is a function of the challenges `y` and `z` and the proof dimensions.
490/// It is needed to correctly aggregate the polynomial checks.
491/// The formula is: `delta(y,z) = (z - z^2) * <1, y^nm> - sum_{j=0}^{m-1} z^(j+3) * <1, 2^n_j>`
492#[cfg(not(target_os = "solana"))]
493fn delta(bit_lengths: &[usize], y: &Scalar, z: &Scalar) -> Scalar {
494    let nm: usize = bit_lengths.iter().sum();
495    let sum_y = util::sum_of_powers(y, nm);
496
497    let mut agg_delta = (z - z * z) * sum_y;
498    let mut exp_z = z * z * z;
499    for n_i in bit_lengths.iter() {
500        let sum_2 = util::sum_of_powers(&Scalar::from(2u64), *n_i);
501        agg_delta -= exp_z * sum_2;
502        exp_z *= z;
503    }
504    agg_delta
505}
