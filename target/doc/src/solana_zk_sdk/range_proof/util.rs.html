2//! This module provides common mathematical operations over scalars and vectors needed to
3//! construct and verify Bulletproofs.

7/// Represents a degree-1 vector polynomial, such as `a + b*x`.
8///
9/// A vector polynomial is a vector of polynomials. For example:
10/// `[a_1 + b_1*x, a_2 + b_2*x, ..., a_n + b_n*x]`
11///
12/// This struct stores the constant terms (the `a_i`'s) and the linear terms (the `b_i`'s) as two
13/// separate vectors of scalars. In the Bulletproofs protocol, this is used to represent the `l(x)`
14/// and `r(x)` vector polynomials.
15pub struct VecPoly1(pub Vec<Scalar>, pub Vec<Scalar>);
16
17impl VecPoly1 {
18    /// Creates a new zero-filled `VecPoly1` of the given size.
19    pub fn zero(n: usize) -> Self {
20        VecPoly1(vec![Scalar::ZERO; n], vec![Scalar::ZERO; n])
21    }
22
23    /// Computes the inner product of two vector polynomials.
24    ///
25    /// The result is a scalar polynomial of degree 2, `t(x) = t_0 + t_1*x + t_2*x^2`,
26    /// which is the inner product `t(x) = <l(x), r(x)>` in the Bulletproofs protocol.
27    ///
28    /// This function uses Karatsuba's method for efficient polynomial multiplication.
29    pub fn inner_product(&self, rhs: &VecPoly1) -> Option<Poly2> {
30        // Uses Karatsuba's method
31        let l = self;
32        let r = rhs;
33
34        let t0 = inner_product(&l.0, &r.0)?;
35        let t2 = inner_product(&l.1, &r.1)?;
36
37        let l0_plus_l1 = add_vec(&l.0, &l.1);
38        let r0_plus_r1 = add_vec(&r.0, &r.1);
39
40        let t1 = inner_product(&l0_plus_l1, &r0_plus_r1)? - t0 - t2;
41
42        Some(Poly2(t0, t1, t2))
43    }
44
45    /// Evaluates the vector polynomial at a given scalar point `x`.
46    ///
47    /// The result is a vector of scalars, where each element is the evaluation
48    /// of the corresponding polynomial at `x`.
49    pub fn eval(&self, x: Scalar) -> Vec<Scalar> {
50        let n = self.0.len();
51        let mut out = vec![Scalar::ZERO; n];
52        #[allow(clippy::needless_range_loop)]
53        for i in 0..n {
54            out[i] = self.0[i] + self.1[i] * x;
55        }
56        out
57    }
58}

60/// Represents a degree-2 scalar polynomial `a + b*x + c*x^2`.
61///
62/// In the Bulletproofs protocol, this represents the `t(x)` polynomial
63/// that results from the inner product of `l(x)` and `r(x)`.
64pub struct Poly2(pub Scalar, pub Scalar, pub Scalar);
65
66impl Poly2 {
67    /// Evaluates the polynomial at a scalar point `x` using Horner's method.
68    pub fn eval(&self, x: Scalar) -> Scalar {
69        self.0 + x * (self.1 + x * self.2)
70    }
71}

73/// Provides an iterator over the powers of a `Scalar` (e.g., `1, x, x^2, x^3, ...`).
74///
75/// This struct is created by the `exp_iter` function.
76pub struct ScalarExp {
77    x: Scalar,
78    next_exp_x: Scalar,
79}
80
81impl Iterator for ScalarExp {
82    type Item = Scalar;
83
84    fn next(&mut self) -> Option<Scalar> {
85        let exp_x = self.next_exp_x;
86        self.next_exp_x *= self.x;
87        Some(exp_x)
88    }
89
90    fn size_hint(&self) -> (usize, Option<usize>) {
91        (usize::MAX, None)
92    }
93}
94
95/// Returns an iterator of the powers of `x`, starting with `x^0 = 1`.
96pub fn exp_iter(x: Scalar) -> ScalarExp {
97    let next_exp_x = Scalar::ONE;
98    ScalarExp { x, next_exp_x }
99}

101/// Computes the component-wise sum of two vectors of scalars.
102/// Panics if the lengths of the vectors two do not match.
103pub fn add_vec(a: &[Scalar], b: &[Scalar]) -> Vec<Scalar> {
104    if a.len() != b.len() {
105        panic!("lengths of vectors don't match for vector addition");
106    }
107    let mut out = vec![Scalar::ZERO; b.len()];
108    for i in 0..a.len() {
109        out[i] = a[i] + b[i];
110    }
111    out
112}

114/// Returns the first 32 bytes of a slice as a fixed-size array.
115/// Panics if the slice is less than 32 bytes long.
116pub fn read32(data: &[u8]) -> [u8; 32] {
117    let mut buf32 = [0u8; 32];
118    buf32[..].copy_from_slice(&data[..32]);
119    buf32
120}

122/// Computes the inner product of two vectors of scalars.
123///
124/// The inner product is defined as:
125/// `<a, b> = a_1*b_1 + a_2*b_2 + ... + a_n*b_n`
126///
127/// Returns `None` if the vectors have different lengths.
128pub fn inner_product(a: &[Scalar], b: &[Scalar]) -> Option<Scalar> {
129    let mut out = Scalar::ZERO;
130    if a.len() != b.len() {
131        return None;
132    }
133    for i in 0..a.len() {
134        out += a[i] * b[i];
135    }
136    Some(out)
137}

139/// Computes the sum of powers of a scalar `x`.
140///
141/// Calculates `1 + x + x^2 + ... + x^(n-1)`.
142///
143/// In the Bulletproofs protocol, vector spaces are padded to a power of two `n`,
144/// so this function uses an efficient algorithm for that case. If `n` is not a
145/// power of two, it falls back to a slower method.
146pub fn sum_of_powers(x: &Scalar, n: usize) -> Scalar {
147    if !n.is_power_of_two() {
148        return sum_of_powers_slow(x, n);
149    }
150    if n == 0 || n == 1 {
151        return Scalar::from(n as u64);
152    }
153    let mut m = n;
154    let mut result = Scalar::ONE + x;
155    let mut factor = *x;
156    while m > 2 {
157        factor = factor * factor;
158        result = result + factor * result;
159        m /= 2;
160    }
161    result
162}
163
164/// A straightforward method to compute the sum of powers of `x`.
165fn sum_of_powers_slow(x: &Scalar, n: usize) -> Scalar {
166    exp_iter(*x).take(n).sum()
