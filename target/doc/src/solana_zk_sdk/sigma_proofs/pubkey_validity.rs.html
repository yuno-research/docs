1//! The public-key (validity) proof system.
2//!
3//! The protocol guarantees computational soundness (by the hardness of discrete log) and perfect
4//! zero-knowledge in the random oracle model.

34/// Byte length of a public key validity proof.
35const PUBKEY_VALIDITY_PROOF_LEN: usize = UNIT_LEN * 2;
36
37/// Public-key proof.
38///
39/// Contains all the elliptic curve and scalar components that make up the sigma protocol.
40#[cfg_attr(target_arch = "wasm32", wasm_bindgen)]
41#[allow(non_snake_case)]
42#[derive(Clone)]
43pub struct PubkeyValidityProof {
44    Y: CompressedRistretto,
45    z: Scalar,
46}
47
48#[allow(non_snake_case)]
49#[cfg(not(target_os = "solana"))]
50impl PubkeyValidityProof {
51    /// Creates a public key validity proof.
52    ///
53    /// The function does *not* hash the public key into the transcript. For
54    /// security, the caller (the main protocol) should hash these public key components prior to
55    /// invoking this constructor.
56    ///
57    /// This function is randomized. It uses `OsRng` internally to generate random scalars.
58    ///
59    /// This function panics if the provided keypair is not valid (i.e. secret key is not
60    /// invertible).
61    ///
62    /// * `elgamal_keypair` = The ElGamal keypair that pertains to the ElGamal public key to be
63    ///   proved
64    /// * `transcript` - The transcript that does the bookkeeping for the Fiat-Shamir heuristic
65    pub fn new(elgamal_keypair: &ElGamalKeypair, transcript: &mut Transcript) -> Self {
66        transcript.pubkey_proof_domain_separator();
67
68        // extract the relevant scalar and Ristretto points from the input
69        let s = elgamal_keypair.secret().get_scalar();
70
71        assert!(s != &Scalar::ZERO);
72        let mut s_inv = s.invert();
73
74        // generate a random masking factor that also serves as a nonce
75        let mut y = Scalar::random(&mut OsRng);
76        let Y = (&y * &(*H)).compress();
77
78        // record masking factors in transcript and get challenges
79        transcript.append_point(b"Y", &Y);
80        let c = transcript.challenge_scalar(b"c");
81
82        // compute masked secret key
83        let z = &(&c * s_inv) + &y;
84
85        // zeroize all sensitive non-reference variables
86        s_inv.zeroize();
87        y.zeroize();
88
89        Self { Y, z }
90    }
91
92    /// Verifies a public key validity proof. The function rejects identity public keys
93    /// even if the verifying algebraic relation holds.
94    ///
95    /// * `elgamal_pubkey` - The ElGamal public key to be proved
96    /// * `transcript` - The transcript that does the bookkeeping for the Fiat-Shamir heuristic
97    pub fn verify(
98        self,
99        elgamal_pubkey: &ElGamalPubkey,
100        transcript: &mut Transcript,
101    ) -> Result<(), PubkeyValidityProofVerificationError> {
102        transcript.pubkey_proof_domain_separator();
103
104        // extract the relevant scalar and Ristretto points from the input
105        let P = elgamal_pubkey.get_point();
106
107        if P.is_identity() {
108            return Err(SigmaProofVerificationError::PubkeyIsIdentity.into());
109        }
110
111        // include Y to transcript and extract challenge
112        transcript.validate_and_append_point(b"Y", &self.Y)?;
113        let c = transcript.challenge_scalar(b"c");
114
115        // check that the required algebraic condition holds
116        let Y = self
117            .Y
118            .decompress()
119            .ok_or(SigmaProofVerificationError::Deserialization)?;
120
121        let check = RistrettoPoint::vartime_multiscalar_mul(
122            vec![&self.z, &(-&c), &(-&Scalar::ONE)],
123            vec![&(*H), P, &Y],
124        );
125
126        if check.is_identity() {
127            Ok(())
128        } else {
129            Err(SigmaProofVerificationError::AlgebraicRelation.into())
130        }
131    }
132
133    pub fn to_bytes(&self) -> [u8; PUBKEY_VALIDITY_PROOF_LEN] {
134        let mut buf = [0_u8; PUBKEY_VALIDITY_PROOF_LEN];
135        let mut chunks = buf.chunks_mut(UNIT_LEN);
136        chunks.next().unwrap().copy_from_slice(self.Y.as_bytes());
137        chunks.next().unwrap().copy_from_slice(self.z.as_bytes());
138        buf
139    }
140
141    pub fn from_bytes(bytes: &[u8]) -> Result<Self, PubkeyValidityProofVerificationError> {
142        let mut chunks = bytes.chunks(UNIT_LEN);
143        let Y = ristretto_point_from_optional_slice(chunks.next())?;
144        let z = canonical_scalar_from_optional_slice(chunks.next())?;
145        Ok(PubkeyValidityProof { Y, z })
146    }
147}

149#[cfg(test)]
150mod test {
151    use {
152        super::*,
153        crate::{
154            encryption::pod::elgamal::PodElGamalPubkey, sigma_proofs::pod::PodPubkeyValidityProof,

162
163    #[test]
164    fn test_pubkey_proof_correctness() {
165        // random ElGamal keypair
166        let keypair = ElGamalKeypair::new_rand();
167
168        let mut prover_transcript = Transcript::new(b"test");
169        let mut verifier_transcript = Transcript::new(b"test");
170
171        let proof = PubkeyValidityProof::new(&keypair, &mut prover_transcript);
172        proof
173            .verify(keypair.pubkey(), &mut verifier_transcript)
174            .unwrap();
175
176        // derived ElGamal keypair
177        let keypair =
178            ElGamalKeypair::new_from_signer(&Keypair::new(), Pubkey::default().as_ref()).unwrap();
179
180        let mut prover_transcript = Transcript::new(b"test");
181        let mut verifier_transcript = Transcript::new(b"test");
182
183        let proof = PubkeyValidityProof::new(&keypair, &mut prover_transcript);
184        proof
185            .verify(keypair.pubkey(), &mut verifier_transcript)
186            .unwrap();
187
188        assert_eq!(
189            prover_transcript.challenge_scalar(b"test"),
190            verifier_transcript.challenge_scalar(b"test"),
191        )
192    }
193

201        let pod_proof = PodPubkeyValidityProof::from_str(proof_str).unwrap();
202        let proof: PubkeyValidityProof = pod_proof.try_into().unwrap();
203

210    fn test_pubkey_proof_verify_identity() {
211        // An identity ElGamal pubkey
212        let identity_pubkey: ElGamalPubkey = PodElGamalPubkey::zeroed().try_into().unwrap();
213
214        // A dummy proof
