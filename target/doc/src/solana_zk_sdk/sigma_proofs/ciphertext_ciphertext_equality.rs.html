1//! The ciphertext-ciphertext equality sigma proof system.
2//!
3//! The protocol guarantees computational soundness (by the hardness of discrete log) and perfect
4//! zero-knowledge in the random oracle model.

35/// Byte length of a ciphertext-ciphertext equality proof.
36const CIPHERTEXT_CIPHERTEXT_EQUALITY_PROOF_LEN: usize = UNIT_LEN * 7;

38/// The ciphertext-ciphertext equality proof.
39///
40/// Contains all the elliptic curve and scalar components that make up the sigma protocol.
41#[cfg_attr(target_arch = "wasm32", wasm_bindgen)]
42#[allow(non_snake_case)]
43#[derive(Clone)]
44pub struct CiphertextCiphertextEqualityProof {
45    Y_0: CompressedRistretto,
46    Y_1: CompressedRistretto,
47    Y_2: CompressedRistretto,
48    Y_3: CompressedRistretto,
49    z_s: Scalar,
50    z_x: Scalar,
51    z_r: Scalar,
52}
53
54#[allow(non_snake_case)]
55#[cfg(not(target_os = "solana"))]
56impl CiphertextCiphertextEqualityProof {
57    /// Creates a ciphertext-ciphertext equality proof.
58    ///
59    /// The function does *not* hash the public keys, first ciphertext, or second ciphertext into the transcript.
60    /// For security, the caller (the main protocol) should hash these public components prior to
61    /// invoking this constructor.
62    ///
63    /// This function is randomized. It uses `OsRng` internally to generate random scalars.
64    ///
65    /// * `first_keypair` - The ElGamal keypair associated with the first ciphertext to be proved
66    /// * `second_pubkey` - The ElGamal pubkey associated with the second ElGamal ciphertext
67    /// * `first_ciphertext` - The first ElGamal ciphertext for which the prover knows a
68    ///   decryption key for
69    /// * `second_opening` - The opening (randomness) associated with the second ElGamal ciphertext
70    /// * `amount` - The message associated with the ElGamal ciphertext and Pedersen commitment
71    /// * `transcript` - The transcript that does the bookkeeping for the Fiat-Shamir heuristic
72    pub fn new(
73        first_keypair: &ElGamalKeypair,
74        second_pubkey: &ElGamalPubkey,
75        first_ciphertext: &ElGamalCiphertext,
76        second_opening: &PedersenOpening,
77        amount: u64,
78        transcript: &mut Transcript,
79    ) -> Self {
80        transcript.ciphertext_ciphertext_equality_proof_domain_separator();
81
82        // extract the relevant scalar and Ristretto points from the inputs
83        let P_first = first_keypair.pubkey().get_point();
84        let D_first = first_ciphertext.handle.get_point();
85        let P_second = second_pubkey.get_point();
86
87        let s = first_keypair.secret().get_scalar();
88        let mut x = Scalar::from(amount);
89        let r = second_opening.get_scalar();
90
91        // generate random masking factors that also serves as nonces
92        let mut y_s = Scalar::random(&mut OsRng);
93        let mut y_x = Scalar::random(&mut OsRng);
94        let mut y_r = Scalar::random(&mut OsRng);
95
96        let Y_0 = (&y_s * P_first).compress();
97        let Y_1 = RistrettoPoint::multiscalar_mul(vec![&y_x, &y_s], vec![&G, D_first]).compress();
98        let Y_2 = RistrettoPoint::multiscalar_mul(vec![&y_x, &y_r], vec![&G, &(*H)]).compress();
99        let Y_3 = (&y_r * P_second).compress();
100
101        // record masking factors in the transcript
102        transcript.append_point(b"Y_0", &Y_0);
103        transcript.append_point(b"Y_1", &Y_1);
104        transcript.append_point(b"Y_2", &Y_2);
105        transcript.append_point(b"Y_3", &Y_3);
106
107        let c = transcript.challenge_scalar(b"c");
108
109        // compute the masked values
110        let z_s = &(&c * s) + &y_s;
111        let z_x = &(&c * &x) + &y_x;
112        let z_r = &(&c * r) + &y_r;
113
114        // compute challenge `w` for consistency with verification
115        transcript.append_scalar(b"z_s", &z_s);
116        transcript.append_scalar(b"z_x", &z_x);
117        transcript.append_scalar(b"z_r", &z_r);
118        let _w = transcript.challenge_scalar(b"w");
119
120        // zeroize all sensitive non-reference variables
121        x.zeroize();
122        y_s.zeroize();
123        y_x.zeroize();
124        y_r.zeroize();
125
126        CiphertextCiphertextEqualityProof {
127            Y_0,
128            Y_1,
129            Y_2,
130            Y_3,
131            z_s,
132            z_x,
133            z_r,
134        }
135    }
136
137    /// Verifies a ciphertext-ciphertext equality proof.
138    ///
139    /// * `first_pubkey` - The ElGamal pubkey associated with the first ciphertext to be proved
140    /// * `second_pubkey` - The ElGamal pubkey associated with the second ciphertext to be proved
141    /// * `first_ciphertext` - The first ElGamal ciphertext to be proved
142    /// * `second_ciphertext` - The second ElGamal ciphertext to be proved
143    /// * `transcript` - The transcript that does the bookkeeping for the Fiat-Shamir heuristic
144    pub fn verify(
145        self,
146        first_pubkey: &ElGamalPubkey,
147        second_pubkey: &ElGamalPubkey,
148        first_ciphertext: &ElGamalCiphertext,
149        second_ciphertext: &ElGamalCiphertext,
150        transcript: &mut Transcript,
151    ) -> Result<(), EqualityProofVerificationError> {
152        transcript.ciphertext_ciphertext_equality_proof_domain_separator();
153
154        // extract the relevant scalar and Ristretto points from the inputs
155        let P_first = first_pubkey.get_point();
156        let C_first = first_ciphertext.commitment.get_point();
157        let D_first = first_ciphertext.handle.get_point();
158
159        let P_second = second_pubkey.get_point();
160        let C_second = second_ciphertext.commitment.get_point();
161        let D_second = second_ciphertext.handle.get_point();
162
163        // include Y_0, Y_1, Y_2 to transcript and extract challenges
164        transcript.validate_and_append_point(b"Y_0", &self.Y_0)?;
165        transcript.validate_and_append_point(b"Y_1", &self.Y_1)?;
166        transcript.validate_and_append_point(b"Y_2", &self.Y_2)?;
167        transcript.validate_and_append_point(b"Y_3", &self.Y_3)?;
168
169        let c = transcript.challenge_scalar(b"c");
170
171        transcript.append_scalar(b"z_s", &self.z_s);
172        transcript.append_scalar(b"z_x", &self.z_x);
173        transcript.append_scalar(b"z_r", &self.z_r);
174        let w = transcript.challenge_scalar(b"w"); // w used for batch verification
175        let ww = &w * &w;
176        let www = &w * &ww;
177
178        let w_negated = -&w;
179        let ww_negated = -&ww;
180        let www_negated = -&www;
181
182        // check that the required algebraic condition holds
183        let Y_0 = self
184            .Y_0
185            .decompress()
186            .ok_or(SigmaProofVerificationError::Deserialization)?;
187        let Y_1 = self
188            .Y_1
189            .decompress()
190            .ok_or(SigmaProofVerificationError::Deserialization)?;
191        let Y_2 = self
192            .Y_2
193            .decompress()
194            .ok_or(SigmaProofVerificationError::Deserialization)?;
195        let Y_3 = self
196            .Y_3
197            .decompress()
198            .ok_or(SigmaProofVerificationError::Deserialization)?;
199
200        let check = RistrettoPoint::vartime_multiscalar_mul(
201            vec![
202                &self.z_s,            // z_s
203                &(-&c),               // -c
204                &(-&Scalar::ONE),     // -identity
205                &(&w * &self.z_x),    // w * z_x
206                &(&w * &self.z_s),    // w * z_s
207                &(&w_negated * &c),   // -w * c
208                &w_negated,           // -w
209                &(&ww * &self.z_x),   // ww * z_x
210                &(&ww * &self.z_r),   // ww * z_r
211                &(&ww_negated * &c),  // -ww * c
212                &ww_negated,          // -ww
213                &(&www * &self.z_r),  // www * z_r
214                &(&www_negated * &c), // -www * c
215                &www_negated,
216            ],
217            vec![
218                P_first,  // P_first
219                &(*H),    // H
220                &Y_0,     // Y_0
221                &G,       // G
222                D_first,  // D_first
223                C_first,  // C_first
224                &Y_1,     // Y_1
225                &G,       // G
226                &(*H),    // H
227                C_second, // C_second
228                &Y_2,     // Y_2
229                P_second, // P_second
230                D_second, // D_second
231                &Y_3,     // Y_3
232            ],
233        );
234
235        if check.is_identity() {
236            Ok(())
237        } else {
238            Err(SigmaProofVerificationError::AlgebraicRelation.into())
239        }
240    }

242    pub fn to_bytes(&self) -> [u8; CIPHERTEXT_CIPHERTEXT_EQUALITY_PROOF_LEN] {
243        let mut buf = [0_u8; CIPHERTEXT_CIPHERTEXT_EQUALITY_PROOF_LEN];
244        let mut chunks = buf.chunks_mut(UNIT_LEN);
245
246        chunks.next().unwrap().copy_from_slice(self.Y_0.as_bytes());
247        chunks.next().unwrap().copy_from_slice(self.Y_1.as_bytes());
248        chunks.next().unwrap().copy_from_slice(self.Y_2.as_bytes());
249        chunks.next().unwrap().copy_from_slice(self.Y_3.as_bytes());
250        chunks.next().unwrap().copy_from_slice(self.z_s.as_bytes());
251        chunks.next().unwrap().copy_from_slice(self.z_x.as_bytes());
252        chunks.next().unwrap().copy_from_slice(self.z_r.as_bytes());
253
254        buf
255    }
256
257    pub fn from_bytes(bytes: &[u8]) -> Result<Self, EqualityProofVerificationError> {
258        let mut chunks = bytes.chunks(UNIT_LEN);
259
260        let Y_0 = ristretto_point_from_optional_slice(chunks.next())?;
261        let Y_1 = ristretto_point_from_optional_slice(chunks.next())?;
262        let Y_2 = ristretto_point_from_optional_slice(chunks.next())?;
263        let Y_3 = ristretto_point_from_optional_slice(chunks.next())?;
264        let z_s = canonical_scalar_from_optional_slice(chunks.next())?;
265        let z_x = canonical_scalar_from_optional_slice(chunks.next())?;
266        let z_r = canonical_scalar_from_optional_slice(chunks.next())?;
267
268        Ok(CiphertextCiphertextEqualityProof {
269            Y_0,
270            Y_1,
271            Y_2,
272            Y_3,
273            z_s,
274            z_x,
275            z_r,
276        })
277    }

291    #[test]
292    fn test_ciphertext_ciphertext_equality_proof_correctness() {
293        // success case
294        let first_keypair = ElGamalKeypair::new_rand();
295        let second_keypair = ElGamalKeypair::new_rand();
296        let message: u64 = 55;
297
298        let first_ciphertext = first_keypair.pubkey().encrypt(message);
299
300        let second_opening = PedersenOpening::new_rand();
301        let second_ciphertext = second_keypair
302            .pubkey()
303            .encrypt_with(message, &second_opening);
304
305        let mut prover_transcript = Transcript::new(b"Test");
306        let mut verifier_transcript = Transcript::new(b"Test");
307
308        let proof = CiphertextCiphertextEqualityProof::new(
309            &first_keypair,
310            second_keypair.pubkey(),
311            &first_ciphertext,
312            &second_opening,
313            message,
314            &mut prover_transcript,
315        );
316
317        proof
318            .verify(
319                first_keypair.pubkey(),
320                second_keypair.pubkey(),
321                &first_ciphertext,
322                &second_ciphertext,
323                &mut verifier_transcript,
324            )
325            .unwrap();
326
327        // fail case: encrypted and committed messages are different
328        let first_message: u64 = 55;
329        let second_message: u64 = 77;
330
331        let first_ciphertext = first_keypair.pubkey().encrypt(first_message);
332
333        let second_opening = PedersenOpening::new_rand();
334        let second_ciphertext = second_keypair
335            .pubkey()
336            .encrypt_with(second_message, &second_opening);
337
338        let mut prover_transcript = Transcript::new(b"Test");
339        let mut verifier_transcript = Transcript::new(b"Test");
340
341        let proof = CiphertextCiphertextEqualityProof::new(
342            &first_keypair,
343            second_keypair.pubkey(),
344            &first_ciphertext,
345            &second_opening,
346            message,
347            &mut prover_transcript,
348        );
349
350        assert!(proof
351            .verify(
352                first_keypair.pubkey(),
353                second_keypair.pubkey(),
354                &first_ciphertext,
355                &second_ciphertext,
356                &mut verifier_transcript
357            )
358            .is_err());
359
360        assert_eq!(
361            prover_transcript.challenge_scalar(b"test"),
362            verifier_transcript.challenge_scalar(b"test"),
363        )
364    }
