1use crate::CentralContext;
2use crate::constants::METEORA_CONSTANTS;
3use crate::types::meteora_vault::MeteoraVault;
4use solana_sdk::pubkey::Pubkey;
5use std::sync::{Arc, RwLock};
6
7/// Get or create a Meteora vault for a given token address
8///
9/// Checks the cache first, and if not found, derives the vault PDA and creates
10/// a new vault instance. Ensures only one vault instance exists per token by
11/// using the cache's mutex protection.
12pub fn get_meteora_vault_from_token_address(
13  token_address: &Pubkey,
14  central_context: Arc<CentralContext>,
15) -> Arc<RwLock<MeteoraVault>> {
16  // 1) If it’s already in the cache, just clone-and-return the Arc<Mutex<...>>
17  let mut cache = central_context.meteora_vault_cache.lock().unwrap();
18  if let Some(vault_arc) = cache.get(token_address) {
19    return Arc::clone(vault_arc);
20  }
21
22  // 2) Otherwise derive all the PDAs
23  let (vault, _) = Pubkey::find_program_address(
24    &[
25      b"vault",
26      token_address.as_ref(),
27      METEORA_CONSTANTS.vault_base_key.as_ref(),
28    ],
29    &METEORA_CONSTANTS.vault_program,
30  );
31
32  let (vault_token_account, _) = Pubkey::find_program_address(
33    &[b"token_vault", vault.as_ref()],
34    &METEORA_CONSTANTS.vault_program,
35  );
36
37  let (lp_token_address, _) = Pubkey::find_program_address(
38    &[b"lp_mint", vault.as_ref()],
39    &METEORA_CONSTANTS.vault_program,
40  );
41
42  // 3) Build the in-memory struct and wrap in Arc<Mutex<...>>
43  let new_vault = MeteoraVault {
44    vault,
45    vault_token_account,
46    token_address: *token_address,
47    lp_token_address,
48    // Data to be fetched later
49    last_updated_locked_profit: 0,
50    last_report: 0,
51    locked_profit_degradation: 0,
52    total_amount: 0,
53    lp_supply: 0,
54  };
55  let vault_arc = Arc::new(RwLock::new(new_vault));
56
57  // 4) Insert into cache and return
58  cache.insert(*token_address, Arc::clone(&vault_arc));
59
60  vault_arc
61}
