1use crate::types::pools::Pools;
2use crate::types::swap_direction::SwapDirection;
3use solana_sdk::pubkey::Pubkey;
4use solana_sdk::signature::Signature;
5use std::collections::HashSet;
6
7#[derive(Clone, Debug)]
8/**
9Standardized swap transaction type that is used to store all swap transactions on the blockchain
10regardless of the protocol. This schema is used in `solana_car` historical ingestion and by
11strategies and data analytics tools working in polars queries that we made.
12*/
13pub struct SwapTx {
14  pub pool: Pools,
15  pub direction: SwapDirection,
16  pub block_time: u64,
17  pub slot: u64,
18  pub index: u64,
19  pub atomic_instruction_index: u8,
20  /*
21  The fraction of the fee paid in lamports. So if the fee is 0.25%, then that fraction of 0.0025 * 10^9 = 2500000
22  */
23  pub fee_fraction_lp: u64,
24
25  // If swap is from A to B, then this is in terms of A tokens spent in
26  pub swapped_amount_in: u64,
27  // If swap is from A to b, then this is in terms fo B tokens recieved
28  pub swapped_amount_received: u64,
29  // Quantity of token in vault AFTER swap is completed
30  pub pool_token_a_vault_amount: u64,
31  pub pool_token_b_vault_amount: u64,
32  pub price_a_b_lp: u128,
33  pub price_b_a_lp: u128,
34
35  pub token_a_address: Pubkey,
36  pub token_b_address: Pubkey,
37  pub market_address: Pubkey,
38  pub signature: Signature,
39  /*
40  A hash set of all the signers of the tx that encapsulates this swap. A tx on solana can have
41  multiple signers in it.
42  */
43  pub signers: HashSet<Pubkey>,
44}
