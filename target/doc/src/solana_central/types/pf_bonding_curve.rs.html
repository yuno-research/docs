14#[derive(Debug)]
15/// Struct to hold data and PoolTrait implementation for Pumpfun Bonding Curve pools/markets.
16pub struct PfBondingCurve {
17  pub virtual_sol_reserves: u64,
18  pub virtual_token_reserves: u64,
19  pub complete: bool,
20  pub token_address: Pubkey,
21  pub bonding_curve_address: Pubkey,
22  pub bonding_curve_associated_token_account_address: Pubkey,
23  // Pubkey::find_program_address(&[b"creator-vault", creator.as_array()], &PUMP_CONSTANTS.bonding_curve_program,);
24  pub creator_vault_address: Pubkey,
25}
26
27impl PoolTrait for PfBondingCurve {
28  fn pool_address(&self) -> &Pubkey {
29    &self.bonding_curve_address
30  }
31  fn token_a_address(&self) -> &Pubkey {
32    &self.token_address
33  }
34  fn token_b_address(&self) -> &Pubkey {
35    &TOKENS.wsol
36  }
37  fn token_a_vault_address(&self) -> &Pubkey {
38    &self.bonding_curve_associated_token_account_address
39  }
40  fn token_b_vault_address(&self) -> &Pubkey {
41    &self.bonding_curve_address
42  }
43  fn pool_type(&self) -> &Pools {
44    &Pools::PfBondingCurve
45  }
46  /*
47  Never going to call this likely but the fee is 1.25% with 0.95% protocol fee and 0.30% creator fee
48  */
49  fn total_swap_fee_lp(&self, _: &Arc<CentralContext>) -> u64 {
50    12500000
51  }
52
53  fn as_any(&self) -> &dyn Any {
54    self
55  }
56  fn as_any_mut(&mut self) -> &mut dyn Any {
57    self
58  }
59
60  fn price_a_over_b_lp(&self) -> u128 {
61    self.virtual_token_reserves as u128 * LAMPORTS_PER_SOL / self.virtual_sol_reserves as u128
62  }
63
64  fn price_b_over_a_lp(&self) -> u128 {
65    self.virtual_sol_reserves as u128 * LAMPORTS_PER_SOL / self.virtual_token_reserves as u128
66  }
67
68  fn fetch_market_state_from_rpc(&mut self, central_context: &Arc<CentralContext>) {
69    // Fetch data from json rpc before calling updater
70    let data = central_context
71      .json_rpc_client
72      .get_account(&self.bonding_curve_address)
73      .unwrap()
74      .data;
75
76    // Deserialize the account buffer into a BondingCurveAccount
77    let decoded_layout: PfBondingCurveIdl = PfBondingCurveIdl::try_from_slice(&data).unwrap();
78    // Update the bonding curve state
79    self.virtual_sol_reserves = decoded_layout.virtual_sol_reserves;
80    self.virtual_token_reserves = decoded_layout.virtual_token_reserves;
81    self.complete = decoded_layout.complete;
82  }
83  /**
84  This will get real token reserves metric
85  */
86  fn token_a_amount_units(&self) -> u64 {
87    self.virtual_token_reserves - PUMP_CONSTANTS.bc_init_virtual_token_reserve_diff
88  }
89
90  /**
91  This will get real sol reserves metric
92  */
93  fn token_b_amount_units(&self) -> u64 {
94    self.virtual_sol_reserves - PUMP_CONSTANTS.bc_init_virtual_sol_reserves
95  }
96
97  fn directional_fees(&self, _: SwapDirection, __: &Arc<CentralContext>) -> (f64, f64) {
98    (1.0, 1.0)
99  }
