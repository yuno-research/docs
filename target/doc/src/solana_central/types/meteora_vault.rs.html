1use borsh::BorshDeserialize;
2use solana_sdk::pubkey::Pubkey;
3
4/// Struct to hold data for Meteora vaults, which are used in Meteora AMMv1 and potentially Meteora
5/// DLMM pools.
6pub struct MeteoraVault {
7  // shows up as aVault/ A Vault
8  pub vault: Pubkey,
9  // A Token Vault, the actual token account where the token is stored
10  pub vault_token_account: Pubkey,
11  // The address of the token
12  pub token_address: Pubkey,
13  /*
14  The address of the liquidity provider token for this token. Liquidity provider tokens are global
15  across all of Meteora
16  */
17  pub lp_token_address: Pubkey,
18  /*
19  The following fields are information about the vault of this token needed to calculate the
20  actual amount in a liquidity pool. Data is pulled directly off of the vault program through
21  IDL and filled in with token account balances.
22  */
23  pub last_updated_locked_profit: u64,
24  pub last_report: u64,
25  pub locked_profit_degradation: u64,
26  pub total_amount: u64,
27  // The total amount of liquidity provider tokens in circulation
28  pub lp_supply: u64,
29}
