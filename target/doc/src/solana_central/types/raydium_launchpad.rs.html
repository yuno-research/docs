13/**
14Struct to hold data and PoolTrait implementation for Raydium AmmV4 pools. As of July 2025, the only
15global config account for Raydium Launchlab uses the constant product pool. We use do token a is
16base and token b is quote here.
17*/
18pub struct RaydiumLaunchpad {
19  pub info: Pool,
20  pub platform_config: Pubkey,
21  /*
22  PDAs required for swap instructions after the creator fee update.
23  */
24  pub platform_vault: Pubkey,
25  pub creator_vault: Pubkey,
26  // The initial reserves that the token pool was created with
27  pub virtual_token_a_reserve: u64,
28  pub virtual_token_b_reserve: u64,
29  pub real_token_a_reserve: u64,
30  pub real_token_b_reserve: u64,
31}
32
33impl PoolTrait for RaydiumLaunchpad {
34  fn token_a_amount_units(&self) -> u64 {
35    // A is the base token which is being sold as the launchpad progresses
36    self.virtual_token_a_reserve - self.real_token_a_reserve
37  }
38  fn token_b_amount_units(&self) -> u64 {
39    // B is the quote token which fills up as the launchpad progresses
40    self.virtual_token_b_reserve + self.real_token_b_reserve
41  }
42
43  fn pool_address(&self) -> &Pubkey {
44    &self.info.pool_address
45  }
46  fn token_a_address(&self) -> &Pubkey {
47    &self.info.token_a_address
48  }
49  fn token_b_address(&self) -> &Pubkey {
50    &self.info.token_b_address
51  }
52  fn token_a_vault_address(&self) -> &Pubkey {
53    &self.info.token_a_vault_address
54  }
55  fn token_b_vault_address(&self) -> &Pubkey {
56    &self.info.token_b_vault_address
57  }
58  fn pool_type(&self) -> &Pools {
59    &self.info.pool_type
60  }
61
62  /*
63  We never would use this because we'd never do arbitrage on raydium launchpad and we do not take
64  into account for fees or price impact when doing slippage calculation in order to avoid
65  fraudulent platform configs or fee configs that are 100% or something.
66  */
67  fn total_swap_fee_lp(&self, _: &Arc<CentralContext>) -> u64 {
68    panic!("total_swap_fee_lp: Called for Raydium Launchpad, should not be used in prod");
69  }
70
71  fn as_any(&self) -> &dyn Any {
72    self
73  }
74  fn as_any_mut(&mut self) -> &mut dyn Any {
75    self
76  }
77
78  fn price_a_over_b_lp(&self) -> u128 {
79    self.token_a_amount_units() as u128 * LAMPORTS_PER_SOL / self.token_b_amount_units() as u128
80  }
81
82  fn price_b_over_a_lp(&self) -> u128 {
83    self.token_b_amount_units() as u128 * LAMPORTS_PER_SOL / self.token_a_amount_units() as u128
84  }
85
86  fn fetch_market_state_from_rpc(&mut self, central_context: &Arc<CentralContext>) {
87    // The only values that change are real reserves, and those are stored in the pool account.
88    let current_pool_state = LaunchpadPoolIdl::try_from_slice(
89      central_context
90        .json_rpc_client
91        .get_account(&self.info.pool_address)
92        .unwrap()
93        .data
94        .as_slice(),
95    )
96    .unwrap();
97    self.real_token_a_reserve = current_pool_state.real_base;
98    self.real_token_b_reserve = current_pool_state.real_quote;
99  }
100
101  fn directional_fees(&self, _: SwapDirection, __: &Arc<CentralContext>) -> (f64, f64) {
102    (0.0, 0.0)
103  }
104}
