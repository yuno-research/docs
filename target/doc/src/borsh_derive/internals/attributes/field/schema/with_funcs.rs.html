1use std::collections::BTreeMap;
2
3use once_cell::sync::Lazy;
4use syn::meta::ParseNestedMeta;
5
6use crate::internals::attributes::{
7    parsing::parse_lit_into,
8    schema_keys::{DECLARATION, DEFINITIONS},
9    Symbol,
10};
11
12pub enum Variants {
13    Declaration(syn::ExprPath),
14    Definitions(syn::ExprPath),
15}
16
17type ParseFn = dyn Fn(Symbol, Symbol, &ParseNestedMeta) -> syn::Result<Variants> + Send + Sync;
18
19pub static WITH_FUNCS_FIELD_PARSE_MAP: Lazy<BTreeMap<Symbol, Box<ParseFn>>> = Lazy::new(|| {
20    let mut m = BTreeMap::new();
21    // assigning closure `let f = |args| {...};` and boxing closure `let f: Box<ParseFn> = Box::new(f);`
22    // on 2 separate lines doesn't work
23    let f_declaration: Box<ParseFn> = Box::new(|attr_name, meta_item_name, meta| {
24        parse_lit_into::<syn::ExprPath>(attr_name, meta_item_name, meta).map(Variants::Declaration)
25    });
26
27    let f_definitions: Box<ParseFn> = Box::new(|attr_name, meta_item_name, meta| {
28        parse_lit_into::<syn::ExprPath>(attr_name, meta_item_name, meta).map(Variants::Definitions)
29    });
30
31    m.insert(DECLARATION, f_declaration);
32    m.insert(DEFINITIONS, f_definitions);
33    m
34});
35
36#[derive(Clone)]
37pub struct WithFuncs {
38    pub declaration: Option<syn::ExprPath>,
39    pub definitions: Option<syn::ExprPath>,
40}
41
42impl From<BTreeMap<Symbol, Variants>> for WithFuncs {
43    fn from(mut map: BTreeMap<Symbol, Variants>) -> Self {
44        let declaration = map.remove(&DECLARATION);
45        let definitions = map.remove(&DEFINITIONS);
46        let declaration = declaration.map(|variant| match variant {
47            Variants::Declaration(declaration) => declaration,
48            _ => unreachable!("only one enum variant is expected to correspond to given map key"),
49        });
50        let definitions = definitions.map(|variant| match variant {
51            Variants::Definitions(definitions) => definitions,
52            _ => unreachable!("only one enum variant is expected to correspond to given map key"),
53        });
54        Self {
55            declaration,
56            definitions,
57        }
58    }
59}
