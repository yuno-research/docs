5pub fn impl_library_api(ast: &DeriveInput) -> proc_macro2::TokenStream {
6    let name = &ast.ident;
7    let fields = get_fields(ast, "SymBorApi");
8
9    let tok_iter = fields.named.iter().map(field_to_tokens);
10    let q = quote! {
11        impl<'a> SymBorApi<'a> for #name<'a> {
12            unsafe fn load(lib: &'a ::dlopen2::symbor::Library) -> ::std::result::Result<#name<'a>,::dlopen2::Error> {
13                ::std::result::Result::Ok(#name {
14                    #(#tok_iter),*
15                })
16            }
17        }
18    };
19
20    q
21}
22
23fn field_to_tokens(field: &Field) -> proc_macro2::TokenStream {
24    let field_name = &field.ident;
25    let symbol_name = symbol_name(field);
26
27    quote! {
28        #field_name: {
29            let raw_result = lib.ptr_or_null_cstr::<()>(
30                ::std::ffi::CStr::from_bytes_with_nul_unchecked(concat!(#symbol_name, "\0").as_bytes())
31            );
32            ::dlopen2::symbor::FromRawResult::from_raw_result(raw_result)?
33        }
34    }
35}
