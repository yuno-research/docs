18use arrow::array::{Array, StructArray};
19use arrow::compute::utils::combine_validities_and;
20use arrow::datatypes::IdxArr;
21
22pub(super) unsafe fn take_unchecked(array: &StructArray, indices: &IdxArr) -> StructArray {
23    let values: Vec<Box<dyn Array>> = array
24        .values()
25        .iter()
26        .map(|a| super::take_unchecked(a.as_ref(), indices))
27        .collect();
28
29    let validity = array
30        .validity()
31        .map(|b| super::bitmap::take_bitmap_nulls_unchecked(b, indices));
32    let validity = combine_validities_and(validity.as_ref(), indices.validity());
33    StructArray::new(array.dtype().clone(), indices.len(), values, validity)
34}
