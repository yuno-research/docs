86fn init(name: &OsStr) {
87    info!("Loading {name:?}");
88    API.get_or_init(|| {
89        unsafe { Container::load(name) }.unwrap_or_else(|err| {
90            error!("Unable to load {name:?}: {err}");
91            std::process::exit(1);
92        })
93    });
94}

96pub fn locate_perf_libs() -> Option<PathBuf> {
97    let exe = env::current_exe().expect("Unable to get executable path");
98    let perf_libs = exe.parent().unwrap().join("perf-libs");
99    if perf_libs.is_dir() {
100        info!("perf-libs found at {perf_libs:?}");
101        return Some(perf_libs);
102    }
103    warn!("{perf_libs:?} does not exist");
104    None
105}

107fn find_cuda_home(perf_libs_path: &Path) -> Option<PathBuf> {
108    if let Ok(cuda_home) = env::var("CUDA_HOME") {
109        let path = PathBuf::from(cuda_home);
110        if path.is_dir() {
111            info!("Using CUDA_HOME: {path:?}");
112            return Some(path);
113        }
114        warn!("Ignoring CUDA_HOME, not a path: {path:?}");
115    }
116
117    // Search /usr/local for a `cuda-` directory that matches a perf-libs subdirectory
118    for entry in fs::read_dir(perf_libs_path).unwrap().flatten() {
119        let path = entry.path();
120        if !path.is_dir() {
121            continue;
122        }
123        let dir_name = path.file_name().unwrap().to_str().unwrap_or("");
124        if !dir_name.starts_with("cuda-") {
125            continue;
126        }
127
128        let cuda_home: PathBuf = ["/", "usr", "local", dir_name].iter().collect();
129        if !cuda_home.is_dir() {
130            continue;
131        }
132
133        info!("CUDA installation found at {cuda_home:?}");
134        return Some(cuda_home);
135    }
136    None
137}

139pub fn append_to_ld_library_path(mut ld_library_path: String) {
140    if let Ok(env_value) = env::var("LD_LIBRARY_PATH") {
141        ld_library_path.push(':');
142        ld_library_path.push_str(&env_value);
143    }
144    info!("setting ld_library_path to: {ld_library_path:?}");
145    env::set_var("LD_LIBRARY_PATH", ld_library_path);
146}

148pub fn init_cuda() {
149    if let Some(perf_libs_path) = locate_perf_libs() {
150        if let Some(cuda_home) = find_cuda_home(&perf_libs_path) {
151            let cuda_lib64_dir = cuda_home.join("lib64");
152            if cuda_lib64_dir.is_dir() {
153                // Prefix LD_LIBRARY_PATH with $CUDA_HOME/lib64 directory
154                // to ensure the correct CUDA version is used
155                append_to_ld_library_path(cuda_lib64_dir.to_str().unwrap_or("").to_string())
156            } else {
157                warn!("CUDA lib64 directory does not exist: {cuda_lib64_dir:?}");
158            }
159
160            let libcuda_crypt = perf_libs_path
161                .join(cuda_home.file_name().unwrap())
162                .join("libcuda-crypt.so");
163            return init(libcuda_crypt.as_os_str());
164        } else {
165            warn!("CUDA installation not found");
166        }
167    }
168
169    // Last resort!  Blindly load the shared object and hope it all works out
170    init(OsStr::new("libcuda-crypt.so"))
171}

173pub fn api() -> Option<&'static Container<Api<'static>>> {
174    {
175        static INIT_HOOK: Once = Once::new();
176        INIT_HOOK.call_once(|| {
177            if std::env::var("TEST_PERF_LIBS_CUDA").is_ok() {
178                init_cuda();
179            }
180        });
181    }
182
183    API.get()
