42
43impl RandomDefaultHasher {
44    #[inline]
45    pub fn seeded(seed: &[u8; 16]) -> Self {
46        Self(SipHasher13::new_with_key(seed))
47    }
48}

50impl Default for RandomDefaultHasher {
51    #[inline]
52    fn default() -> Self {
53        let mut seed = [0u8; 16];
54        #[cfg(not(feature = "rand"))]
55        {
56            getrandom::fill(&mut seed).expect("Unable to obtain entropy from OS/Hardware sources");
57        }
58        #[cfg(feature = "rand")]
59        {
60            use rand::RngCore;
61            rand::rng().fill_bytes(&mut seed);
62        }
63
64        Self::seeded(&seed)
65    }
66}

158#[derive(Clone, Copy)]
159pub(crate) struct DoubleHasher {
160    h1: u64,
161    h2: u64,
162}
163
164impl DoubleHasher {
165    /// The first two hashes of the value, h1 and h2.
166    ///
167    /// Subsequent hashes, h, are efficiently derived from these two using `next_hash`.
168    ///
169    /// This strategy is adapted from <https://www.eecs.harvard.edu/~michaelm/postscripts/rsa2008.pdf>,
170    /// in which a keyed hash function is used to generate two real hashes, h1 and h2, which are then used to produce
171    /// many more "fake hahes" h, using h = h1 + i * h2.
172    ///
173    /// However, here we only use 1 real hash, for performance, and derive h1 and h2:
174    /// First, we'll think of the 64 bit real hash as two seperate 32 bit hashes, h1 and h2.
175    ///     - Using h = h1 + i * h2 generates entropy in at least the lower 32 bits
176    /// Second, for more entropy in the upper 32 bits, we'll populate the upper 32 bits for both h1 and h2:
177    /// For h1, we'll use the original upper bits 32 of the real hash.
178    ///     - h1 is the same as the real hash
179    /// For h2 we'll use lower 32 bits of h, and multiply by a large constant (same constant as FxHash)
180    ///     - h2 is basically a "weak hash" of h1
181    #[inline]
182    pub(crate) fn new(hash: u64) -> Self {
183        let h2 = hash
184            .wrapping_shr(32)
185            .wrapping_mul(0x51_7c_c1_b7_27_22_0a_95); // 0xffff_ffff_ffff_ffff / 0x517c_c1b7_2722_0a95 = π
186        Self { h1: hash, h2 }
187    }
188
189    /// "Double hashing" produces a new hash efficiently from two orignal hashes.
190    ///
191    /// Modified from <https://www.eecs.harvard.edu/~michaelm/postscripts/rsa2008.pdf>.
192    #[inline]
193    pub(crate) fn next(&mut self) -> u64 {
194        self.h1 = self.h1.wrapping_add(self.h2).rotate_left(5);
195        self.h1
196    }
197}
