12pub(in crate::parse_token) fn parse_interest_bearing_mint_instruction(
13    instruction_data: &[u8],
14    account_indexes: &[u8],
15    account_keys: &AccountKeys,
16) -> Result<ParsedInstructionEnum, ParseInstructionError> {
17    match decode_instruction_type(instruction_data)
18        .map_err(|_| ParseInstructionError::InstructionNotParsable(ParsableProgram::SplToken))?
19    {
20        InterestBearingMintInstruction::Initialize => {
21            check_num_token_accounts(account_indexes, 1)?;
22            let InitializeInstructionData {
23                rate_authority,
24                rate,
25            } = *decode_instruction_data(instruction_data).map_err(|_| {
26                ParseInstructionError::InstructionNotParsable(ParsableProgram::SplToken)
27            })?;
28            let rate_authority: Option<Pubkey> = rate_authority.into();
29            Ok(ParsedInstructionEnum {
30                instruction_type: "initializeInterestBearingConfig".to_string(),
31                info: json!({
32                    "mint": account_keys[account_indexes[0] as usize].to_string(),
33                    "rateAuthority": rate_authority.map(|pubkey| pubkey.to_string()),
34                    "rate": i16::from(rate),
35                }),
36            })
37        }
38        InterestBearingMintInstruction::UpdateRate => {
39            check_num_token_accounts(account_indexes, 2)?;
40            let new_rate: BasisPoints =
41                *decode_instruction_data(instruction_data).map_err(|_| {
42                    ParseInstructionError::InstructionNotParsable(ParsableProgram::SplToken)
43                })?;
44            let mut value = json!({
45                "mint": account_keys[account_indexes[0] as usize].to_string(),
46                "newRate": i16::from(new_rate),
47            });
48            let map = value.as_object_mut().unwrap();
49            parse_signers(
50                map,
51                1,
52                account_keys,
53                account_indexes,
54                "rateAuthority",
55                "multisigRateAuthority",
56            );
57            Ok(ParsedInstructionEnum {
58                instruction_type: "updateInterestBearingConfigRate".to_string(),
59                info: value,
60            })
61        }
62    }
63}
