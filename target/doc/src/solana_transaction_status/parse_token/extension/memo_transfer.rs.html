9pub(in crate::parse_token) fn parse_memo_transfer_instruction(
10    instruction_data: &[u8],
11    account_indexes: &[u8],
12    account_keys: &AccountKeys,
13) -> Result<ParsedInstructionEnum, ParseInstructionError> {
14    check_num_token_accounts(account_indexes, 2)?;
15    let instruction_type_str = match decode_instruction_type(instruction_data)
16        .map_err(|_| ParseInstructionError::InstructionNotParsable(ParsableProgram::SplToken))?
17    {
18        RequiredMemoTransfersInstruction::Enable => "enable",
19        RequiredMemoTransfersInstruction::Disable => "disable",
20    };
21    let mut value = json!({
22        "account": account_keys[account_indexes[0] as usize].to_string(),
23    });
24    let map = value.as_object_mut().unwrap();
25    parse_signers(
26        map,
27        1,
28        account_keys,
29        account_indexes,
30        "owner",
31        "multisigOwner",
32    );
33    Ok(ParsedInstructionEnum {
34        instruction_type: format!("{instruction_type_str}RequiredMemoTransfers"),
35        info: value,
36    })
37}

59            &owner_pubkey,
