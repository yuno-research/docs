24/// Applies to all operations with the echo server.
25pub(crate) const TIMEOUT: Duration = Duration::from_secs(5);

27/// Make a request to the echo server, binding the client socket to the provided IP.
28pub(crate) async fn ip_echo_server_request_with_binding(
29    ip_echo_server_addr: SocketAddr,
30    msg: IpEchoServerMessage,
31    bind_address: IpAddr,
32) -> anyhow::Result<IpEchoServerResponse> {
33    let socket = TcpSocket::new_v4()?;
34    socket.bind(SocketAddr::new(bind_address, 0))?;
35
36    let response =
37        tokio::time::timeout(TIMEOUT, make_request(socket, ip_echo_server_addr, msg)).await??;
38    parse_response(response, ip_echo_server_addr)
39}

41/// Make a request to the echo server, client socket will be bound by the OS.
42pub(crate) async fn ip_echo_server_request(
43    ip_echo_server_addr: SocketAddr,
44    msg: IpEchoServerMessage,
45) -> anyhow::Result<IpEchoServerResponse> {

49    parse_response(response, ip_echo_server_addr)

52/// Makes the request to the specified server and returns reply as Bytes.
53async fn make_request(
54    socket: TcpSocket,
55    ip_echo_server_addr: SocketAddr,
56    msg: IpEchoServerMessage,
57) -> anyhow::Result<BytesMut> {
58    let mut stream = socket.connect(ip_echo_server_addr).await?;
59    // Start with HEADER_LENGTH null bytes to avoid looking like an HTTP GET/POST request
60    let mut bytes = BytesMut::with_capacity(IP_ECHO_SERVER_RESPONSE_LENGTH);
61    bytes.extend_from_slice(&[0u8; HEADER_LENGTH]);
62    bytes.extend_from_slice(&bincode::serialize(&msg)?);
63
64    // End with '\n' to make this request look HTTP-ish and tickle an error response back
65    // from an HTTP server
66    bytes.put_u8(b'\n');
67    stream.write_all(&bytes).await?;
68    stream.flush().await?;
69
70    bytes.clear();
71    let _n = stream.read_buf(&mut bytes).await?;
72    stream.shutdown().await?;
73
74    Ok(bytes)
75}

77fn parse_response(
78    response: BytesMut,
79    ip_echo_server_addr: SocketAddr,
80) -> anyhow::Result<IpEchoServerResponse> {
81    // It's common for users to accidentally confuse the validator's gossip port and JSON
82    // RPC port.  Attempt to detect when this occurs by looking for the standard HTTP
83    // response header and provide the user with a helpful error message
84    if response.len() < HEADER_LENGTH {
85        bail!("Response too short, received {} bytes", response.len());
86    }
87
88    let (response_header, body) =
89        response
90            .split_first_chunk::<HEADER_LENGTH>()
91            .ok_or(anyhow::anyhow!(
92                "Not enough data in the response from {ip_echo_server_addr}!"
93            ))?;
94    let payload = match response_header {
95        [0, 0, 0, 0] => {
96            bincode::deserialize(&response[HEADER_LENGTH..IP_ECHO_SERVER_RESPONSE_LENGTH])?
97        }
98        [b'H', b'T', b'T', b'P'] => {
99            let http_response = std::str::from_utf8(body);
100            match http_response {
101                Ok(r) => bail!(
102                    "Invalid gossip entrypoint. {ip_echo_server_addr} looks to be an HTTP port \
103                     replying with {r}"
104                ),
105                Err(_) => bail!(
106                    "Invalid gossip entrypoint. {ip_echo_server_addr} looks to be an HTTP port."
107                ),
108            }
109        }
110        _ => {
111            bail!(
112                "Invalid gossip entrypoint. {ip_echo_server_addr} provided unexpected header \
113                 bytes {response_header:?} "
114            );
115        }
116    };
117    Ok(payload)
118}

122/// Checks if all of the provided TCP ports are reachable by the machine at
123/// `ip_echo_server_addr`. Tests must complete within timeout provided.
124/// Tests will run concurrently to avoid head-of-line blocking. Will return false on any error.
125/// This function may panic.
126/// All listening sockets should be bound to the same IP.
127pub(crate) async fn verify_all_reachable_tcp(
128    ip_echo_server_addr: SocketAddr,
129    listeners: Vec<TcpListener>,
130    timeout: Duration,
131) -> bool {
132    if listeners.is_empty() {
133        warn!("No ports provided for verify_all_reachable_tcp to check");
134        return true;
135    }
136
137    // Extract the bind address for requests to remote server
138    let bind_address = listeners[0]
139        .local_addr()
140        .expect("Sockets should be bound")
141        .ip();
142
143    // Verify that all other sockets are bound to the same address
144    for listener in listeners.iter() {
145        let local_binding = listener.local_addr().expect("Sockets should be bound");
146        assert_eq!(
147            local_binding.ip(),
148            bind_address,
149            "All sockets should be bound to the same IP"
150        );
151    }
152    let mut checkers = Vec::new();
153    let mut ok = true;
154
155    // Chunk the port range into slices small enough to fit into one packet
156    for chunk in &listeners.into_iter().chunks(MAX_PORT_COUNT_PER_MESSAGE) {
157        let listeners = chunk.collect_vec();
158        let ports = listeners
159            .iter()
160            .map(|l| l.local_addr().expect("Sockets should be bound").port())
161            .collect_vec();
162        info!(
163            "Checking that tcp ports {:?} are reachable from {:?}",
164            &ports, ip_echo_server_addr
165        );
166
167        // make request to the echo server
168        let _ = ip_echo_server_request_with_binding(
169            ip_echo_server_addr,
170            IpEchoServerMessage::new(&ports, &[]),
171            bind_address,
172        )
173        .await
174        .map_err(|err| warn!("ip_echo_server request failed: {err}"));
175
176        // spawn checker to wait for reply
177        // since we do not know if tcp_listeners are nonblocking, we have to run them in native threads.
178        for (port, tcp_listener) in ports.into_iter().zip(listeners) {
179            let listening_addr = tcp_listener.local_addr().unwrap();
180            let (sender, receiver) = oneshot::channel();
181
182            // Use blocking API since we have no idea if sockets given to us are nonblocking or not
183            let thread_handle = tokio::task::spawn_blocking(move || {
184                debug!("Waiting for incoming connection on tcp/{port}");
185                match tcp_listener.incoming().next() {
186                    Some(_) => {
187                        // ignore errors here since this can only happen if a timeout was detected.
188                        // timeout drops the receiver part of the channel resulting in failure to send.
189                        let _ = sender.send(());
190                    }
191                    None => warn!("tcp incoming failed"),
192                }
193            });
194
195            // Set the timeout on the receiver
196            let receiver = tokio::time::timeout(timeout, receiver);
197            checkers.push((listening_addr, thread_handle, receiver));
198        }
199    }
200
201    // now wait for notifications from all the tasks we have spawned.
202    for (listening_addr, thread_handle, receiver) in checkers.drain(..) {
203        match receiver.await {
204            Ok(Ok(_)) => {
205                info!("tcp/{} is reachable", listening_addr.port());
206            }
207            Ok(Err(_v)) => {
208                unreachable!("The receive on oneshot channel should never fail");
209            }
210            Err(_t) => {
211                error!(
212                    "Received no response at tcp/{}, check your port configuration",
213                    listening_addr.port()
214                );
215                // Ugh, std rustc doesn't provide accepting with timeout or restoring original
216                // nonblocking-status of sockets because of lack of getter, only the setter...
217                // So, to close the thread cleanly, just connect from here.
218                // ref: https://github.com/rust-lang/rust/issues/31615
219                TcpStream::connect_timeout(&listening_addr, timeout).unwrap();
220                // Mark that we have found error, but do  not exit yet, as we will have stuck ports otherwise.
221                ok = false;
222            }
223        }
224        thread_handle.await.expect("Thread should exit cleanly");
225    }
226
227    ok
228}

230/// Checks if all of the provided UDP ports on all of the provided IPs are
231/// reachable by the machine at `ip_echo_server_addr`.
232/// This function will test a few ports at a time, retrying if necessary.
233/// Tests must complete within timeout provided, so a longer timeout may be
234/// necessary if checking many ports.
235/// A given amount of retries will be made to accommodate packet loss.
236/// This function may panic.
237///
238pub(crate) async fn verify_all_reachable_udp(
239    ip_echo_server_addr: SocketAddr,
240    sockets: &[&UdpSocket],
241    timeout: Duration,
242    retry_count: usize,
243) -> bool {
244    if sockets.is_empty() {
245        warn!("No ports provided for verify_all_reachable_udp to check");
246        return true;
247    }
248    let mut ip_to_ports: HashMap<IpAddr, BTreeMap<u16, Vec<&UdpSocket>>> = HashMap::new();
249    for &socket in sockets.iter() {
250        let local_addr = socket.local_addr().expect("Socket must be bound");
251        ip_to_ports
252            .entry(local_addr.ip())
253            .or_default()
254            .entry(local_addr.port())
255            .or_default()
256            .push(socket);
257    }
258    for (bind_ip, ports_to_socks_map) in ip_to_ports {
259        let ports: Vec<u16> = ports_to_socks_map.keys().copied().collect();
260
261        info!("Checking that udp ports {ports:?} are reachable from bind IP {bind_ip:?}");
262
263        'outer: for chunk_to_check in ports.chunks(MAX_PORT_COUNT_PER_MESSAGE) {
264            let ports_to_check = chunk_to_check.to_vec();
265
266            for attempt in 0..retry_count {
267                if attempt > 0 {
268                    error!("There are some udp ports with no response!! Retrying...");
269                }
270                // clone off the sockets that use ports within our chunk
271                let sockets_to_check: Vec<UdpSocket> = ports_to_check
272                    .iter()
273                    .flat_map(|port| ports_to_socks_map.get(port).unwrap())
274                    .map(|&s| s.try_clone().expect("Unable to clone UDP socket"))
275                    .collect();
276
277                let _ = ip_echo_server_request_with_binding(
278                    ip_echo_server_addr,
279                    IpEchoServerMessage::new(&[], &ports_to_check),
280                    bind_ip,
281                )
282                .await
283                .map_err(|err| warn!("ip_echo_server request failed: {err}"));
284
285                let reachable_ports = Arc::new(RwLock::new(HashSet::new()));
286                // Spawn threads for each socket to check
287                let mut checkers = JoinSet::new();
288                for socket in sockets_to_check {
289                    let port = socket.local_addr().expect("Socket should be bound").port();
290                    let reachable_ports = reachable_ports.clone();
291
292                    checkers.spawn_blocking(move || {
293                        let start = Instant::now();
294
295                        let original_read_timeout = socket.read_timeout().unwrap();
296                        socket
297                            .set_read_timeout(Some(Duration::from_millis(250)))
298                            .unwrap();
299
300                        loop {
301                            if reachable_ports.read().unwrap().contains(&port)
302                                || Instant::now().duration_since(start) >= timeout
303                            {
304                                break;
305                            }
306
307                            let recv_result = socket.recv(&mut [0; 1]);
308                            debug!("Waited for incoming datagram on udp/{port}: {recv_result:?}");
309
310                            if recv_result.is_ok() {
311                                reachable_ports.write().unwrap().insert(port);
312                                break;
313                            }
314                        }
315
316                        socket.set_read_timeout(original_read_timeout).unwrap();
317                    });
318                }
319                loop {
320                    let next = checkers.join_next().await;
321                    let Some(r) = next else {
322                        break;
323                    };
324                    r.expect("Threads should exit cleanly");
325                }
326                // Might have lost a UDP packet, check that all ports were reached
327                let reachable_ports = Arc::into_inner(reachable_ports)
328                    .expect("Single owner expected")
329                    .into_inner()
330                    .expect("No threads should hold the lock");
331                info!(
332                    "checked udp ports: {ports_to_check:?}, reachable udp ports: \
333                     {reachable_ports:?}"
334                );
335                if reachable_ports.len() == ports_to_check.len() {
336                    continue 'outer; // starts checking next chunk of ports, if any
337                }
338            }
339
340            error!("Maximum retry count reached. Some ports for IP {bind_ip} unreachable.");
341            return false;
342        }
343    }
344    true
