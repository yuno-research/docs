8/// Raw SIMD operations
9/// These operations are used by the SIMD algorithm and have to be implemented for each
10/// data type - SIMD instruction set combination.
11/// The operations are implemented in the `simd_*.rs` files.
12///
13/// Note that for floating point dataypes two implementations are required:
14/// - one for the ignore NaN case (uses a floating point SIMDVecDtype)
15///   (see the `simd_f*_ignore_nan.rs` files)
16/// - one for the return NaN case (uses an integer SIMDVecDtype - as we use the
17///   ord_transform to view the floating point data as ordinal integer data).
18///   (see the `simd_f*_return_nan.rs` files)
19///
20#[doc(hidden)]
21pub trait SIMDOps<ScalarDType, SIMDVecDtype, SIMDMaskDtype, const LANE_SIZE: usize>
22where
23    ScalarDType: Copy + PartialOrd + AsPrimitive<usize>,
24    SIMDVecDtype: Copy,
25    SIMDMaskDtype: Copy,
26{
27    /// Integers > this value **cannot** be accurately represented in SIMDVecDtype
28    const MAX_INDEX: usize;
29    /// Initial index value for the SIMD vector
30    const INITIAL_INDEX: SIMDVecDtype;
31    /// Increment value for the SIMD vector
32    const INDEX_INCREMENT: SIMDVecDtype;
33
34    /// Convert a SIMD register to array
35    unsafe fn _reg_to_arr(reg: SIMDVecDtype) -> [ScalarDType; LANE_SIZE];
36
37    /// Load a SIMD register from memory
38    unsafe fn _mm_loadu(data: *const ScalarDType) -> SIMDVecDtype;
39
40    /// Add two SIMD registers
41    unsafe fn _mm_add(a: SIMDVecDtype, b: SIMDVecDtype) -> SIMDVecDtype;
42
43    /// Compare two SIMD registers for greater-than (gt): a > b
44    /// Returns a SIMD mask
45    unsafe fn _mm_cmpgt(a: SIMDVecDtype, b: SIMDVecDtype) -> SIMDMaskDtype;
46
47    /// Compare two SIMD registers for less-than (lt): a < b
48    unsafe fn _mm_cmplt(a: SIMDVecDtype, b: SIMDVecDtype) -> SIMDMaskDtype;
49
50    /// Blend two SIMD registers using a SIMD mask (selects elements from a or b)
51    unsafe fn _mm_blendv(a: SIMDVecDtype, b: SIMDVecDtype, mask: SIMDMaskDtype) -> SIMDVecDtype;
52
53    /// Horizontal min: get the minimum value from the value SIMD register and its
54    /// corresponding index from the index SIMD register
55    #[inline(always)]
56    unsafe fn _horiz_min(index: SIMDVecDtype, value: SIMDVecDtype) -> (usize, ScalarDType) {
57        // This becomes the bottleneck when using 8-bit data types, as for  every 2**7
58        // or 2**8 elements, the SIMD inner loop is executed (& thus also terminated)
59        // to avoid overflow.
60        // To tackle this bottleneck, we use a different approach for 8-bit data types:
61        // -> we overwrite this method to perform (in SIMD) the horizontal min
62        //    see: https://stackoverflow.com/a/9798369
63        // Note: this is not a bottleneck for 16-bit data types, as the termination of
64        // the SIMD inner loop is 2**8 times less frequent.
65        let index_arr = Self::_reg_to_arr(index);
66        let value_arr = Self::_reg_to_arr(value);
67        let (min_index, min_value) = min_index_value(&index_arr, &value_arr);
68        (min_index.as_(), min_value)
69    }
70
71    /// Horizontal max: get the maximum value from the value SIMD register and its
72    /// corresponding index from the index SIMD register
73    #[inline(always)]
74    unsafe fn _horiz_max(index: SIMDVecDtype, value: SIMDVecDtype) -> (usize, ScalarDType) {
75        // This becomes the bottleneck when using 8-bit data types, as for  every 2**7
76        // or 2**8 elements, the SIMD inner loop is executed (& thus also terminated)
77        // to avoid overflow.
78        // To tackle this bottleneck, we use a different approach for 8-bit data types:
79        // -> we overwrite this method to perform (in SIMD) the horizontal max
80        //    see: https://stackoverflow.com/a/9798369
81        // Note: this is not a bottleneck for 16-bit data types, as the termination of
82        // the SIMD inner loop is 2**8 times less frequent.
83        let index_arr = Self::_reg_to_arr(index);
84        let value_arr = Self::_reg_to_arr(value);
85        let (max_index, max_value) = max_index_value(&index_arr, &value_arr);
86        (max_index.as_(), max_value)
87    }
88
89    /// Get the largest multiple of LANE_SIZE that is <= MAX_INDEX
90    #[inline(always)]
91    fn _get_overflow_lane_size_limit() -> usize {
92        Self::MAX_INDEX - Self::MAX_INDEX % LANE_SIZE
93    }
94
95    /// ----------------- SIMD operations necessary for ignoring NaNs ------------------
96
97    #[inline(always)]
98    unsafe fn _mm_set1(_value: ScalarDType) -> SIMDVecDtype {
99        // This is a dummy method that is only used for the ignore NaN case.
100        // For the Integer and Float return NaN case, this method is not used.
101        unreachable!()
102    }
103}

105/// SIMD initialization operations
106/// These operations are used by the SIMD algorithm and have to be implemented for each
107/// data type - SIMD instruction set combination.
108///
109/// These operations are implemented in the `simd_*.rs` files through calling one of the
110/// three macros below:
111/// - `impl_SIMDInit_Int!`
112///     - called in the `simd_i*.rs` files
113///     - called in the `simd_u*.rs` files
114/// - `impl_SIMDInit_FloatReturnNaN!`
115///     - see the `simd_f*_return_nan.rs` files
116/// - `impl_SIMDInit_FloatIgnoreNaN!`
117///     - see the `simd_f*_ignore_nan.rs` files
118///
119/// The current (default) implementation is for the Int case - see `impl_SIMDInit_Int!`
120/// macro below for more details.
121/// For the Float Return NaN case,only the _return_check method is changed - see
122/// `impl_SIMDInit_FloatReturnNaN!` macro below for more details.
123/// For the Float Ignore NaN case, all the initialization methods are changed - see
124/// `impl_SIMDInit_FloatIgnoreNaN!` macro below for more details. Note that for this
125/// case, the SIMDOps should implement the `_mm_set1` method.
126///
127#[doc(hidden)]
128pub trait SIMDInit<ScalarDType, SIMDVecDtype, SIMDMaskDtype, const LANE_SIZE: usize>:
129    SIMDOps<ScalarDType, SIMDVecDtype, SIMDMaskDtype, LANE_SIZE>
130where
131    ScalarDType: Copy + PartialOrd + AsPrimitive<usize>,
132    SIMDVecDtype: Copy,
133    SIMDMaskDtype: Copy,
134{
135    const IGNORE_NAN: bool = false;
136
137    /// Initialization for _core_argminmax
138
139    #[inline(always)]
140    unsafe fn _initialize_index_values_low(
141        arr_ptr: *const ScalarDType,
142    ) -> (SIMDVecDtype, SIMDVecDtype) {
143        // Initialize the index and value SIMD registers
144        (Self::INITIAL_INDEX, Self::_mm_loadu(arr_ptr))
145    }
146
147    #[inline(always)]
148    unsafe fn _initialize_index_values_high(
149        arr_ptr: *const ScalarDType,
150    ) -> (SIMDVecDtype, SIMDVecDtype) {
151        // Initialize the index and value SIMD registers
152        (Self::INITIAL_INDEX, Self::_mm_loadu(arr_ptr))
153    }
154
155    /// Initialization for _overflow_safe_core_argminmax
156
157    #[inline(always)]
158    fn _initialize_min_value(arr: &[ScalarDType]) -> ScalarDType {
159        unsafe { *arr.get_unchecked(0) }
160    }
161
162    #[inline(always)]
163    fn _initialize_max_value(arr: &[ScalarDType]) -> ScalarDType {
164        unsafe { *arr.get_unchecked(0) }
165    }
166
167    /// Checks
168
169    /// Return case for the algorithm
170    #[inline(always)]
171    fn _return_check(_v: ScalarDType) -> bool {
172        false
173    }
174
175    /// Check if the value is NaN
176    #[inline(always)]
177    fn _nan_check(_v: ScalarDType) -> bool {
178        false
179    }
180}

269                // Initialize the index and value SIMD registers
270                let new_values = Self::_mm_loadu(arr_ptr);

284            }

288                arr_ptr: *const $scalar_dtype,
289            ) -> ($simd_vec_dtype, $simd_vec_dtype) {
290                // Initialize the index and value SIMD registers
291                let new_values = Self::_mm_loadu(arr_ptr);
292                let mask_high =
293                    Self::_mm_cmpgt(new_values, Self::_mm_set1(<$scalar_dtype>::NEG_INFINITY));

320            }

336/// The SIMDCore trait (for all data types).
337/// This trait contains the core of the argminmax algorithm.
338///
339/// This trait is auto-implemented below for all structs - iff the SIMDOps and the
340/// SIMDInit traits are implemented for the struct
341///
342#[doc(hidden)]
343pub trait SIMDCore<ScalarDType, SIMDVecDtype, SIMDMaskDtype, const LANE_SIZE: usize>:
344    SIMDOps<ScalarDType, SIMDVecDtype, SIMDMaskDtype, LANE_SIZE>
345    + SIMDInit<ScalarDType, SIMDVecDtype, SIMDMaskDtype, LANE_SIZE>
346where
347    ScalarDType: Copy + PartialOrd + AsPrimitive<usize>,
348    SIMDVecDtype: Copy,
349    SIMDMaskDtype: Copy,
350{
351    /// Core argminmax algorithm - returns (argmin, min, argmax, max)
352    ///
353    /// This method asserts:
354    /// - the array length is a multiple of LANE_SIZE
355    /// This method assumes:
356    /// - the array length is <= MAX_INDEX
357    ///
358    /// Note that this method is not overflow safe, as it assumes that the array length
359    /// is <= MAX_INDEX. The `_overflow_safe_core_argminmax` method is overflow safe.
360    ///
361    #[inline(always)]
362    unsafe fn _core_argminmax(arr: &[ScalarDType]) -> (usize, ScalarDType, usize, ScalarDType) {
363        assert_eq!(arr.len() % LANE_SIZE, 0);
364        // Efficient calculation of argmin and argmax together
365
366        let mut arr_ptr = arr.as_ptr(); // Array pointer we will increment in the loop
367        let mut new_index = Self::INITIAL_INDEX; // Index we will increment in the loop
368        let (mut index_low, mut values_low) = Self::_initialize_index_values_low(arr_ptr);
369        let (mut index_high, mut values_high) = Self::_initialize_index_values_high(arr_ptr);
370
371        // This is (40%-5%) slower than the loop below (depending on the data type)
372        // arr.chunks_exact(LANE_SIZE)
373        //     .into_iter()
374        //     .skip(1)
375        //     .for_each(|step| {
376        //         new_index = Self::_mm_add(new_index, increment);
377
378        //         let new_values = Self::_mm_loadu(step.as_ptr());
379
380        //         let lt_mask = Self::_mm_cmplt(new_values, values_low);
381        //         let gt_mask = Self::_mm_cmpgt(new_values, values_high);
382
383        //         index_low = Self::_mm_blendv(index_low, new_index, lt_mask);
384        //         index_high = Self::_mm_blendv(index_high, new_index, gt_mask);
385
386        //         values_low = Self::_mm_blendv(values_low, new_values, lt_mask);
387        //         values_high = Self::_mm_blendv(values_high, new_values, gt_mask);
388        //     });
389
390        for _ in 0..arr.len() / LANE_SIZE - 1 {
391            // Increment the index
392            new_index = Self::_mm_add(new_index, Self::INDEX_INCREMENT);
393            // Load the next chunk of data
394            arr_ptr = arr_ptr.add(LANE_SIZE);
395            let new_values = Self::_mm_loadu(arr_ptr);
396
397            // Update the lowest values and index
398            let mask_low = Self::_mm_cmplt(new_values, values_low);
399            values_low = Self::_mm_blendv(values_low, new_values, mask_low);
400            index_low = Self::_mm_blendv(index_low, new_index, mask_low);
401
402            // Update the highest values and index
403            let mask_high = Self::_mm_cmpgt(new_values, values_high);
404            values_high = Self::_mm_blendv(values_high, new_values, mask_high);
405            index_high = Self::_mm_blendv(index_high, new_index, mask_high);
406        }
407
408        // Get the min/max index and corresponding value from the SIMD vectors and return
409        let (min_index, min_value) = Self::_horiz_min(index_low, values_low);
410        let (max_index, max_value) = Self::_horiz_max(index_high, values_high);
411        (min_index, min_value, max_index, max_value)
412    }
413
414    /// Core argmin algorithm - returns (argmin, min)
415    ///
416    /// This method asserts:
417    /// - the array length is a multiple of LANE_SIZE
418    /// This method assumes:
419    /// - the array length is <= MAX_INDEX
420    ///
421    /// Note that this method is not overflow safe, as it assumes that the array length
422    /// is <= MAX_INDEX. The `_overflow_safe_core_argmin` method is overflow safe.
423    ///
424    #[inline(always)]
425    unsafe fn _core_argmin(arr: &[ScalarDType]) -> (usize, ScalarDType) {
426        let mut arr_ptr = arr.as_ptr(); // Array pointer we will increment in the loop
427        let mut new_index = Self::INITIAL_INDEX; // Index we will increment in the loop
428        let (mut index_low, mut values_low) = Self::_initialize_index_values_low(arr_ptr);
429
430        for _ in 0..arr.len() / LANE_SIZE - 1 {
431            // Increment the index
432            new_index = Self::_mm_add(new_index, Self::INDEX_INCREMENT);
433            // Load the next chunk of data
434            arr_ptr = arr_ptr.add(LANE_SIZE);
435            let new_values = Self::_mm_loadu(arr_ptr);
436
437            // Update the lowest values and index
438            let mask_low = Self::_mm_cmplt(new_values, values_low);
439            values_low = Self::_mm_blendv(values_low, new_values, mask_low);
440            index_low = Self::_mm_blendv(index_low, new_index, mask_low);
441        }
442
443        // Get the min index and corresponding value from the SIMD vectors and return
444        Self::_horiz_min(index_low, values_low)
445    }
446
447    /// Core argmax algorithm - returns (argmax, max)
448    ///
449    /// This method asserts:
450    /// - the array length is a multiple of LANE_SIZE
451    /// This method assumes:
452    /// - the array length is <= MAX_INDEX
453    ///
454    /// Note that this method is not overflow safe, as it assumes that the array length
455    /// is <= MAX_INDEX. The `_overflow_safe_core_argmax` method is overflow safe.
456    ///
457    #[inline(always)]
458    unsafe fn _core_argmax(arr: &[ScalarDType]) -> (usize, ScalarDType) {
459        let mut arr_ptr = arr.as_ptr(); // Array pointer we will increment in the loop
460        let mut new_index = Self::INITIAL_INDEX; // Index we will increment in the loop
461        let (mut index_high, mut values_high) = Self::_initialize_index_values_high(arr_ptr);
462
463        for _ in 0..arr.len() / LANE_SIZE - 1 {
464            // Increment the index
465            new_index = Self::_mm_add(new_index, Self::INDEX_INCREMENT);
466            // Load the next chunk of data
467            arr_ptr = arr_ptr.add(LANE_SIZE);
468            let new_values = Self::_mm_loadu(arr_ptr);
469
470            // Update the highest values and index
471            let mask_high = Self::_mm_cmpgt(new_values, values_high);
472            values_high = Self::_mm_blendv(values_high, new_values, mask_high);
473            index_high = Self::_mm_blendv(index_high, new_index, mask_high);
474        }
475
476        // Get the max index and corresponding value from the SIMD vectors and return
477        Self::_horiz_max(index_high, values_high)
478    }

480    /// Overflow-safe core argminmax algorithm - returns (argmin, min, argmax, max)
481    ///
482    /// This method asserts:
483    /// - the array is not empty
484    /// - the array length is a multiple of LANE_SIZE
485    ///
486    #[inline(always)]
487    unsafe fn _overflow_safe_core_argminmax(
488        arr: &[ScalarDType],
489    ) -> (usize, ScalarDType, usize, ScalarDType) {
490        assert!(!arr.is_empty());
491        assert_eq!(arr.len() % LANE_SIZE, 0);
492        // 0. Get the max value of the data type - which needs to be divided by LANE_SIZE
493        let dtype_max = Self::_get_overflow_lane_size_limit();
494
495        // 1. Determine the number of loops needed
496        // let n_loops = (arr.len() + dtype_max - 1) / dtype_max; // ceil division
497        let n_loops = arr.len() / dtype_max; // floor division
498
499        // 2. Perform overflow-safe _core_argminmax
500        let mut min_index: usize = 0;
501        let mut min_value: ScalarDType = Self::_initialize_min_value(arr);
502        let mut max_index: usize = 0;
503        let mut max_value: ScalarDType = Self::_initialize_max_value(arr);
504        let mut start: usize = 0;
505        // 2.0 Perform the full loops
506        for _ in 0..n_loops {
507            if Self::_return_check(min_value) || Self::_return_check(max_value) {
508                // We can return immediately
509                return (min_index, min_value, max_index, max_value);
510            }
511            let (min_index_, min_value_, max_index_, max_value_) =
512                Self::_core_argminmax(&arr[start..start + dtype_max]);
513            if min_value_ < min_value || Self::_return_check(min_value_) {
514                min_index = start + min_index_;
515                min_value = min_value_;
516            }
517            if max_value_ > max_value || Self::_return_check(max_value_) {
518                max_index = start + max_index_;
519                max_value = max_value_;
520            }
521            start += dtype_max;
522        }
523        // 2.1 Handle the remainder
524        if start < arr.len() {
525            if Self::_return_check(min_value) || Self::_return_check(max_value) {
526                // We can return immediately
527                return (min_index, min_value, max_index, max_value);
528            }
529            let (min_index_, min_value_, max_index_, max_value_) =
530                Self::_core_argminmax(&arr[start..]);
531            if min_value_ < min_value || Self::_return_check(min_value_) {
532                min_index = start + min_index_;
533                min_value = min_value_;
534            }
535            if max_value_ > max_value || Self::_return_check(max_value_) {
536                max_index = start + max_index_;
537                max_value = max_value_;
538            }
539        }
540
541        // 3. Return the min/max index and corresponding value
542        (min_index, min_value, max_index, max_value)
543    }

545    /// Overflow-safe core argmin algorithm - returns (argmin, min)
546    ///
547    /// This method asserts:
548    /// - the array is not empty
549    /// - the array length is a multiple of LANE_SIZE
550    ///
551    #[inline(always)]
552    unsafe fn _overflow_safe_core_argmin(arr: &[ScalarDType]) -> (usize, ScalarDType) {
553        assert!(!arr.is_empty());
554        assert_eq!(arr.len() % LANE_SIZE, 0);
555        // 0. Get the max value of the data type - which needs to be divided by LANE_SIZE
556        let dtype_max = Self::_get_overflow_lane_size_limit();
557
558        // 1. Determine the number of loops needed
559        let n_loops = arr.len() / dtype_max; // floor division
560
561        // 2. Perform overflow-safe _core_argminmax
562        let mut min_index: usize = 0;
563        let mut min_value: ScalarDType = Self::_initialize_min_value(arr);
564        let mut start: usize = 0;
565        // 2.0 Perform the full loops
566        for _ in 0..n_loops {
567            if Self::_return_check(min_value) {
568                // We can return immediately
569                return (min_index, min_value);
570            }
571            let (min_index_, min_value_) = Self::_core_argmin(&arr[start..start + dtype_max]);
572            if min_value_ < min_value || Self::_return_check(min_value_) {
573                min_index = start + min_index_;
574                min_value = min_value_;
575            }
576            start += dtype_max;
577        }
578        // 2.1 Handle the remainder
579        if start < arr.len() {
580            if Self::_return_check(min_value) {
581                // We can return immediately
582                return (min_index, min_value);
583            }
584            let (min_index_, min_value_) = Self::_core_argmin(&arr[start..]);
585            if min_value_ < min_value || Self::_return_check(min_value_) {
586                min_index = start + min_index_;
587                min_value = min_value_;
588            }
589        }
590
591        // 3. Return the min/max index and corresponding value
592        (min_index, min_value)
593    }
594
595    /// Overflow-safe core argmax algorithm - returns (argmax, max)
596    ///
597    /// This method asserts:
598    /// - the array is not empty
599    /// - the array length is a multiple of LANE_SIZE
600    ///
601    #[inline(always)]
602    unsafe fn _overflow_safe_core_argmax(arr: &[ScalarDType]) -> (usize, ScalarDType) {
603        assert!(!arr.is_empty());
604        assert_eq!(arr.len() % LANE_SIZE, 0);
605        // 0. Get the max value of the data type - which needs to be divided by LANE_SIZE
606        let dtype_max = Self::_get_overflow_lane_size_limit();
607
608        // 1. Determine the number of loops needed
609        let n_loops = arr.len() / dtype_max; // floor division
610
611        // 2. Perform overflow-safe _core_argminmax
612        let mut max_index: usize = 0;
613        let mut max_value: ScalarDType = Self::_initialize_max_value(arr);
614        let mut start: usize = 0;
615        // 2.0 Perform the full loops
616        for _ in 0..n_loops {
617            if Self::_return_check(max_value) {
618                // We can return immediately
619                return (max_index, max_value);
620            }
621            let (max_index_, max_value_) = Self::_core_argmax(&arr[start..start + dtype_max]);
622            if max_value_ > max_value || Self::_return_check(max_value_) {
623                max_index = start + max_index_;
624                max_value = max_value_;
625            }
626            start += dtype_max;
627        }
628        // 2.1 Handle the remainder
629        if start < arr.len() {
630            if Self::_return_check(max_value) {
631                // We can return immediately
632                return (max_index, max_value);
633            }
634            let (max_index_, max_value_) = Self::_core_argmax(&arr[start..]);
635            if max_value_ > max_value || Self::_return_check(max_value_) {
636                max_index = start + max_index_;
637                max_value = max_value_;
638            }
639        }
640
641        // 3. Return the min/max index and corresponding value
642        (max_index, max_value)
643    }
644}
