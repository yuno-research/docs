3#[inline(always)]
4pub(crate) fn argminmax_generic<T: Copy + PartialOrd>(
5    arr: &[T],
6    lane_size: usize,
7    core_argminmax: unsafe fn(&[T]) -> (usize, T, usize, T),
8    scalar_argminmax: fn(&[T]) -> (usize, usize),
9    nan_check: fn(T) -> bool, // returns true if value is NaN
10    ignore_nan: bool,         // if false, NaNs will be returned
11) -> (usize, usize) {
12    assert!(!arr.is_empty()); // split_array should never return (None, None)
13    match split_array(arr, lane_size) {
14        (Some(simd_arr), Some(rem)) => {
15            // Perform SIMD operation on the first part of the array
16            let simd_result = unsafe { core_argminmax(simd_arr) };
17            // Perform scalar operation on the remainder of the array
18            let (rem_min_index, rem_max_index) = scalar_argminmax(rem);
19            // let (rem_min_index, rem_max_index) = SCALAR::argminmax(rem);
20            let rem_result = (
21                rem_min_index + simd_arr.len(),
22                rem[rem_min_index],
23                rem_max_index + simd_arr.len(),
24                rem[rem_max_index],
25            );
26            // Find the final min and max values
27            let (min_index, min_value) = find_final_index_min(
28                (simd_result.0, simd_result.1),
29                (rem_result.0, rem_result.1),
30                nan_check,
31                ignore_nan,
32            );
33            let (max_index, max_value) = find_final_index_max(
34                (simd_result.2, simd_result.3),
35                (rem_result.2, rem_result.3),
36                nan_check,
37                ignore_nan,
38            );
39            get_correct_argminmax_result(
40                min_index, min_value, max_index, max_value, nan_check, ignore_nan,
41            )
42        }
43        (Some(simd_arr), None) => {
44            let (min_index, min_value, max_index, max_value) = unsafe { core_argminmax(simd_arr) };
45            get_correct_argminmax_result(
46                min_index, min_value, max_index, max_value, nan_check, ignore_nan,
47            )
48        }
49        (None, Some(rem)) => {
50            let (rem_min_index, rem_max_index) = scalar_argminmax(rem);
51            (rem_min_index, rem_max_index)
52        }
53        (None, None) => panic!("Array is empty"), // Should never occur because of assert
54    }
55}

119#[inline(always)]
120fn split_array<T: Copy>(arr: &[T], lane_size: usize) -> (Option<&[T]>, Option<&[T]>) {
121    let n = arr.len();
122
123    // if n < lane_size * 2 {
124    //     // TODO: check if this is the best threshold
125    //     return (None, Some(arr));
126    // };
127
128    let (left_arr, right_arr) = arr.split_at(n - n % lane_size);
129
130    match (left_arr.is_empty(), right_arr.is_empty()) {
131        (true, true) => (None, None),
132        (false, false) => (Some(left_arr), Some(right_arr)),
133        (true, false) => (None, Some(right_arr)),
134        (false, true) => (Some(left_arr), None),
135    }
136}

138/// Get the final index of the min value when both a SIMD and scalar result is available
139/// If not ignoring NaNs (thus returning NaN index if any present):
140/// - If both values are NaN, returns the index of the simd result (as the first part
141/// of the array is passed to the SIMD function)
142/// - If one value is NaN, returns the index of the non-NaN value
143/// - If neither value is NaN, returns the index of the min value
144/// If ignoring NaNs: returns the index of the min value
145///
146/// Note: when the values are equal, the index of the simd result is returned (as the
147/// first part of the array is passed to the SIMD function)
148#[inline(always)]
149fn find_final_index_min<T: Copy + PartialOrd>(
150    simd_result: (usize, T),
151    remainder_result: (usize, T),
152    nan_check: fn(T) -> bool,
153    ignore_nan: bool,
154) -> (usize, T) {
155    let (min_index, min_value) = match simd_result.1.partial_cmp(&remainder_result.1) {
156        Some(Ordering::Less) => simd_result,
157        Some(Ordering::Equal) => simd_result,
158        Some(Ordering::Greater) => remainder_result,
159        None => {
160            if !ignore_nan {
161                // --- Return NaNs
162                // Should prefer simd result over remainder result if both are NaN
163                if nan_check(simd_result.1) {
164                    simd_result
165                } else {
166                    remainder_result
167                }
168            } else {
169                // --- Ignore NaNs
170                // If both are NaN raise panic, else return index of the non-NaN value
171                if nan_check(simd_result.1) && nan_check(remainder_result.1) {
172                    panic!("Data contains only NaNs (or +/- inf)")
173                } else if nan_check(remainder_result.1) {
174                    simd_result
175                } else {
176                    remainder_result
177                }
178            }
179        }
180    };
181    (min_index, min_value)
182}

230/// Get the correct index(es) for the argmin and argmax functions
231/// If not ignoring NaNs (thus returning NaN index if any present):
232/// - If both values are NaN, returns the lowest index twice
233/// - If one value is NaN, returns the index of the non-NaN value twice
234/// - If neither value is NaN, returns the min_index and max_index
235/// If ignoring NaNs: returns the min_index and max_index
236fn get_correct_argminmax_result<T: Copy + PartialOrd>(
237    min_index: usize,
238    min_value: T,
239    max_index: usize,
240    max_value: T,
241    nan_check: fn(T) -> bool,
242    ignore_nan: bool,
243) -> (usize, usize) {
244    if !ignore_nan && (nan_check(min_value) || nan_check(max_value)) {
245        // --- Return NaNs
246        // -> at least one of the values is NaN
247        if nan_check(min_value) && nan_check(max_value) {
248            // If both are NaN, return lowest index
249            let lowest_index = std::cmp::min(min_index, max_index);
250            return (lowest_index, lowest_index);
251        } else if nan_check(min_value) {
252            // If min is the only NaN, return min index
253            return (min_index, min_index);
254        } else {
255            // If max is the only NaN, return max index
256            return (max_index, max_index);
257        }
258    }
259    (min_index, max_index)
260}

265pub(crate) fn min_index_value<T: Copy + PartialOrd>(index: &[T], values: &[T]) -> (T, T) {
266    assert!(!index.is_empty());
267    assert_eq!(index.len(), values.len());
268    let mut min_index: T = unsafe { *index.get_unchecked(0) };
269    let mut min_value: T = unsafe { *values.get_unchecked(0) };
270    for i in 0..values.len() {
271        let v: T = unsafe { *values.get_unchecked(i) };
272        let idx: T = unsafe { *index.get_unchecked(i) };
273        if v < min_value || (v == min_value && idx < min_index) {
274            min_value = v;
275            min_index = idx;
276        }
277    }
278    (min_index, min_value)
279}
