1//! A crate for finding the index of the minimum and maximum values in an array.
2//!
3//! These operations are optimized for speed using [SIMD](https://en.wikipedia.org/wiki/Single_instruction,_multiple_data) instructions (when available).  
4//! The SIMD implementation is branchless, ensuring that there is no best case / worst case.
5//! Furthermore, runtime CPU feature detection is used to choose the fastest implementation for the current CPU (with a scalar fallback).
6//!
7//! The SIMD implementation is enabled for the following architectures:
8//! - `x86` / `x86_64`: [`SSE`](https://en.wikipedia.org/wiki/Streaming_SIMD_Extensions), [`AVX2`](https://en.wikipedia.org/wiki/Advanced_Vector_Extensions#Advanced_Vector_Extensions_2), [`AVX512`](https://en.wikipedia.org/wiki/Advanced_Vector_Extensions#AVX-512)
9//! - `arm` / `aarch64`: [`NEON`](https://en.wikipedia.org/wiki/ARM_architecture#Advanced_SIMD_(Neon))
10//!
11//! # Description
12//!
13//! This crate provides two traits: [`ArgMinMax`](trait.ArgMinMax.html) and [`NaNArgMinMax`](trait.NaNArgMinMax.html).
14//!
15//! These traits are implemented for [`slice`](https://doc.rust-lang.org/std/primitive.slice.html) and [`Vec`](https://doc.rust-lang.org/std/vec/struct.Vec.html).  
16//! - For [`ArgMinMax`](trait.ArgMinMax.html) the supported data types are
17//!   - ints: `i8`, `i16`, `i32`, `i64`
18//!   - uints: `u8`, `u16`, `u32`, `u64`
19//!   - floats: `f16`, `f32`, `f64` (see [Features](#features))
20//! - For [`NaNArgMinMax`](trait.NaNArgMinMax.html) the supported data types are
21//!   - floats: `f16`, `f32`, `f64` (see [Features](#features))
22//!
23//! Both traits differ in how they handle NaNs:
24//! - [`ArgMinMax`](trait.ArgMinMax.html) ignores NaNs and returns the index of the minimum and maximum values in an array.
25//! - [`NaNArgMinMax`](trait.NaNArgMinMax.html) returns the index of the first NaN in an array if there is one, otherwise it returns the index of the minimum and maximum values in an array.
26//!
27//! ### Caution
28//! When dealing with floats and you are sure that there are no NaNs in the array, you should use [`ArgMinMax`](trait.ArgMinMax.html) instead of [`NaNArgMinMax`](trait.NaNArgMinMax.html) for performance reasons. The former is 5%-30% faster than the latter.
29//!
30//!
31//! # Features
32//! This crate has several features.
33//!
34//! - **`nightly_simd`** *(default)* - enables the use of AVX512 & (often) NEON SIMD instructions (requires a nightly compiler).
35//! - **`float`** *(default)* - enables the traits for floats (`f32` and `f64`).
36//! - **`half`** - enables the traits for `f16` (requires the [`half`](https://crates.io/crates/half) crate).
37//! - **`ndarray`** - adds the traits to [`ndarray::ArrayBase`](https://docs.rs/ndarray/latest/ndarray/struct.ArrayBase.html) (requires the `ndarray` crate).
38//! - **`arrow`** - adds the traits to [`arrow::array::PrimitiveArray`](https://docs.rs/arrow/latest/arrow/array/struct.PrimitiveArray.html) (requires the `arrow` crate).
39//! - **`arrow2`** - adds the traits to [`arrow2::array::PrimitiveArray`](https://docs.rs/arrow2/latest/arrow2/array/struct.PrimitiveArray.html) (requires the `arrow2` crate).
40//!
41//!
42//! # Examples
43//!
44//! Two examples are provided below.
45//!
46//! ## Example with integers
47//! ```
48//! use argminmax::ArgMinMax;
49//!
50//! let a: Vec<i32> = vec![0, 1, 2, 3, 4, 5];
51//! let (imin, imax) = a.argminmax();
52//! assert_eq!(imin, 0);
53//! assert_eq!(imax, 5);
54//! ```
55//!
56//! ## Example with NaNs (default `float` feature)
57//! ```ignore
58//! use argminmax::ArgMinMax; // argminmax ignores NaNs
59//! use argminmax::NaNArgMinMax; // nanargminmax returns index of first NaN
60//!
61//! let a: Vec<f32> = vec![f32::NAN, 1.0, f32::NAN, 3.0, 4.0, 5.0];
62//! let (imin, imax) = a.argminmax(); // ArgMinMax::argminmax
63//! assert_eq!(imin, 1);
64//! assert_eq!(imax, 5);
65//! let (imin, imax) = a.nanargminmax(); // NaNArgMinMax::nanargminmax
66//! assert_eq!(imin, 0);
67//! assert_eq!(imax, 0);
68//!```
69//!

128/// Trait for finding the minimum and maximum values in an array. For floats, NaNs are ignored.  
129///
130/// This trait is implemented for slices (or other array-like) of integers and floats.
131///  
132/// See the [feature documentation](index.html#features) for more information on the supported data types and array types.
133///
134pub trait ArgMinMax {
135    // TODO: future work implement these other functions?
136    // fn min(self) -> Self::Item;
137    // fn max(self) -> Self::Item;
138    // fn minmax(self) -> (T, T);
139
140    /// Get the index of the minimum and maximum values in the array.
141    ///
142    /// When dealing with floats, NaNs are ignored.  
143    /// Note that this differs from numpy, where the `argmin` and `argmax` functions
144    /// return the index of the first NaN (which is the behavior of our nanargminmax
145    /// function).
146    ///
147    /// # Returns
148    /// A tuple of the index of the minimum and maximum values in the array
149    /// `(min_index, max_index)`.
150    ///
151    /// # Caution
152    /// When a float array contains *only* NaNs and / or infinities unexpected behavior
153    /// may occur (in which case index 0 is returned for both).
154    ///
155    fn argminmax(&self) -> (usize, usize);
156
157    /// Get the index of the minimum value in the array.
158    ///
159    /// When dealing with floats, NaNs are ignored.
160    /// Note that this differs from numpy, where the `argmin` function returns the index
161    /// of the first NaN (which is the behavior of our nanargmin function).
162    ///
163    /// # Returns
164    /// The index of the minimum value in the array.
165    ///
166    /// # Caution
167    /// When a float array contains *only* NaNs and / or infinities unexpected behavior
168    /// may occur (in which case index 0 is returned).
169    ///
170    fn argmin(&self) -> usize;
171
172    /// Get the index of the maximum value in the array.
173    ///
174    /// When dealing with floats, NaNs are ignored.
175    /// Note that this differs from numpy, where the `argmax` function returns the index
176    /// of the first NaN (which is the behavior of our nanargmax function).
177    ///
178    /// # Returns
179    /// The index of the maximum value in the array.
180    ///
181    /// # Caution
182    /// When a float array contains *only* NaNs and / or infinities unexpected behavior
183    /// may occur (in which case index 0 is returned).
184    ///
185    fn argmax(&self) -> usize;
186}

188/// Trait for finding the minimum and maximum values in an array. For floats, NaNs are propagated - index of the first NaN is returned.  
189///
190/// This trait is implemented for slices (or other array-like) of floats.
191///  
192/// See the [feature documentation](index.html#features) for more information on the supported data types and array types.
193///

195pub trait NaNArgMinMax {
196    /// Get the index of the minimum and maximum values in the array.
197    ///
198    /// When dealing with floats, NaNs are propagated - index of the first NaN is
199    /// returned.  
200    /// Note that this differs from numpy, where the `nanargmin` and `nanargmax`
201    /// functions ignore NaNs (which is the behavior of our argminmax function).
202    ///
203    /// # Returns
204    /// A tuple of the index of the minimum and maximum values in the array
205    /// `(min_index, max_index)`.
206    ///
207    /// # Caution
208    /// When multiple bit-representations for NaNs are used, no guarantee is made
209    /// that the first NaN is returned.
210    ///
211    fn nanargminmax(&self) -> (usize, usize);
212
213    /// Get the index of the minimum value in the array.
214    ///
215    /// When dealing with floats, NaNs are propagated - index of the first NaN is
216    /// returned.
217    /// Note that this differs from numpy, where the `nanargmin` function ignores
218    /// NaNs (which is the behavior of our argmin function).
219    ///
220    /// # Returns
221    /// The index of the minimum value in the array.
222    ///
223    /// # Caution
224    /// When multiple bit-representations for NaNs are used, no guarantee is made
225    /// that the first NaN is returned.
226    ///
227    fn nanargmin(&self) -> usize;
228
229    /// Get the index of the maximum value in the array.
230    ///
231    /// When dealing with floats, NaNs are propagated - index of the first NaN is
232    /// returned.
233    /// Note that this differs from numpy, where the `nanargmax` function ignores
234    /// NaNs (which is the behavior of our argmax function).
235    ///
236    /// # Returns
237    /// The index of the maximum value in the array.
238    ///
239    /// # Caution
240    /// When multiple bit-representations for NaNs are used, no guarantee is made
241    /// that the first NaN is returned.
242    ///
243    fn nanargmax(&self) -> usize;
244}

272macro_rules! impl_argminmax_int {
273    // $int_type is the integer data type of the array (e.g. i32)
274    // you can pass multiple types (separated by commas) to this macro
275    ($($int_type:ty),*) => {
276        $(
277            impl ArgMinMax for &[$int_type] {
278                fn argminmax(&self) -> (usize, usize) {
279                    #[cfg(any(target_arch = "x86", target_arch = "x86_64"))]
280                    {
281                        if is_x86_feature_detected!("sse4.1") & (<$int_type>::NB_BITS == 8) {
282                            // 8-bit numbers are best handled by SSE4.1
283                            return unsafe { SSE::<Int>::argminmax(self) }
284                        }
285                        #[cfg(feature = "nightly_simd")]
286                        {
287                            if is_x86_feature_detected!("avx512bw") & (<$int_type>::NB_BITS <= 16) {
288                                // BW (ByteWord) instructions are needed for 8 or 16-bit avx512
289                                return unsafe { AVX512::<Int>::argminmax(self) }
290                            }
291                            else if is_x86_feature_detected!("avx512f") {  // TODO: check if avx512bw is included in avx512f
292                                return unsafe { AVX512::<Int>::argminmax(self) }
293                            }
294                        }
295                        if is_x86_feature_detected!("avx2") {
296                            return unsafe { AVX2::<Int>::argminmax(self) }
297                        // SKIP SSE4.2 bc scalar is faster or equivalent for 64 bit numbers
298                        // // } else if is_x86_feature_detected!("sse4.2") & (<$int_type>::NB_BITS == 64) & (<$int_type>::IS_FLOAT == false) {
299                        //     // SSE4.2 is needed for comparing 64-bit integers
300                        //     return unsafe { SSE::argminmax(self) }
301                        } else if is_x86_feature_detected!("sse4.1") & (<$int_type>::NB_BITS < 64) {
302                            // Scalar is faster for 64-bit numbers
303                            return unsafe { SSE::<Int>::argminmax(self) }
304                        }
305                    }
306                    #[cfg(target_arch = "aarch64")]
307                    {
308                        if std::arch::is_aarch64_feature_detected!("neon") & (<$int_type>::NB_BITS < 64) {
309                            // Scalar is faster for 64-bit numbers
310                            return unsafe { NEON::<Int>::argminmax(self) }

312                    }
313                    #[cfg(all(target_arch = "arm", feature = "nightly_simd"))]
314                    {

316                            // TODO: requires v7?
317                            // We miss some NEON instructions for 64-bit numbers
